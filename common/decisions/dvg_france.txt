fra_annex_occ = {
	is_shown = {
		exists = c:FRA
		root = c:FRA
		exists = c:OCC
		NOT = { has_variable = fra_dealing_with_occ }
	}

    possible = {
		has_technology_researched = nationalism
	}

    when_taken = {
		add_journal_entry = {
			type = je_fra_annex_occ
		}

		ig:ig_landowners = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_manifest_destiny
				months = normal_modifier_time
				multiplier = -1
			}
		}
		# Add Claims
		s:STATE_LANGUEDOC = { 
			add_claim = ROOT
		}
		s:STATE_RHONE = { 
			add_claim = ROOT
		}
		s:STATE_ARMAGNAC = { 
			add_claim = ROOT
		}
		s:STATE_AUVERGNE = { 
			add_claim = ROOT
		}
		s:STATE_AQUITAINE = { 
			add_claim = ROOT
		}
		set_variable = fra_dealing_with_occ
    }

	ai_chance = {
		value = 100
	}
}

pan_europe = {
	is_shown = {
		exists = c:BRG
		root = c:BRG
		has_variable = embrace_rheinish
		has_variable = embrace_dutch
		has_game_rule = all_formable_nations
	}

	possible = {
		has_technology_researched = pan-nationalism
		OR = {
			has_law = law_type:law_racial_segregation
			has_law = law_type:law_multicultural
			has_law = law_type:law_cultural_exclusion
		}
	}
	
    when_taken = {
		
		add_primary_culture = cu:dutch
		add_primary_culture = cu:rheinish

		create_incident = {
			country = ROOT
			value = 15 
		}

		every_subject_or_below = {
			limit = { 
				is_subject_of = ROOT
				country_is_in_europe = yes
			}
			save_scope_as = subject_to_annex
			ROOT = {
				annex = scope:subject_to_annex
			}
		}
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:rheinish
				is_homeland = cu:dutch
				}
			state_is_in_europe = yes
			}
			add_claim = ROOT
		}
		change_tag = EUR
    }

	ai_chance = {
		value = 100
	}
}
