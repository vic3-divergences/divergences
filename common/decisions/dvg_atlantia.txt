dvg_atlantia_referendum = {
	is_shown = {
		exists = c:ATL
		this = c:ATL
		NOT = { has_global_variable = atlantia_independent }
		c:ATL = {
			is_subject = yes
		}
		overlord = {
			is_player = no
		}
	}

    possible = {
		year >= 1840
		
	}

    when_taken = {
		c:ATL =  {trigger_event = { id = dvg_atlantia.1 popup = yes}}
		set_global_variable = atlantia_independent
		every_country = {
			limit = {
				OR = {
					country_has_primary_culture = cu:lotharian
					country_has_primary_culture = cu:amazonien
				}
				
			}
			trigger_event = { id = dvg_atlantia.11 popup = yes}

			
		}
    }

	ai_chance = {
		value = 100
	}
}

	}

    possible = {
	}

    when_taken = {

		}
    }

	ai_chance = {
		value = 100
	}
}

dvg_nouvellsatian_integration = {
	is_shown = {
		exists = c:NEA
		any_neighbouring_state = {
			owner = c:NEA
		}
		NOT = { c:NEA = {is_subject_of = ROOT
		is_subject_type = subject_type_puppet
		} }
	}

    possible = {
		c:NEA = {
			is_at_war = no
			owner = {
				is_ai = yes
				relations:root >= relations_threshold:amicable
				market = {
					owner = ROOT
				}
				this.gdp < root.gdp
			}
			
		}
		OR = {
			c:NEA = {
				is_in_same_power_bloc = ROOT
			}
			c:NEA = {
				gdp_ownership_ratio = {
					target = ROOT
 					value > 0.50
				}
			}	
			c:NEA = {
				is_subject_of = ROOT
				liberty_desire < 25
			}
		}
	}
		
    when_taken = {
		trigger_event = { id = dvg_atlantia.7 popup = yes }
    }

	ai_chance = {
		value = 100
	}
}
