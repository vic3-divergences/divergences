dvg_restore_tuscany = {
	is_shown = {
		exists = c:PRO
        this = c:PRO
        has_law = law_type:law_monarchy
        NOT = { owns_entire_state_region = STATE_TUSCANY }
		NOT = { has_global_variable = medici_reconquest }
		is_diplomatic_play_committed_participant = no	
	}
	
	
    possible = {
		exists = c:TUS
        c:TUS = {
		
		NOT = { has_law = law_type:law_monarchy }

		}
	}

    when_taken = {
		
        set_global_variable = {
            name = medici_reconquest
            value = 0
        }
        create_diplomatic_play = {
            
        
            target_state = s:STATE_TUSCANY.region_state:TUS
            
            war = yes
            
            type = dp_return_state
            
            add_war_goal = {
                holder = c:TUS
                type = revoke_claim
                target_state = s:STATE_TUSCANY.region_state:TUS
                target_country = c:PRO
            }
            
           
        }
    }
	ai_chance = {
		value = 100
	}

	
}

dvg_reclaim_france = {
	is_shown = {
		
        NOT = { has_global_variable = valois_reconquest }	
		
        NOT = { exists = c:TAP }

		OR = {
		AND = { exists = c:PRO

		c:PRO = ROOT 
		}
		AND = {
		was_formed_from = PRO
		}
		}
	}

    possible = {
		country_rank >= rank_value:major_power
        
        has_law = law_type:law_monarchy
		
	}
		
		

    when_taken = {
		every_state_region = {
			limit = {
			
			is_homeland = cu:french
				
			
			}
			add_claim = ROOT
		}
		every_state_region = {
			limit = {
			
			is_homeland = cu:occitan
				
			
			}
			add_claim = ROOT
		}

        add_journal_entry = {
			type = je_valois_claim
		}
		
		
		create_incident = {
			country = c:PRO
			value = 27
		}
		set_global_variable = {
				name = valois_reconquest
				value = 0
			}
    }

	ai_chance = {
		value = 100
	}
}

			}
			if = {
				limit = {
					has_journal_entry = je_total_revolution
				}
				change_variable = { name = dua_total_revolution_revolt add = 10 }
			}
			
			


		}
		c:DUA = {trigger_event = dvg_dua.119}
    }

	ai_chance = {
		value = 100
	}
