dvg_mcd_acceptgreeks = {
	is_shown = {
		exists = c:YUG
		c:YUG = ROOT
		NOT = {
			c:YUG = {
				country_has_primary_culture = cu:greek
			}
		}
		was_formed_from = MCD
	}

	possible = {
		root = {
			is_at_war = no
		}
	}
	
    when_taken = {
		c:YUG = {
			add_primary_culture = cu:greek
		}
		add_radicals = {
			value = 0.10
			culture = cu:greek
		}
		add_radicals = {
			value = 0.40
			culture = cu:serb
		}
		add_radicals = {
			value = 0.40
			culture = cu:croat
		}
		add_radicals = {
			value = 0.40
			culture = cu:bosniak
		}
    }

	ai_chance = {
		value = 50
	}
}

dvg_rumelian = {
	is_shown = {
		country_is_in_europe = yes
		country_has_state_religion = rel:sunni
		OR = {
			any_primary_culture = {
				has_discrimination_trait_group = heritage_group_european
			}
			any_primary_culture = {
				has_discrimination_trait = heritage_anatolian
			}
		}
		
		any_scope_state = {
			is_capital = yes
			is_incorporated = yes
			state_region = {
				is_homeland = cu:turkish
			}
		}
		NOT = {
			country_has_primary_culture = cu:rumelian
		}
	}

	possible = {
		has_technology_researched = pan-nationalism
	}
	
    when_taken = {
		add_primary_culture = cu:rumelian
		
    }

	ai_chance = {
		value = 50
	}
}
