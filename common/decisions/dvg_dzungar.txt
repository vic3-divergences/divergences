dvg_embrace_han = {
	is_shown = {
		NOT = { country_has_primary_culture = cu:han }
		NOT = { has_journal_entry = je_china_reunification }
		OR = {
		has_interest_marker_in_region = region_south_china
		has_interest_marker_in_region = region_north_china
		}
		ROOT = {
			culture_percent_country = {
			  target = cu:han
			  value >= 0.35
			}
		}
		OR = {
		any_primary_culture = {
							has_discrimination_trait = tradition_sinosphere
						}
		any_primary_culture = {
							has_discrimination_trait_group = heritage_group_east_asian
						}
		any_primary_culture = {
							has_discrimination_trait_group = language_group_mongolic
						}

			}
		any_scope_state = {
						is_incorporated = yes
						state_region = { 
							OR = {
								is_homeland = cu:han
							}
						}
						
					}
	}

    possible = {
		OR = {
		has_law = law_type:law_racial_segregation
		has_law = law_type:law_cultural_exclusion
		}
		OR = {
		  any_primary_culture = {
				has_discrimination_trait = tradition_sinosphere
			}
		   any_primary_culture = {
				has_discrimination_trait = heritage_han
			}
		ROOT = {
			culture_percent_country = {
			  target = cu:han
			  value >= 0.65
			}
		}
		}
	}

    when_taken = {
		
		add_primary_culture = cu:han

		add_journal_entry = {
				type = je_china_reunification_outsider
			}
	
		
    }

	ai_chance = {
		value = 100
	}
}

dvg_dzu_tib_union = {
	is_shown = {
	
		country_has_primary_culture = cu:mongol
		country_has_state_religion = rel:gelugpa
		
		exists = c:TIB

		OR = {
		
		has_law = law_type:law_monarchy
		has_law = law_type:law_theocracy

		}
		
		c:TIB = {has_law = law_type:law_theocracy }

		NOT = {
			has_global_variable = dzu_tib_united
		}
	}

    possible = {

		has_law = law_type:law_state_religion
		has_law = law_type:law_cultural_exclusion
		NOT = {has_journal_entry = je_galdan_khan}
		has_technology_researched = pan-nationalism
		c:TIB = {
			is_at_war = no
			has_law = law_type:law_theocracy
			owner = {
				is_ai = yes
					
				relations:root >= relations_threshold:amicable
					
				this.gdp < root.gdp
			}
		}
	}

    when_taken = {
		trigger_event = { id = dvg_dzungaria.7 popup = yes }
		set_global_variable = {
			name = dzu_tib_united
			value = 0
		}
    }

	ai_chance = {
		value = 100
	}
}

dvg_annex_siberia = {
	is_shown = {
	
		NOT = { is_country_type = decentralized }
		exists = c:SIB

		has_game_rule = siberia_railroaded

		any_primary_culture = {
			has_discrimination_trait_group = heritage_group_north_asian
		}
		any_neighbouring_state = {
			owner = {is_country_type = decentralized}
		}
		has_interest_marker_in_region = sr:region_west_siberia
		
	}

    possible = {

		OR = {
			is_player = yes
			year >= 1900
			is_country_type = recognized
			has_technology_researched = pan-nationalism
		}
		has_technology_researched = colonization
		NOT = { has_law = law_type:law_no_colonial_affairs }
		army_size >= 20
		
	
	}

    when_taken = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					is_country_type = decentralized
					any_primary_culture = {
						has_discrimination_trait_group = heritage_group_north_asian
						
					}
					any_scope_state = {
						region = sr:region_west_siberia
					}

				}			
				root = { annex = prev }
			}
		}
		if = {
			limit = {
				exists = c:SBR
			}
			save_temporary_scope_value_as = {
				name = war_reparations_amount
				value = {
					value = income
					multiply = define:NDiplomacy|DEFAULT_WAR_REPARATIONS_MONEY_TRANSFER
				}
			}
			
			create_diplomatic_play = {
				type = dp_annex_war
				target_country = c:SBR

				add_war_goal = {
					holder = c:SBR
					target_country = ROOT
					type = enforce_treaty_article
					article = {
						article = money_transfer
						source_country = ROOT
						target_country = c:SBR
						inputs = {
							{ quantity = scope:war_reparations_amount }
						}
					}
				}
			}
		}
		
		create_incident = {
			country = ROOT
			value = 5
		}
    }

	ai_chance = {
		value = 100
	}
}

dvg_accept_mizrahi = {
	is_shown = {
		OR = {
			country_is_in_middle_east = yes
			country_is_in_central_asia = yes
		}
		any_scope_state = {
			state_region = {
				is_homeland = cu:mizrahi
			}
		}
		any_scope_pop = {
			culture = cu:mizrahi
			OR = {
				religion = {
					has_discrimination_trait = heritage_islamic
				}
				religion = {
					has_discrimination_trait = heritage_christian
				}
				religion = {
					has_discrimination_trait = heritage_jewish
				}
			}
		}
	}

    possible = {
		cultural_acceptance_base = {
			target = cu:mizrahi
			value > acceptance_status_3
		}
		has_law = law_type:law_cultural_exclusion
	}

    when_taken = {
		add_primary_culture = cu:mizrahi
    }

	ai_chance = {
		value = 100
	}
}
