dvg_new_capital = {
	is_shown = {
		exists = c:PRU
		this = c:PRU
		NOT = {
			has_global_variable = new_capital_pru
		}
		OR = {
			owns_entire_state_region = STATE_BRANDENBURG
			owns_entire_state_region = STATE_LATVIA
		}
		owns_entire_state_region = STATE_EAST_PRUSSIA
	}

    possible = {
		has_technology_researched = railways
		has_technology_researched = modern_sewerage
		gold_reserves >= 300000
		OR = {
			owns_entire_state_region = STATE_BRANDENBURG
			owns_entire_state_region = STATE_LATVIA
		}
	}

    when_taken = {
		add_treasury = -300000
		set_global_variable = new_capital_pru
		trigger_event = { id = dvg_prussia.3 }
    }

	ai_chance = {
		value = 100
	}
}
dvg_scandi_prussia = {
	is_shown = {
		exists = c:PRU
		this = c:PRU
		NOT = {
			has_global_variable = scandi_say_yes
		}
	}
	cooldown = { days = 365 }
	possible = {
		is_subject = no
		AND = {
			AND = {
				gold_reserves >= 400000
				has_attitude = {
					who = c:SCA
					attitude = genial
				}
			}
			OR = {
				c:SCA = {
					taking_loans = yes
				}
				# (scandi je fail) has_global_variable = scandi_je_failed 
				any_country = {
					any_primary_culture = {
						OR = {
							cu:czech ?= this
							cu:north_german ?= this
							cu:south_german ?= this
							cu:rheinish ?= this
							cu:balters ?= this
						}
					}
				global_country_ranking > 5
				}
			}
		}
	}	

	when_taken = {
		c:PRU = {
			trigger_event = { id = dvg_prussia.13 }
		}
	}
}
