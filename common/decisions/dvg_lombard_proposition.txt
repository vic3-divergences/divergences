dvg_lombard_proposition = {
	is_shown = {
	    exists = c:LOM
		this = c:LOM
		OR = {
			exists = c:PAR
			exists = c:MOD
		}
		NOT = {
			c:MOD = {
				is_player = yes
			}
			c:PAR = {
				is_player = yes
			}
		}
	}

    possible = {
		has_technology_researched = nationalism
	}

    when_taken = {
		if = {
			limit = {
				exists = c:PAR
			}
			create_incident = {
				country = c:PAR
				value = 10
			}
			annex = c:PAR
		}
		if = {
			limit = {
				exists = c:MOD
			}
			create_incident = {
				country = c:MOD
				value = 10
			}
			annex = c:MOD
		}
    }

	ai_chance = {
		value = 100
	}
}
