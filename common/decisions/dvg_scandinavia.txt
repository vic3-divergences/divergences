dvg_sca_renovate_cities_of_scotland = {

	is_shown = {
		OR = {
			AND = {
				exists = c:SCA
				c:SCA = ROOT			
			}
			AND = {
				exists = c:SCO
				c:SCO = ROOT			
			}
		}
		ROOT = {
			owns_entire_state_region = STATE_LOWLANDS
			owns_entire_state_region = STATE_HIGHLANDS
		}
		NOT = {
			has_variable = dvg_sca_renovate_cities_of_scotland_decision_taken
		}
	}

	possible = {
		ROOT = {
			owns_entire_state_region = STATE_LOWLANDS
			owns_entire_state_region = STATE_HIGHLANDS
		}
		OR = {
			custom_tooltip = {
				text = dvg_sca_renovate_cities_of_scotland_event_enable_tt
				has_variable = var_sca_highland_clearances_enable_renovation_decision
			}
			AND = {
				has_technology_researched = railways
				has_technology_researched = steel_frame_buildings
				has_technology_researched = nitroglycerin
			}
		}
	}	

	when_taken = {
		set_variable = dvg_sca_renovate_cities_of_scotland_decision_taken
		add_modifier = {
			name = dvg_sca_or_sco_paying_for_scottish_city_renovations
			days = normal_modifier_time
		}
		s:STATE_LOWLANDS.region_state:ROOT = { 
			state_region = {
				add_arable_land = 5
				add_state_trait = state_trait_renovated_scotland_ports
			}
		}
		every_scope_state = {
			limit = {
				has_modifier = dvg_sca_minor_economic_downturn_scotland_clearances
			}
			remove_modifier = dvg_sca_minor_economic_downturn_scotland_clearances
		}
	}
}
