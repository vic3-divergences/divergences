
dvg_annex_taiping = {
	is_shown = {
		exists = c:TPG
		country_has_state_religion = rel:taiping
		OR = {
			c:JIA = { is_player = no }
			NOT = { exists = c:JIA }		
		}
		
	}

    possible = {

		any_country = {
			AND = {
			country_has_state_religion = rel:taiping
			country_has_primary_culture = cu:han
			}
			
		}
	
	}

    when_taken = {
		every_country = {
			if = {
				limit = {
					is_ai = yes			
					country_has_state_religion = rel:taiping
				}			
				root = { annex = prev }
			}
		}
		create_incident = {
			country = ROOT
			value = 4
		}
    }

	ai_chance = {
		value = 100
	}
}

dvg_treaty_ports = {
	is_shown = {
		any_scope_state = {
			is_isolated_from_market = yes
			state_region = {
				is_homeland = cu:han
			}
			is_sea_adjacent = yes
			NOT = {		has_building = building_port}
		}
		
	}

    possible = {

	
	}

    when_taken = {
		every_scope_state = {
			limit = {
				is_isolated_from_market = yes
				state_region = {
				is_homeland = cu:han
				}

			}
			set_state_type = unincorporated
			create_building={
				building="building_port"
				level=1
				reserves=1
				activate_production_methods={ "pm_basic_port" }
			}
			
		}
		
		
    }

	ai_chance = {
		value = 100
	}
}

dvg_china_states = {
	is_shown = {
		
		country_has_primary_culture = cu:han

		NOT = {
			has_variable = china_unifer
		}
	}

    possible = {
		
		is_diplomatic_play_committed_participant = no
		OR = {
			country_rank >= rank_value:unrecognized_major_power 
			country_rank >= rank_value:major_power 
		}
		any_scope_state = {
			count > 2
			state_region = {
				is_homeland = cu:han
			}
		}
		
	}
		
		

    when_taken = {
		every_state_region = {
			limit = {
			is_homeland = cu:han
			state_is_in_china  = yes
			}
			add_claim = ROOT
		}
		
		set_variable = china_unifer

		if = {
			limit = {
				NOT = {
					has_journal_entry = je_china_reunification
				}
			}
			add_journal_entry = {
				type = je_china_reunification
			}
		}
    }

	ai_chance = {
		value = 100
	}
}
