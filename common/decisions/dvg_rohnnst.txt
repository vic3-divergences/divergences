dvg_integrate_new_lands_rohnnst = {

	is_shown = {
		root = c:RHN
		exists = c:RHN
		root = {
			OR = {
				AND = {
					has_state_in_state_region = STATE_WENYUAN
					s:STATE_WENYUAN = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}					
				}
				AND = {
					has_state_in_state_region = STATE_BAIZEHU
					s:STATE_BAIZEHU = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}					
				}
				AND = {
					has_state_in_state_region = STATE_AIOUEZ
					s:STATE_AIOUEZ = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}					
				}
			}
		}
		NOT = {
			has_law = law_type:law_multicultural
		}
	}
	
	possible = {
		root = {
			OR = {
				AND = {
					owns_entire_state_region = STATE_WENYUAN
					s:STATE_WENYUAN = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}
				}
				AND = {
					owns_entire_state_region = STATE_BAIZEHU
					s:STATE_BAIZEHU = {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}
				}
				AND = {
					owns_entire_state_region = STATE_AIOUEZ
					s:STATE_AIOUEZ= {
						NOT = {
							is_homeland = cu:rohnnst
						}
					}
				}
			}
		}	
	}

    when_taken = {
		IF = {
			limit = {
				c:RHN = {
					owns_entire_state_region = STATE_BAIZEHU
					}
				s:STATE_BAIZEHU = {
					NOT = {
						is_homeland = cu:rohnnst
					}
				}
			}
			s:STATE_BAIZEHU = {
				add_homeland = cu:rohnnst
			}
		}
		IF = {
			limit = {
				c:RHN = {
					owns_entire_state_region = STATE_WENYUAN
					}
				s:STATE_WENYUAN = {
					NOT = {
						is_homeland = cu:rohnnst
					}
				}
			}
			s:STATE_WENYUAN = {
				add_homeland = cu:rohnnst
			}
		}
		IF = {
			limit = {
				c:RHN = {
					owns_entire_state_region = STATE_AIOUEZ
					}
				s:STATE_AIOUEZ = {
					NOT = {
						is_homeland = cu:rohnnst
					}
				}
			}
			s:STATE_AIOUEZ = {
				add_homeland = cu:rohnnst
			}
		}
    }
}

dvg_attack_rohnnst = {
	is_shown = {
		exists = c:RHN
		NOT = {
			root = c:RHN
		}
		is_ai = yes
		NOT = {
			has_variable = rhn_attacked
		}
		any_neighbouring_state = {
			owner = c:RHN
		}
	}
    possible = {
		NOT = {
			c:RHN = {
				has_truce_with = ROOT
				is_ai = yes
			}
		}
		has_technology_researched = nationalism
	}

    when_taken = {
		set_variable = rhn_attacked
		create_diplomatic_play = {
			type = dp_annex_war
			target_country = c:RHN
			add_war_goal = {
				holder = c:RHN
				type = war_reparations
			 }
		}
    }

	ai_chance = {
		base = 100
	}
}


dvg_embrace_cree = {
	is_shown = {
		exists = c:RHN
		root = c:RHN

		NOT = {
			country_has_primary_culture = cu:cree
		}
	}
    possible = {
		cultural_acceptance_base = {
			target = cu:cree
			value > acceptance_status_3
		}
		any_scope_state = {
			state_region = {
				is_homeland = cu:cree
			}
		}
	}

    when_taken = {
		add_primary_culture = cu:cree
    }

	ai_chance = {
		base = 100
	}
}
