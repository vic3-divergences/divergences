dvg_decision_renewal_paris = {
	is_shown = {
		NOT = {
			has_global_variable = paris_refounded_global_var
		}
		any_scope_state = {
			state_region = s:STATE_ILE_DE_FRANCE
		}
	}

	possible = {
		has_technology_researched = modern_sewerage
		any_scope_state = {
			state_region = s:STATE_ILE_DE_FRANCE
			is_capital = yes
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 3
			}
			any_scope_building = {
				is_building_type = building_railway
			}
		}
		trigger_if = {
			limit = {
				c:DUA ?= this
			}
		}
	}

	when_taken = {
		set_global_variable = paris_refounded_global_var
		hidden_effect = { # To deal with tooltip wonkiness
			trigger_event = { id = dvg_dua.529 popup = yes }
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_ILE_DE_FRANCE
			}
			STATE_ILE_DE_FRANCE_state_name_assign = yes
		}
		custom_tooltip = choose_one_of_tt
		show_as_tooltip = {
			s:STATE_ILE_DE_FRANCE.region_state:DUA = {
				add_modifier = {
					name = modifier_paris_steam_mills
					days = long_modifier_time
				}
			}
			s:STATE_ILE_DE_FRANCE.region_state:DUA = {
				add_modifier = {
					name = modifier_greater_paris
					days = long_modifier_time
				}
			}
			trigger_event = { id = dvg_dua.529 popup = yes }
		}
	}

	ai_chance = {
		value = 10
	}
}
