integrate_granada = {
	is_shown = {
		OR = {
			AND = {
				exists = c:SPC
				this = c:SPC
			}
			AND = {
				exists = c:IBE
				this = c:IBE
			}
			AND = {
				exists = c:SPA
				this = c:SPA
			}
		}
		exists = c:GRA
		c:GRA = {
			is_subject_of = root
		}
		NOT = { has_global_variable = granada_annexed }
	}
	possible = {
		country_rank = rank_value:great_power
		has_technology_researched = pan-nationalism
		c:GRA = {
			is_subject = yes
			liberty_desire <=20
		}
		
		is_at_war = no
	}
	when_taken = {
		annex_with_incorporation = c:GRA
		add_primary_culture = cu:granadan
		set_global_variable = granada_annexed
	}
}

establish_mauretania = {
	is_shown = {
		country_has_primary_culture = cu:portuguese
		is_in_geographic_region = geographic_region_europe
		any_scope_state = {
			state_region = {
				is_homeland = cu:berber
			}
		}
		
		NOT = { has_global_variable = mauretania_established }
	}
	possible = {
		country_rank = rank_value:great_power
		has_technology_researched = civilizing_mission
		is_at_war = no
		any_scope_state = {
			state_region = {
							is_homeland = cu:berber

			}
		}
	}
	when_taken = {
		random_scope_state = {
			limit = {
				state_region = {
					is_homeland = cu:berber

				}
			}
			save_scope_as = mau_state
		}
		if = {
			limit = { 
				has_game_rule = no_breakaway
			}
			
			create_country = {
				tag = MAU
				origin = this
				state = scope:mau_state
				
				on_created = {
					remove_primary_culture = cu:mauretanian
					add_primary_culture = cu:portuguese
					set_country_type = colonial
					activate_law = law_type:law_presidential_republic
					activate_law = law_type:law_oligarchy
					
					c:SPA = {
						create_diplomatic_pact = {
							country = c:MAU
							type = colony
						}
					}
				}
			}
			every_scope_state = {
				limit = {
					state_is_in_africa = yes
					state_region = {
						is_homeland = cu:berber 
					}
				}
				set_state_owner = c:MAU
				state_region = {
					add_homeland = cu:portuguese
				}
			}
		}
		else_if = {
			limit = { 
				has_game_rule = breakaway
			}
			
			create_country = {
				tag = MAU
				origin = this
				state = scope:mau_state

				
				on_created = {
					activate_law = law_type:law_presidential_republic
					activate_law = law_type:law_oligarchy
					set_country_type = colonial
					every_scope_pop = {
						limit = { 
							culture = { has_discrimination_trait_group = heritage_group_european }
							NOT = { culture = cu:sephardic }
						}
						change_pop_culture = { target = cu:mauretanian value = 1 }
					}
					c:SPA = {
						create_diplomatic_pact = {
							country = c:MAU
							type = colony
						}
					}
				}
			}
			every_scope_state = {
				limit = {
					state_is_in_africa = yes
					state_region = {
						is_homeland = cu:berber 
					}
				}
				set_state_owner = c:MAU
				state_region = {
					add_homeland = cu:mauretanian
				}
			}
			
		}
		
		set_global_variable = mauretania_established
	}
}
