	is_shown = {
		NOT = {has_global_variable = autonomies_status}
		OR = {
			country_has_primary_culture = cu:north_italian
			country_has_primary_culture = cu:south_italian
			country_has_primary_culture = cu:central_italian
			country_has_primary_culture = cu:venetian
			country_has_primary_culture = cu:italian
		}
	}

    possible = {
		owns_entire_state_region = STATE_PIEDMONT
		owns_entire_state_region = STATE_ROMAGNA
		has_technology_researched = pan-nationalism
	}

    when_taken = {
	set_global_variable = {
		name = autonomies_status
	}
	trigger_event = dvg_italian_confederation.6
		
    }
	ai_chance = {
		value = 1
	}
	

}
