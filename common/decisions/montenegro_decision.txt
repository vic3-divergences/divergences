mon_build_biljarda = {
	is_shown = {
		has_dlc_feature = ip3_content
		c:MON ?= ROOT
		NOT = {
			has_variable = biljarda_built
		}
	}

	possible = {
		is_at_war = no
		is_subject = no
		gold_reserves > 20000
	}

	when_taken = {
		trigger_event = {
			id = mon_state_formation.9
			popup = yes
		}
		custom_tooltip = {
			text = mon_state_formation_9_effects_summary
		}
		add_modifier = {
			name = biljarda_modifier
		}
		add_modifier = {
			name = newly_built_biljarda_modifier
			days = very_long_modifier_time
			is_decaying = yes
		}
		set_variable = biljarda_built
	}

	ai_chance = {
		value = 100
	}
}
