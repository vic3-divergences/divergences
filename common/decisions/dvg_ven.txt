dvg_ven_reform_patrician_system_decision = {
	is_shown = {
		exists = c:VEN
		c:VEN = THIS
		has_modifier = dvg_ven_patricians
	}

	possible = {
		AND = {
			ig:ig_landowners = {
				is_powerful = no
				is_in_government = no
			}
			ig:ig_petty_bourgeoisie = {
				is_powerful = no
				is_in_government = no
			}
			NOT = { has_law = law_type:law_oligarchy }
		}
	}

	when_taken = {
		if = {
			limit = {
				has_variable = dvg_ven_patricians_victory_var
			}
			ig:ig_landowners = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_venician_nuovo_patrician
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_venician_nuovo_patrician
			}
			remove_modifier = dvg_ven_patricians
		}
		else = {
			ig:ig_landowners = {
				remove_ideology = ideology_venician_patrician
				add_modifier = {
					name = dvg_ven_patricians_diluted_power
					days = long_modifier_time
				}
			}
			ig:ig_petty_bourgeoisie = {
				remove_ideology = ideology_venician_patrician
				add_ideology = ideology_reactionary
				add_modifier = {
					name = dvg_ven_patricians_diluted_power
					days = long_modifier_time
				}
			}
			remove_modifier = dvg_ven_patricians
		}
	}

	ai_chance = {
		value = 100
	}
}

dvg_italoafrica = {
	is_shown = {
	
		country_is_in_africa = yes

		any_scope_state = {
			state_is_in_africa = yes
			is_capital = yes
		}
		any_primary_culture = {
			has_discrimination_trait = heritage_italic
		}
		
	}

    possible = {

		has_technology_researched = egalitarianism
		OR = {
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_universal_suffrage
		}
		is_subject = no
	
	}

    when_taken = {
		add_primary_culture = cu:afro_latino
		every_scope_pop = {
			limit = {
				religion = rel:catholic
				culture = {
					has_discrimination_trait_group = heritage_group_african
				}
			}
			change_pop_culture = {
				target = cu:afro_latino
				value = 1
			}
		}
    }
	ai_chance = {
		value = 1
	}
	
}
