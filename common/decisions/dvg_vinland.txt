dvg_skraeling_integration = {
	is_shown = {
		country_is_in_north_america = yes
		exists = c:VIN
		exists = c:SKR
		OR = {
			c:VIN = {
				has_law = law_type:law_multicultural
			}
			any_primary_culture = {
				has_discrimination_trait = north_american_group
			}
		}
		NOT = {
			has_variable = skr_to_integrate
		}
	}
    possible = {
		OR = {
			has_technology_researched = pan-nationalism
			any_country = {
				country_is_in_north_america = yes
				NOT = {
					has_law = law_type:law_no_colonial_affairs
				}
			}
			
		}
		c:VIN = {
			NOT = {
				has_truce_with = c:SKR
			}
		}
		c:SKR = {
			OR = {
				AND = {
					is_subject_of = c:VIN
					liberty_desire < 25
				}
				calc_true_if = {
					amount = 3

					has_diplomatic_pact = {
						who = c:VIN
						type = defensive_pact
					}
					has_diplomatic_pact = {
						who = c:VIN
						type = alliance
					}
					has_diplomatic_pact = {
						who = c:VIN
						type = trade_agreement
					}
					has_diplomatic_pact = {
						who = c:VIN
						type = foreign_investment_agreement
					}
					AND = {
						has_variable = recently_lost_war
						is_country_type = unrecognized
					}
				}
				
			}
		}
	}

    when_taken = {
		set_variable = skr_to_integrate
		c:SKR = {
			trigger_event = {
				id = dvg_skraeling.24
			}
		}
		
    }
	ai_chance = {
		base = 100
	}
}

dvg_vinland_treachery = {
	is_shown = {
		exists = c:VIN
		exists = c:SCA
		c:VIN = ROOT
		is_subject_of = c:SCA
		NOT = {
			is_diplomatic_play_enemy_of = c:SCA
		}
		has_journal_entry = je_vinland_independence
	}
	possible = {
		OR = {
			any_diplomatic_play = {
				initiator_is = c:SCA
				target_is = c:SKR
			}
			any_diplomatic_play = {
				initiator_is = c:SCA
				target_is = c:MSG
			}

		}
		
	}
	when_taken = {
		every_diplomatic_play = {
			limit = {
				initiator_is = c:SCA
				target_is = c:SKR
			}
			add_target_backers = {
				c:VIN
			}
			if = {
				limit = {
					c:MSG = {
						liberty_desire > 75
						is_ai = yes
						has_attitude = {
							who = c:VIN
							attitude > disinterested
						}
					}
				}
				add_target_backers = {
					c:MSG
				}
				add_war_goal = {
					holder = c:MSG
					type = independence
					target_country = c:SCA
				}
				
				
			}
		}
		every_diplomatic_play = {
			limit = {
				initiator_is = c:SCA
				target_is = c:MSG
			}
			add_target_backers = {
				c:VIN
			}
			add_war_goal = {
				holder = c:VIN
				type = independence
			}
			if = {
				limit = {
					c:SKR = {
						liberty_desire > 75
						is_ai = yes
						has_attitude = {
							who = c:VIN
							attitude > disinterested
						}
					}
				}
				add_target_backers = {
					c:SKR
				}
				add_war_goal = {
					holder = c:SKR
					type = independence
					target_country = c:SCA
				}
			}
		}
	}
}

    }

		base = 100
	}
}
