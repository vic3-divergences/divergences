mus_southern_expansion = {
	is_shown = {
		exists = c:MUS
		c:MUS = ROOT
	}

	possible = {
		has_technology_researched = nationalism
		NOT = { has_variable = mus_southern_expansion }
	}

    when_taken = {
		add_journal_entry = {
			type = je_mus_southern_expansion
		}

		ig:ig_landowners = {
			add_modifier = {
				name = ig_manifest_destiny
				days = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_manifest_destiny
				days = normal_modifier_time
				multiplier = -1
			}
		}

		# Add Claims
		s:STATE_SLOBOZ = {
			add_claim = ROOT
		}
		s:STATE_DANU = { 
			add_claim = ROOT
		}
		s:STATE_DVG_ROSTOV = { 
			add_claim = ROOT
		}
		set_variable = mus_southern_expansion
    }

	ai_chance = {
		value = 100
	}
}

mus_northern_expansion = {
	is_shown = {
		exists = c:MUS
		c:MUS = ROOT
		exists = c:NOV
	}

    possible = {
		has_technology_researched = nationalism
		NOT = { has_variable = mus_northern_expansion }
		owns_entire_state_region = STATE_DANU
		owns_entire_state_region = STATE_DVG_ROSTOV
	}

    when_taken = {
		add_journal_entry = {
			type = je_mus_northern_expansion
		}
	
		ig:ig_landowners = {
			add_modifier = {
				name = ig_manifest_destiny
				days = normal_modifier_time
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_manifest_destiny
				days = normal_modifier_time
				multiplier = -1
			}
		}

		set_variable = mus_northern_expansion
    }

	ai_chance = {
		value = 100
	}
}
