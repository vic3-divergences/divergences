
dvg_bharat_subcontinent = {
	is_shown = {
		any_primary_culture = {
			has_discrimination_trait_group = heritage_group_south_asian
		}

		any_primary_culture = {
			has_discrimination_trait = language_hindustani
		}
		NOT = {
			has_global_variable = dvg_bharata
		}
	}

    possible = {
		OR = {
			country_tier = empire
			country_tier = hegemony
		}
	}

    when_taken = {

		create_incident = {
			country = ROOT
			value = 20
		}
		
		every_state_region = {
			limit = {
				region = sr:region_indus
				region = sr:region_ganges
			}
			add_claim = ROOT
		}
		
		set_global_variable = dvg_bharata

}
ai_chance = {
		value = 100
	}
}

dvg_sikh_embrace = {
	is_shown = {
	
		has_variable = dvg_sikh_accepted
		cu:panjabi = {
			root = {
				cultural_acceptance_base = { target = prev value >= acceptance_status_5 }
			}
		}
		NOT = {
		   has_variable = dvg_convert_sikh
		   exists = c:PAN
		}
	}

    possible = {
		
		NOT = {
			has_modifier = decision_cooldown
			ruler = {
				has_trait = bigoted
			}
		}
		any_interest_group = {
			leader = {
				religion = rel:sikh
			}
			is_powerful = yes
		}
		religion_percent_country = {
			target = rel:sikh
			value > 0.25
		}

		has_law = law_type:law_state_religion
	}

    when_taken = {

		trigger_event = {
			id = dvg_afghanistan.4
		}
		set_variable = dvg_convert_sikh
	}
	ai_chance = {
		value = 100
	}
}

dvg_tamil_embrace = {
	is_shown = {
	
		culture_percent_country = {
			target = cu:tamil
			value > 0.05
		}
		NOT = {
			country_has_primary_culture = cu:tamil
		}
		OR = {
			country_is_in_india = yes
			country_is_in_southeast_asia = yes
			is_country_type = colonial
			
		}
		scope:state_religion_scope = {
			has_discrimination_trait_group = heritage_group_eastern
		}
	}

    possible = {

		OR = {
			culture_percent_country = {
				target = cu:tamil
				value > 0.25
			}
			has_law = law_type:law_multicultural
		}
		
		any_scope_state = {
			state_region = {
				is_homeland = cu:tamil
			}
		}
		has_technology_researched = egalitarianism
		cultural_acceptance_base = {
			target = cu:tamil
			value > acceptance_status_3
		}
	}

    when_taken = {

		add_primary_culture = cu:tamil
	}
	ai_chance = {
		base = 100
	}
}

dvg_become_hindustan = {
	is_shown = {
		any_primary_culture = {
			has_discrimination_trait = language_hindustani
		}
		NOT = {
			exists = c:MUG
			exists = c:HND
		}
	}

    possible = {

		has_technology_researched = nationalism

		NOT = {
			exists = c:TME
		}
		calc_true_if = {
			amount > 3
			country_or_subject_owns_entire_state_region = STATE_DELHI
			country_or_subject_owns_entire_state_region = STATE_AWADH
			country_or_subject_owns_entire_state_region = STATE_BIHAR
			country_or_subject_owns_entire_state_region = STATE_JHARKHAND
			country_or_subject_owns_entire_state_region = STATE_NORTH_BENGAL
			country_or_subject_owns_entire_state_region = STATE_SOUTH_BENGAL
			country_or_subject_owns_entire_state_region = STATE_SINDH
			country_or_subject_owns_entire_state_region = STATE_RAJPUTANA
		}
	}
		
		
	

    when_taken = {

		change_tag = HND

		if = {
			limit = {
				NOT = {
					country_has_primary_culture = cu:language_hindustani
				}
			}
			add_primary_culture = cu:language_hindustani
		}

		add_primary_culture = cu:language_hindustani

		every_subject_or_below = {
			limit = {
				any_primary_culture = {
				has_discrimination_trait = language_hindustani
				}
				NOT = {
					country_has_primary_culture = cu:language_hindustani
				}
			}
			add_primary_culture = cu:language_hindustani
		}

	}
	ai_chance = {
		base = 100
	}
}
