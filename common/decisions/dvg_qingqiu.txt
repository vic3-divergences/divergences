dvg_integrate_east_qq = {

	is_shown = {
		root = c:QIN
		exists = c:QIN
		root = {
			OR = {
				AND = {
					owns_entire_state_region = STATE_JINSHAN
					s:STATE_JINSHAN = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
				AND = {
					owns_entire_state_region = STATE_YICHENG
					s:STATE_YICHENG = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
				OR = {
					AND = {
						has_state_in_state_region = STATE_COMANCHERIE
						s:STATE_COMANCHERIE = {
							NOT = {
								is_homeland = cu:qingqiuan
							}
						}
					}
					AND = {
						has_state_in_state_region = STATE_TERRAS_VERMELHAS
						s:STATE_TERRAS_VERMELHAS = {
							NOT = {
								is_homeland = cu:qingqiuan
							}
						}
					}
				}
				any_scope_state = {
					state_is_in_north_america = yes 
					culture_percent_state = {
						target = cu:qingqiuan
						value >= 0.33
					}
					state_region = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
			}
		}
		NOT = {
			has_law = law_type:law_multicultural
		}
	}

    when_taken = {
		IF = {
			limit = {
				c:QIN = {
					owns_entire_state_region = STATE_JINSHAN
					}
				s:STATE_JINSHAN = {
					NOT = {
						is_homeland = cu:qingqiuan
					}
				}
			}
			s:STATE_JINSHAN = {
				add_homeland = cu:qingqiuan
			}
		}
		IF = {
			limit = {
				c:QIN = {
					owns_entire_state_region = STATE_YICHENG
					}
				s:STATE_YICHENG = {
					NOT = {
						is_homeland = cu:qingqiuan
					}
				}
			}
			s:STATE_YICHENG = {
				add_homeland = cu:qingqiuan
			}
		}
		IF = {
			limit = {
				c:QIN = {
					has_state_in_state_region = STATE_COMANCHERIE
					s:STATE_COMANCHERIE = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
			}
			s:STATE_COMANCHERIE = {
				add_homeland = cu:qingqiuan
			}
		}
		IF = {
			limit = {
				c:QIN = {
					has_state_in_state_region = STATE_TERRAS_VERMELHAS
					s:STATE_TERRAS_VERMELHAS = {
						NOT = {
							is_homeland = cu:qingqiuan
						}
					}
				}
			}
			s:STATE_TERRAS_VERMELHAS = {
				add_homeland = cu:qingqiuan
			}
		}
		every_scope_state = {

dvg_embrace_rhonnst = {

	is_shown = {
		NOT = {
			exists = c:RHN
			has_variable = dvg_qq_embraced_rohnnst_var
		}
		any_scope_state = {
			state_region = {
				is_homeland = cu:rohnnst
			}
		}
	}
	
	possible = {
		country_is_in_north_america = yes
		OR = {
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural 
		}
		any_scope_state = {
			state_region = {
				is_homeland = cu:rohnnst
			}
			is_incorporated = yes
		}
	}
	
	when_taken = {
		add_primary_culture = cu:rohnnst
		set_variable = dvg_qq_embraced_rohnnst_var
    }
	ai_chance = {
		base = 100
	}
}


dvg_attack_blf = {
	is_shown = {
		exists = c:BLF
		NOT = {
			root = c:BLF
		}
		is_ai = yes
		NOT = {
			has_variable = blf_attacked
		}
		any_neighbouring_state = {
			owner = c:BLF
		}
	}

    possible = {
		NOT = {
			c:BLF = {
				has_truce_with = ROOT
				is_ai = yes
			}
		}
		has_technology_researched = civilizing_mission
	}

    when_taken = {
		set_variable = blf_attacked
		create_diplomatic_play = {
			type = dp_annex_war
			target_country = c:BLF
			add_war_goal = {
				holder = c:BLF
				type = war_reparations
			 }
		}
    }

	ai_chance = {
		base = 100
	}
}
dvg_manifest_destiny_qin = {
	is_shown = {
		exists = c:QIN
		root = c:QIN
		NOT = {
			has_variable = manifested_destiny_qingqiu
		}
		
	}

    possible = {
		has_technology_researched = nationalism
		has_technology_researched = romanticism
		country_rank >= rank_value:minor_power
		is_subject = no
		is_at_war = no 
		NOT = {
			has_law = law_type:law_no_colonial_affairs
		}
	}

    when_taken = {
		set_variable = manifested_destiny_qingqiu
		create_incident = {
			country = THIS
			value = 8
		}
		every_state = {
			limit = {
				OR = {
					state_region = {
						is_homeland = cu:navajo
					}
					state_region = {
						is_homeland = cu:paiute
					}
					state_region = {
						is_homeland = cu:nez_perce
					}
					state_region = {
						is_homeland = cu:athabaskan
					}
					state_region = {
						is_homeland = cu:kaiganese
					}
				}
				
				NOT = {
					owner = ROOT
				}
				state_is_in_north_america = yes
			}
			state_region = {
					
				add_claim = ROOT	
			}	
		}
		
    }

	ai_chance = {
		base = 100
	}
}
