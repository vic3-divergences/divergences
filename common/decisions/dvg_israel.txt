
dvg_jewish_state = {
	is_shown = {
		country_has_state_religion = rel:jewish
		any_primary_culture = {
			has_discrimination_trait = tradition_israelite
		}
		NOT = {
		has_global_variable = israel_global_var
		}
	}

    possible = {
		any_scope_state = {
			state_region = s:STATE_PALESTINE
		}
		has_technology_researched = nationalism
	}

    when_taken = {
		set_global_variable = israel_global_var
		trigger_event = dvg_israel.2
		if = {
			limit = {
				NOT = {
					any_scope_state = {
					state_region = s:STATE_PALESTINE
					is_capital = yes
					}
				}
			}
			set_capital = STATE_PALESTINE
		}
		s:STATE_PALESTINE.region_state:ISR = {
			add_modifier = {
				name = dvg_moderate_immigration_policy
				months = 60
				is_decaying = yes
			}
		}
		create_incident = {
			country = ROOT
			value = 20
		}
		if = {
			limit = {
				NOT = {
					exists = c:ISR
				}
			}
			change_tag = ISR
		}	
	}
ai_chance = {
		value = 100
	}
}
