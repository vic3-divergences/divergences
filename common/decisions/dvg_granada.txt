dvg_rocafuerte_afro_latino_citizens = {
	is_shown = {
		exists = c:GRA
		this = c:GRA
		ruler = { has_variable = is_rocafuerte }
		has_law = law_type:law_slavery_banned
		is_subject = no
	}
	possible = {
		has_technology_researched = egalitarianism
		has_law = law_type:law_universal_suffrage
		OR = {
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural
		}
	}
	when_taken = {
		add_primary_culture = cu:afro_latino
		every_scope_pop = {
			limit = {
				religion = rel:catholic
				culture = {
					has_discrimination_trait_group = heritage_group_african
				}
			}
			change_pop_culture = {
				target = cu:afro_latino
				value = 1
			}
		}
		add_loyalists = {
			culture = cu:afro_latino
			value = 0.6
		}
	}
}
dvg_de_blanco_revenge = {
	is_shown = {
		exists = c:GRA
		this = c:GRA
		exists = c:QUI
		exists = c:TWT
		ruler = { has_variable = is_de_blanco }
		NOR = { 
			has_variable = de_blanco_revenge
			has_variable = glory_to_de_blanco
		}
	}
	possible = {
		custom_tooltip = {
			text = qui_twt_war
			c:QUI = { has_war_with = c:TWT }
		}
	}
	when_taken = {
		set_variable = de_blanco_revenge
		if = {
			limit = {
				c:QUI = {
					NOT = { has_variable = dvg_QUI_ballivian }
				}
			}
			trigger_event = { id = dvg_granada.29 }
		}
		else = {
			limit = {
				c:QUI = { has_variable = dvg_QUI_ballivian }
			}
			trigger_event = { id = dvg_granada.36 }
		}
	}
	ai_chance = {
		value = 100
	}
}
