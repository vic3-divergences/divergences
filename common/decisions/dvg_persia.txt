

dvg_embrace_persian = {
	is_shown = {
		NOT = {country_has_primary_culture = cu:persian }
		OR = {
		any_scope_culture = {
			OR = {
				has_discrimination_trait_group = heritage_group_middle_eastern
				has_discrimination_trait_group = heritage_group_central_asian
			}
			is_primary_culture_of  = ROOT
		}
		AND = {
		exists = c:AFG
		c:AFG = ROOT 
		}
		AND = {
		exists = c:AZB
		c:AZB = ROOT 
		ruler = {has_trait = ambitious}
		}

		}
	}

    possible = {
		
		has_technology_researched = pan-nationalism
		
		country_rank > rank_value:insignificant_power
		
		has_law = law_type:law_cultural_exclusion

		has_law = law_type:law_monarchy
		
		

		
	}

    when_taken = {
		 
		add_primary_culture = cu:persian
    }

	ai_chance = {
		value = 100
	}
}

dvg_armenian_liberation = {
	is_shown = {
		
	country_has_primary_culture = cu:armenian
		
	NOT = { has_global_variable = armenia_claims }	
	}
    possible = {
		has_technology_researched = nationalism
		country_rank > rank_value:insignificant_power
	}
    when_taken = {
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:armenian
				}
			}
			add_claim = ROOT
		}

			country = ROOT
			value = 15
		}
		set_global_variable = {
				name = armenia_claims
				value = 0
		}
    }

	ai_chance = {
		value = 100
	}
}

demand_qeshm = {
	is_shown = {
		country_rank >= rank_value:major_power
		NOT = { has_global_variable = qeshm_demanded }
		OR = {
			AND = {
				exists = c:SPA
				this = c:SPA
			}
			AND = {
				exists = c:IBE
				this = c:IBE
			}
			AND = {
				exists = c:SPC
				this = c:SPC
			}
		} 
		is_diplomatic_play_committed_participant = no
	}

    possible = {
		country_rank >= rank_value:major_power
		has_interest_marker_in_region = region_persia
		has_interest_marker_in_region = region_arabic
		navy_size >= 20
		NOT = { has_truce_with = c:SHZ 
		owns_entire_state_region = STATE_LARISTAN
		
		}
		c:SHZ = {
				owns_entire_state_region = STATE_LARISTAN
			}
	}

    when_taken = {
		c:SHZ = {
			trigger_event = { id = dvg_persia.7 days = 10 }
		}
		set_variable = qeshm_demander
		set_global_variable = qeshm_demanded
    }

	ai_chance = {
		value = 100
	}
}
