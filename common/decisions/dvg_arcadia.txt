dvg_afroarcadian_liberation = {
	is_shown = {
		
		exists = c:LIB
		c:LIB = ROOT

		NOT = {
			has_global_variable = afroarcadian_liberation
		}
	}

    possible = {
		has_technology_researched = pan-nationalism
		
		NOT = {
			ruler = {
				has_ideology = ideology:ideology_pacifist
			}
		}
		has_law = law_type:law_racial_segregation

	}
		
		

    when_taken = {
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:afro_arcadian
				is_homeland = cu:afro_latino
				is_homeland = cu:afro_germanic
				}
			state_is_in_north_america = yes
			}
			add_claim = ROOT
		}
		
		
		add_primary_culture = cu:afro_latino
		
		create_incident = {
			country = ROOT
			value = 20
		}
		set_global_variable = {
				name = afroarcadian_liberation
				value = 0
			}
    }

	ai_chance = {
		base = 100
	}
}

dvg_isolation_failsafe = {
	is_shown = {
		
		is_subject = yes
		overlord = {
			is_ai = yes
		}
		is_ai = yes
	}

    possible = {
		
		any_scope_state = {
			is_capital = yes
			is_isolated_from_market = yes
		}
		is_diplomatic_play_committed_participant = no
		
	}
		
		

    when_taken = {
		
		make_independent = yes

		if = {
			limit = {
				has_law = law_type:law_closed_borders
			}
			activate_law = law_type:law_migration_controls
		}

		if = {
			limit = {
				has_law = law_type:law_isolationism
			}
			activate_law = law_type:law_protectionism
		}
		add_modifier = {
			name = dvg_approval_buff
			months = 30
			is_decaying = yes
		}
    }

	ai_chance = {
		base = 100
	}
}
