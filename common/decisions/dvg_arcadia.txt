dvg_afroarcadian_liberation = {
	is_shown = {
		
		exists = c:LIB
		c:LIB = ROOT

		NOT = {
			has_global_variable = afroarcadian_liberation
		}
	}

    possible = {
		has_technology_researched = pan-nationalism
		
		NOT = {
			ruler = {
				has_ideology = ideology:ideology_pacifist
			}
		}
		has_law = law_type:law_racial_segregation

	}
		
		

    when_taken = {
		every_state_region = {
			limit = {
			OR = {
				is_homeland = cu:afro_arcadian
				is_homeland = cu:afro_latino
				is_homeland = cu:afro_germanic
				}
			state_is_in_north_america = yes
			}
			add_claim = ROOT
		}
		
		
		add_primary_culture = cu:afro_latino
		
		create_incident = {
			country = ROOT
			value = 20
		}
		every_country = {
			limit = {
				any_scope_state = {
					state_region = {
						is_homeland = cu:afro_arcadian
					}
				}
			}
			add_radicals = {
				value = 0.40
				culture = cu:afro_arcadian
			}
		}
		set_global_variable = {
				name = afroarcadian_liberation
				value = 0
			}
    }

	ai_chance = {
		base = 100
	}
}

dvg_manifest_destiny_cadien = {
	is_shown = {
		
		country_has_primary_culture = cu:cadien

		NOT = {
			has_variable = manifested_destiny_plantagenia
		}
	}
    possible = {
		has_technology_researched = nationalism
		has_technology_researched = romanticism
		country_rank >= rank_value:minor_power
		is_subject = no
		is_at_war = no 
		NOT = {
			has_law = law_type:law_no_colonial_affairs
		}
	}
    when_taken = {
		trigger_event = {
			id = dvg_plantagenia.10
		}
		every_state = {
			limit = {
				
				state_region = {
					
				is_homeland = cu:cadien
					
				}
				NOT = {
					owner = ROOT
				}
				state_is_in_north_america = yes
			}
			state_region = {
					
				add_claim = ROOT
					
			}
			
			
		}
		s:STATE_BAHIA = {
			add_claim = ROOT
		}
		s:STATE_APACHERIA = {
			add_claim = ROOT
		}
		s:STATE_COMANCHERIE = {
			add_claim = ROOT
		}
		s:STATE_TERRAS_VERMELHAS = {
			add_claim = ROOT
		}
		s:STATE_TEJAS = {
			add_claim = ROOT
		}
		if = {
			limit = {
				NOT = {
					has_journal_entry = je_manifest_destiny_pla
				}
			}
			add_journal_entry = {
				type = je_manifest_destiny_pla
			}
		}
		if = {
			limit = {
				OR = {
					ruler = {
						has_ideology = ideology:ideology_jingoist_leader
					}
					any_interest_group = {
						OR = {
							has_ideology = ideology:ideology_jingoist
							leader = {
								has_ideology = ideology:ideology_jingoist_leader
							}
						}
						is_powerful = yes
					}

				}
				
			}
			s:STATE_BAJA_CALIFORNIA = {
				add_claim = ROOT
			}
			s:STATE_XINGTIAN = {
				add_claim = ROOT
			}
		}
		create_incident = {
			country = THIS
			value = 15
		}
		set_variable = manifested_destiny_plantagenia
    }

	ai_chance = {
		base = 100
	}
}

dvg_buy_athesia = {
	is_shown = {
		country_is_in_north_america = yes
		OR = {
			has_interest_marker_in_region = sr:region_caribbean
			has_interest_marker_in_region = sr:region_dixie
		}

		NOT = {
			owns_entire_state_region = STATE_ATHESIA
			ROOT = {is_subject = yes}
		}
		AND = {
			exists = c:ATH
			c:ATH = {
				is_ai = yes
				is_subject_of = c:VEN
			}
		}
		
	}

    possible = {

		gold_reserves > 100000
		c:VEN = {
			global_country_ranking < rank_value:major_power 
			relations:root >= relations_threshold:amicable
			OR = {
				has_attitude = {
					who = ROOT
					attitude = genial
				}
				has_attitude = {
					who = ROOT
					attitude = loyal
				}

			}
		}
		NOT = {
			has_law = law_type:law_no_colonial_affairs
		}
	}		

    when_taken = {
		
		c:VEN = {
			add_modifier = {
				name = dvg_falklands_sale
				months = 12
			}
		}

		add_modifier = {
			name = dvg_falklands_buy
			months = 12
			
		}
		
		annex = c:ATH
	 }

	ai_chance = {
		base = 100
	}
}
