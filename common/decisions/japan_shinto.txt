
  shinto_decision = {
	is_shown = {
		exists = c:JAP
		c:JAP = THIS
		NOT = { country_has_state_religion = rel:shinto }
	}
	possible = {
		#custom_tooltip = {
		#	text = is_meiji_japan
		#	has_variable = meiji_restoration_complete
		#}
		ig:ig_devout = {
			is_marginal = no
		}
		has_law = law_type:law_monarchy
		NOT = { has_law = law_type:law_state_religion }
	}

	when_taken = {
		set_state_religion = rel:shinto
		ig:ig_devout = {
			hidden_effect = { set_interest_group_name = ig_shinto_monks }
			remove_ideology = ideology_buddhist_moralist
			add_ideology = ideology_shinto_moralist
		}
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:shinto
				value = 0.1
			}
		}
	}

	ai_chance = {
		value = 10
	}
}

 shingon_decision = {
	is_shown = {
		country_has_primary_culture = cu:japanese
		NOT = { country_has_state_religion = rel:gelugpa }
	}
	possible = {
		#custom_tooltip = {
		#	text = is_meiji_japan
		#	has_variable = meiji_restoration_complete
		#}
		ig:ig_devout = {
			is_powerful = yes
		}
		NOT = { has_law = law_type:law_state_religion }
		religion_percent_country = {
			target = rel:gelugpa
			value > 0.40
		}
		
	}

	when_taken = {
		set_state_religion = rel:gelugpa

		activate_law = law_type:law_state_religion
		
		every_country = {
			limit = {
				country_has_state_religion = rel:gelugpa
			}
			change_relations = {
				country = THIS
				value = 20
			}
		}

		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:gelugpa
				value = 0.1
			}
		}
		add_loyalists = {
			value = 0.15
			religion = rel:gelugpa
		}
	}

	ai_chance = {
		value = 10
	}
}
