de_aigle_progressbar = {
	name = "de_aigle_progress_bar"
	desc = "de_aigle_progress_bar_desc"

	default_bad = yes

	monthly_progress = {
		if = {
			limit = {
				average_sol_for_culture = {
					target = cu:lotharian
					value < 10
				} 
			}
			add = {
				desc = "earning_recognition_low_sol_tt"
				value = 1
			}
		}
		if = {
			limit = {
				# Your Standard of Living is relatively high
				sol_ranking <= 20
			}
			add = {
				desc = "AVG_STANDARD_OF_LIVING"
				value = 1
			}
		}
		if = {
			limit = {
				has_modifier = declared_bankruptcy
			}
			add = {
				desc = "austrian_neo_absolutism_declared_bankruptcy"
				value = 10
			}
		}
		if = {
			limit = {
				taking_loans = yes
			}
			add = {
				desc = "austrian_neo_absolutism_taking_loans"
				value = 1
			}
		}
		if = {
			limit = {
				any_scope_state = {
					turmoil >= 0.10
				}
			}
			add = {
				desc = "austrian_neo_absolutism_radicals"
				every_scope_state = {
					limit = {
						turmoil >= 0.10
					}
					add = this.turmoil
				}
			}
		}
		if = {
				limit = {
					ig:ig_armed_forces = {
						is_powerful = yes
						is_in_government = yes
					}
					any_interest_group = {
					count > 1
					ig_approval < -6
					}
				}
				add = {
				desc = army_disapproval_tt
				value = 1
				}
		}
		if = {
				limit = {
					has_law = law_type:law_autocracy
				}
			add = {
				desc = autocracy_enacted_tt
				value = 1
			}
		}
		if = {
			limit = {
					cu:atlantian = {
						culture_current_fervor >= 50
					}
					any_scope_state = {
						state_region = {
							is_homeland = cu:atlantian
						}
					}
			}
			add = {
				desc = atlantian_frevor_tt
				value = 1
			}
		}
		if = {
			limit = {
					cu:afro_amerigan = {
						culture_current_fervor >= 50
					}
			}
			add = {
				desc = afro_amerigan_fervor_tt
				value = 1
			}
		}
		if = {
			limit = {
				cu:guarani = {
						culture_current_fervor >= 50
				}
				any_scope_state = {
						state_region = {
							is_homeland = cu:guarani
						}
				}
			}
			add = {
				desc = guarani_fervor_tt
				value = 1
			}
		}
		if = {
			limit = {
				scope:de_aigle_progressbar > 100
			}
			add = {
				value = 100
			}
		}
	}

	
 	start_value = 50
 	min_value = 0
 	max_value = 100
 }
