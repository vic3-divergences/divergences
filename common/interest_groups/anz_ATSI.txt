			}
			if = {
				limit = {
					owner = {
						country_has_state_religion = rel:protestant

				}
				random_list = {
					20 = {}

						change_character_religion = rel:protestant

						}

	pop_weight = {
		value = 0


		if = {
			limit = {
				has_pop_culture = aborigine
			}
			add = { 
				desc = "POP_Aboriginal"
				value = 225
			}
		}
		else_if = {
			limit = {
				has_pop_culture = aborigine_mixed
			}
			add = { 
				desc = "POP_Aboriginal_mixed"
				value = 125
			}
		}
		else_if = {
			limit = {
				has_pop_culture = torres_strait_islander #Still has common interest, but less of a unifying cultural connection
			}

				value = 225
			}
		}

			limit = {
				owner = {
					ig:ig_atsi = {
						leader = {
							not = {
								has_religion = protestant
								has_ideology = ideology:ideology_traditionalist
							}
						}
					}
				}
			}
			add = {
				desc = "POP_atsi_religion"
				if = {
					limit = {
						has_state_religion = yes
					}
					add = -40
				}
			}
		}

		#more likely when discriminated
		add = {
			desc = "POP_ATSI_ACCEPTANCE"
			if = {
				limit = {
					pop_acceptance < 20
				}
				add = 65
			}
			else_if = {
				limit = {
					pop_acceptance < 40
				}
				add = 40
			}
			else_if = {
				limit = {
					pop_acceptance < 60
				}
				add = 20
			}
			else_if = { 
				limit = {
					pop_acceptance >= 80
				}
				add = -55
			}
		}

		#less relevant when equal rights
		add = {
			desc = "POP_ATSI_EQUAL_RIGHTS"
			if = {
				limit = {
					owner = {
						has_law_or_variant = law_type:law_aboriginal_equal
					}
				}
				add = -115
			}
		}

		#rally together when colonising
		if = { 
			limit = {
				or = {
					has_pop_culture = aborigine
					has_pop_culture = torres_strait_islander
				}
			}
			if = {
				limit = {
					owner = {
						any_scope_state = {
							state_region = {
								is_homeland = cu:aborigine
							}
							is_under_colonization = yes
						}
					}
				}
				add = {
					desc = "COLONISATION_INST"
					value = owner.institution:institution_colonial_affairs.investment
					multiply = 15
				}
			}
		}
		else_if = {
			limit = { #excluding others because cultures are fragmented and might need more performance than worth?
				has_pop_culture = maori 
			}
			if = {
				limit = {
					owner = {
						any_scope_state = {
							state_region = {
								is_homeland = cu:maori
							}
							is_under_colonization = yes
						}
					}
				}
				add = {
					desc = "COLONISATION_INST"
					value = owner.institution:institution_colonial_affairs.investment
					multiply = 15
				}
			}
		}

		#very high attraction while oppressed
		#lose support when getting less oppressed
		# new land reform law? aboriginal poptypes would be helpful here

		#######
		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader ?= {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.15
					min = 0.85
				}
			}
		}

		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}

		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
			}
			multiply = { 
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				divide = 3
				add = 1
			}
		}

		if = {
			limit = {
				NOR = {
					has_pop_culture = aborigine
					has_pop_culture = aborigine_mixed
					has_pop_culture = torres_strait_islander
				}
			}
			multiply = {
				desc = "POP_not_atsi"
				value = 0.333
			}
		}
	}
