# Dnieper Region

STATE_KIEV_state_name_assign = {
	if = {
        limit = {
            owner = {
                any_primary_culture = {
                    cu:russian ?= this
                }
				NOT = {
					any_primary_culture = {
						cu:ukrainian ?= this
					}
				}
            }
        }
		
        set_state_name = STATE_KIEV_russian
        
        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_KIEV_city_russian
        }
		
        set_hub_name = {
            type = farm
            name = HUB_NAME_STATE_KIEV_farm_russian
        }
		
        set_hub_name = {
            type = wood
            name = HUB_NAME_STATE_KIEV_wood_russian
        }
    }
	else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
		reset_state_name = yes
		reset_hub_names = yes
	}
}

				has_law = law_type:law_council_republic
			}
		}

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_INGRIA_city_russian_sov # St. Petersburg only becomes Leningrad if Lenin has been around to be a namesake
        }
	}
	else_if = {
		limit = {
			owner = {
				country_has_primary_culture = cu:russian
				has_technology_researched = nationalism
				NOT = { has_law = law_type:law_council_republic }

				OR = {
					any_rival_country = {
						any_primary_culture = {
							has_discrimination_trait = german_speaking
						}
					}
					any_rivaling_country = {
						any_primary_culture = {
							has_discrimination_trait = german_speaking
						}					
					}
				}
			}
		}

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_INGRIA_city_russian_imp # St. Petersburg only becomes Petrograd when Germanophobia becomes a thing
        }
	}
	else = { # switch back to defaults
		reset_state_name = yes
		reset_hub_names = yes
	}
}
