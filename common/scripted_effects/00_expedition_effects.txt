	if = {
		limit = {
			OR = {
				ROOT.var:current_expedition_location_var ?= {
					NOT = {
						state_region = prev.state_region
					}
				}
				NOT = {
					exists = var:current_expedition_location_var
				}
			}
		}
		ROOT = {
			set_variable = { name = current_expedition_location_var value = prev }
		}
	}
}
