# East Aegean Islands
STATE_EAST_AEGEAN_ISLANDS_state_name_assign = {
    if = { # Greek
        limit = {
            owner = {
                any_primary_culture = {
                    has_discrimination_trait = language_greek
                }
            }
        }

        set_state_name = STATE_EAST_AEGEAN_ISLANDS_greek

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_city_greek
        }

        set_hub_name = {
            type = farm
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_farm_greek
        }

        set_hub_name = {
            type = wood
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_wood_greek
        }

        set_hub_name = {
            type = port
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_port_greek
        }
    }
    else_if = { # Turkish
        limit = {
            owner = {
                any_primary_culture = {
                    has_discrimination_trait_group = language_group_turkic
                }
            }
        }

        set_state_name = STATE_EAST_AEGEAN_ISLANDS_turkish

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_city_turkish
        }

        set_hub_name = {
            type = farm
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_farm_turkish
        }

        set_hub_name = {
            type = wood
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_wood_turkish
        }

        set_hub_name = {
            type = port
            name = HUB_NAME_STATE_EAST_AEGEAN_ISLANDS_port_turkish
        }
    }
    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}

    if = { # South Asian
        limit = {
            owner = {
                any_primary_culture = {
                    has_discrimination_trait_group = heritage_group_south_asian
                }
                NOT = {
                    any_overlord_or_above = {
                        any_primary_culture = {
                            has_discrimination_trait = language_anglophone
                        }
                    }
                }
            }
        }

        set_state_name = STATE_MADRAS_indian

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_MADRAS_city_indian
        }

        set_hub_name = {
            type = farm
            name = HUB_NAME_STATE_MADRAS_farm_indian
        }

        set_hub_name = {
            type = wood
            name = HUB_NAME_STATE_MADRAS_wood_indian
        }

        set_hub_name = {
            type = mine
            name = HUB_NAME_STATE_MADRAS_mine_indian
        }

        set_hub_name = {
            type = port
            name = HUB_NAME_STATE_MADRAS_port_indian
        }
    }
    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}

    else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
        reset_state_name = yes
        reset_hub_names = yes
    }
}
