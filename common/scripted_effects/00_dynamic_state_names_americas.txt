STATE_XINGTIAN_state_name_assign = { #Colombian
	if = {
		limit = {
			owner = {
				any_primary_culture = {
					OR = { 
						cu:colombian ?= this
						cu:mexican ?= this
						cu:central_american ?= this
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_colombian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_colombian

	if = {
		limit = {
			owner = {
				country_has_primary_culture = cu:beornian
			}
		}
		set_state_name = STATE_NEAR_COLLASUYU_anglo
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_NEAR_COLLASUYU_city_anglo
		}
	}
	
	if = {
		limit = {
			owner = {
				any_primary_culture = { 
					has_discrimination_trait = language_hispanophone
				}
				NOR = {
					this ?= c:TWT
					this ?= c:IPM
				}
			}
		}
		set_state_name = STATE_NEAR_COLLASUYU_castilian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_NEAR_COLLASUYU_city_castilian
		}
		set_hub_name = {
			type = farm
			name = HUB_NAME_STATE_NEAR_COLLASUYU_farm_castilian
		}
		set_hub_name = {
			type = wood
			name = HUB_NAME_STATE_NEAR_COLLASUYU_wood_castilian
		}
		set_hub_name = {
			type = mine
			name = HUB_NAME_STATE_NEAR_COLLASUYU_mine_castilian
		}
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_NEAR_COLLASUYU_port_castilian
		}
	}
	else_if = {
		limit = {
			owner = {
				any_primary_culture = { 
					this ?= cu:scandinavian
				}
			}
		}
		set_state_name = STATE_NEAR_COLLASUYU_scandinavian
		set_hub_name = {
			type = port
			name = HUB_NAME_STATE_NEAR_COLLASUYU_port_scandinavian
		}
	}
	else = {
		reset_state_name = yes
		reset_hub_names = yes
	}
}
