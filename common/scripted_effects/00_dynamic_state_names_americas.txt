STATE_XINGTIAN_state_name_assign = { #Colombian
	if = {
		limit = {
			owner = {
				any_primary_culture = {
					OR = { 
						cu:colombian ?= this
						cu:mexican ?= this
						cu:central_american ?= this
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_colombian
		set_hub_name = {
			type = city
			name = HUB_NAME_STATE_XINGTIAN_city_colombian
		}
		set_hub_name = {
			type = farm

				}
				NOT = {
					any_primary_culture = {
						OR = {
							cu:colombian ?= this 
							cu:mexican ?= this 
							cu:central_american ?= this
						}
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_lusitanian
		set_hub_name = {
