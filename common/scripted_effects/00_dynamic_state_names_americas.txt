STATE_XINGTIAN_state_name_assign = { #Colombian
	if = {
		limit = {
			owner = {
				any_primary_culture = {
					OR = { 
						cu:colombian ?= this
						cu:mexican ?= this
						cu:central_american ?= this
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_colombian
		set_hub_names = colombian
	}
	else_if = { #Lusitanian
		limit = {
			owner = {
				any_primary_culture = {
					cu:lusitanian ?= this
				}
				NOT = {
					any_primary_culture = {
						OR = {
							cu:colombian ?= this 
							cu:mexican ?= this 
							cu:central_american ?= this
						}
					}
				}
			}
		}
		set_state_name = STATE_XINGTIAN_lusitanian
		set_hub_names = lusitanian
	}
	else_if = { #Cadien
		limit = {
			owner = {
				any_primary_culture = {
					cu:cadien ?= this
				}
			}
		}
		set_state_name = STATE_XINGTIAN_cadien
		set_hub_names = cadien
	}
	else_if = { #Buyeoan/Korean
		limit = {
			owner = {
				any_primary_culture = {
					has_discrimination_trait = language_korean
				}
				NOT = { 
					any_primary_culture = { has_discrimination_trait_group = language_group_sinitic } 
				}
			}
		}
		set_state_name = STATE_XINGTIAN_korean
		set_hub_names = korean
	}
	else_if = { #Kaiganese/Japanese
		limit = {
			owner = {
				any_primary_culture = {
					has_discrimination_trait = language_japanese
				}
			}
		}
		set_state_name = STATE_XINGTIAN_japanese
		set_hub_names = japanese
	}
	else = {
		reset_state_name = yes
		reset_hub_names = yes
	}
}
