country_legislature_name = {
	type = country
	text = {
		trigger = {
			has_law_or_variant = law_type:law_technocracy
		}
		localization_key = legislature_board_of_directors
	}
	text = {
		trigger = {
			country_has_primary_culture = cu:russian
			has_law_or_variant = law_type:law_council_republic
		}
		localization_key = legislature_congress_of_soviets
	}
	text = {
		trigger = {
			NOT = {
				country_has_primary_culture = cu:russian
			}
			has_law_or_variant = law_type:law_council_republic
		}
		localization_key = legislature_congress_of_councils
	}

	text = {
		trigger = {
			AND = {
				c:RUS ?= this
				OR = {
					has_law_or_variant = law_type:law_monarchy
					has_law_or_variant = law_type:law_presidential_republic
					has_law_or_variant = law_type:law_parliamentary_republic
				}
			}
		}
		localization_key = legislature_duma
	}
	text = {
		trigger = {
			c:SWI ?= this
		}
		localization_key = legislature_federal_diet
	}
	text = {
		trigger = {
			OR = {
				c:SWE ?= this
				AND = {
					c:SCA ?= this
					was_formed_from = SWE
				}
			}
		}
		localization_key = legislature_riksdag
	}

	text = {
		trigger = {
			OR = {
				c:DEN ?= this
				AND = {
					c:SCA ?= this
					was_formed_from = DEN
				}
			}
		}
		localization_key = legislature_folketing
	}

	text = {
		trigger = {
			OR = {
				c:PRU ?= this
				c:BAV ?= this
			}
		}
		localization_key = legislature_landtag
	}
	text = {
		trigger = {
			AND = {
				c:CHI ?= this
				has_law_or_variant = law_type:law_monarchy
				country_has_voting_franchise = no
			}
		}
		localization_key = legislature_grand_council
	}
	text = {
		trigger = {
			AND = {
				c:CHI ?= this
				has_law_or_variant = law_type:law_monarchy
				country_has_voting_franchise = yes
			}
		}
		localization_key = legislature_advisory_council
	}
	text = {
		trigger = {
			OR = {
				c:FRA ?= this
				c:BUL ?= this
				has_law_or_variant = law_type:law_parliamentary_republic
				has_law_or_variant = law_type:law_presidential_republic
			}
		}
		localization_key = legislature_national_assembly
	}
	text = { # Fallback
		trigger = {
			always = yes
		}
		localization_key = legislature_parliament
	}
}

country_red_season_name_dynamic = {
	type = country
	text = {
		trigger = {
			OR = {
				month >= 11
				month < 2
			}
		}
		localization_key = red_winter_loc
	}
	text = {
		trigger = {
			month >= 2
			month < 5
		}
		localization_key = red_spring_loc
	}
	text = {
		trigger = {
			month >= 5
			month < 8
		}
		localization_key = red_summer_loc
	}
	text = {
		trigger = {
			month >= 8
			month < 11
		}
		localization_key = red_autumn_loc
	}
	text = { # Fallback
		trigger = {
			always = yes
		}
		localization_key = red_summer_loc
	}
}
