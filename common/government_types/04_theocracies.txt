		has_law_or_variant = law_type:law_theocracy
		OR = {
			AND = {

		has_law_or_variant = law_type:law_theocracy

			AND = {
				exists = c:PAP
				c:PAP = ROOT
			}
			AND = {
				exists = c:AVN
				c:AVN = ROOT
				NOT = { exists = c:PAP }
				NOT = {
					exists = c:ITA
					has_government_type = gov_papacy
				}
				NOT = {
					exists = c:ITC
					has_government_type = gov_papacy
				}

			AND = {
				exists = c:ITA
