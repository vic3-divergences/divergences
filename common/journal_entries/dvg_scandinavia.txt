je_sca_cufc_finland = { #Dealing with Finland

	group = je_group_internal_affairs
	
	possible = {
		ROOT = {
			has_journal_entry = je_sca_chaos_under_four_crowns
			any_scope_state = {
				owner = ROOT
				state_region = {
					is_homeland = cu:finnish
				}
			}
		}
	}
	
	is_shown_when_inactive = {
		c:SCA = ROOT
	}
}
