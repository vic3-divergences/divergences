	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		
	}
	possible = {
		
	}
	immediate = {
		random_primary_culture = {
			save_scope_as = primary_culture_scope
		}
	}
	scripted_button = je_the_eastern_border_treaty_of_aigun
	on_monthly_pulse = {
		random_events = {
			85 = 0
			15 = dvg_ryuukidou.6
		}
		effect = {
			if = {
				limit = {
				has_modifier = dvg_monastic_teachings		
				}
				convert_population = {			
					target = rel:gelugpa
					value = 0.01
				}
			}
		}
	}

	complete = {
		NOT = {
			any_neighbouring_state = {
				region = sr:region_west_siberia
				owner = {
					is_country_type = decentralized
				}
			}
			any_neighbouring_state = {
				region = sr:region_east_siberia
				owner = {
					is_country_type = decentralized
				}
			}
		}
	}

	on_complete = {
		every_scope_state = {
			limit = {
				OR = {
					region = sr:region_west_siberia
					region = sr:region_east_siberia
				}
			}
			state_region = {
				add_homeland = scope:primary_culture_scope
			}
			random_scope_pop = {
				change_pop_culture = {
					target = scope:primary_culture_scope
					value = 1
				}
			}
		}
		trigger_event = {
			id = dvg_ryuukidou.2
		}
	}

	

	transferable = yes

	weight = 5000

	should_be_pinned_by_default = yes
