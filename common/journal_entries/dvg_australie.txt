
je_nvz_colonization = {
	icon = "gfx/interface/icons/event_icons/event_zhusha.dds"

	group = je_group_foreign_affairs

	is_shown_in_lobby = {
		country_has_primary_culture = cu:australian
	}
	is_shown_when_inactive = {
		country_has_primary_culture = cu:australian
	}
	possible = {
		has_technology_researched = colonization
		NOT = {
			has_law = law_type:law_no_colonial_affairs
		}
	}
	immediate = {
		
	}

	on_monthly_pulse = {
		fallback = on_yearly_events

		random_events = {
			15 = dvg_australie.10
		}
	}

	complete = {
		NOT = {
			any_neighbouring_state = {
				owner = {
					is_country_type = decentralized
				}
			}
		}
	}
	
	fail = {
		has_law = law_type:law_no_colonial_affairs
	}

    on_complete = {
		every_scope_state = {
			
			state_region = {
				add_homeland = cu:australian
			}
			if = {
				limit = {
				culture_percent_state = {
					target = cu:javan
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:javan
				}
			}

			if = {
				limit = {
				culture_percent_state = {
					target = cu:sundanese
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:sundanese
				}
			}

			if = {
				limit = {
				culture_percent_state = {
					target = cu:minangkabau
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:minangkabau
				}
			}
			if = {
				limit = {
				culture_percent_state = {
					target = cu:malay
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:malay
				}
			}
		}
		
	}
	
    on_fail = {

	}
	
	weight = 100
	should_be_pinned_by_default_involved = yes

}


je_chougie_unite = {
	icon = "gfx/interface/icons/event_icons/event_zhusha.dds"

	group = je_group_historical_content

	is_shown_in_lobby = {
		exists = c:NJY
		c:NJY = this
	}
	is_shown_when_inactive = {
		exists = c:NJY
		c:NJY = this
	}
	immediate = {
		
	}
	possible = {
		has_technology_researched = colonization
	}
	on_monthly_pulse = {
		fallback = on_yearly_events
		effect = {
			if = {
				limit = {
				exists = c:YLN
				exists = c:NJY
				c:YLN = {
					is_ai = yes
				}
				c:NJY = {
					is_ai = yes
				}
				}
				c:NJY = {
					change_relations = {
						country = c:YLN
						value = 1
					}
				}
			}
		}
		
	}

	complete = {
		has_technology_researched = egalitarianism
		AND = {
			exists = c:YLN
			c:YLN = {
				is_ai = yes
				relations:root >= relations_threshold:amicable
				top_overlord = {
							THIS = root.top_overlord
						}
				this.gdp < root.gdp
				OR = {
					has_attitude = {
					who = ROOT
					attitude = cooperative
					}
					has_attitude = {
					who = ROOT
					attitude = genial
					}
					liberty_desire < 25
				}

				

				top_overlord = {
							THIS = root.top_overlord
				}
			}
			
		}
		OR = {
			has_law = law_type:law_subjecthood
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural
		}
	}
	
	fail = {
		has_technology_researched = nationalism
		OR = {
			cu:aborigine_malay = {
				culture_current_fervor > 24
			}
			cu:aborigine_french = {
				culture_current_fervor > 24
			}
		}
		any_scope_state = {
			state_region = s:STATE_YOLNGU
			turmoil > 0.50
		}
		
	}

    on_complete = {
		trigger_event = {
			id = dvg_australie.7
		}
	}
	
    on_fail = {
		trigger_event = {
			id = dvg_australie.9
		}
	}
	
	weight = 100
	should_be_pinned_by_default_involved = yes

}
