je_chougie_unite = {
	icon = "gfx/interface/icons/event_icons/event_zhusha.dds"

	group = je_group_historical_content

	is_shown_in_lobby = {
		exists = c:NJY
		c:NJY = this
	}
	is_shown_when_inactive = {
		exists = c:NJY
		c:NJY = this
	}
	possible = {
		has_technology_researched = colonization
	}

	complete = {
		has_technology_researched = egalitarianism
		AND = {
			exists = c:YLN
			c:YLN = {
				liberty_desire < 25

				scope:target_country.relations:root >= relations_threshold:amicable

				OR = {
					has_attitude = {
					who = ROOT
					attitude = cooperative
					}
					has_attitude = {
					who = ROOT
					attitude = genial
					}
				}

				

				top_overlord = {
							THIS = root.top_overlord
				}
			}
			
		}
		OR = {
			has_law = law_type:law_subjecthood
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural
		}
	}
	
	fail = {
		is_at_war = yes
		any_enemy_in_war = {
			c:YLN = this	
		}
	}

    on_complete = {
		
	}
	
    on_fail = {

	}
	
	weight = 100
	should_be_pinned_by_default = yes
}

	}

		owns_entire_state_region = STATE_YOLNGU

			any_scope_state = {
				turmoil > 0.20
			}

				owner = {
					is_country_type = decentralized
				}
			}
		}
		OR = {
			has_technology_researched = pan-nationalism
			has_technology_researched = human_rights

			culture_current_fervor < 25
		}

			}

			}
		}
	}

    on_complete = {
		
	}
	
    on_fail = {

	}
	
	weight = 100
