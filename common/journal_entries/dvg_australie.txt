
je_nvz_colonization = {
	icon = "gfx/interface/icons/event_icons/event_zhusha.dds"

	group = je_group_foreign_affairs

	scripted_button = more_convicts_button
	scripted_button = end_convicts_button

	is_shown_in_lobby = {
		exists = c:NVZ
		c:NVZ = this
	}

	complete = {
		NOT = {
			any_neighbouring_state = {
				owner = {
					is_country_type = decentralized
				}
			}
		}
	}
	
	fail = {
		has_law = law_type:law_no_colonial_affairs
	}

    on_complete = {
		every_scope_state = {
			
			state_region = {
				add_homeland = cu:australian
			}
			if = {
				limit = {
				culture_percent_state = {
					target = cu:javan
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:javan
				}
			}

			if = {
				limit = {
				culture_percent_state = {
					target = cu:sundanese
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:sundanese
				}
			}

			if = {
				limit = {
				culture_percent_state = {
					target = cu:minangkabau
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:minangkabau
				}
			}
			if = {
				limit = {
				culture_percent_state = {
					target = cu:malay
					value > 0.33
				}
				}
				state_region = {
				add_homeland = cu:malay
				}
			}
		}
		
	}
	
    on_fail = {

	}

	on_yearly_pulse = {
		fallback = on_yearly_events

		effect = {
			if = {
					limit = {
						OR = {
							top_overlord = {
								has_technology_researched = human_rights
							}
						}
						has_technology_researched = human_rights
					}
					trigger_event = {
						id = dvg_australie.2
					}
			
			}
		}
	}
	
	weight = 100
	should_be_pinned_by_default = yes
}


je_chougie_unite = {
	icon = "gfx/interface/icons/event_icons/event_zhusha.dds"

	group = je_group_historical_content

	is_shown_in_lobby = {
		exists = c:NJY
		c:NJY = this
	}
	is_shown_when_inactive = {
		exists = c:NJY
		c:NJY = this
	}
	immediate = {
		
	}
	possible = {
		has_technology_researched = colonization
	}

	complete = {
		has_technology_researched = egalitarianism
		AND = {
			exists = c:YLN
			c:YLN = {
				is_ai = yes
				relations:root >= relations_threshold:amicable
				top_overlord = {
							THIS = root.top_overlord
						}
				this.gdp < root.gdp
				OR = {
					has_attitude = {
					who = ROOT
					attitude = cooperative
					}
					has_attitude = {
					who = ROOT
					attitude = genial
					}
					liberty_desire < 25
				}

				

				top_overlord = {
							THIS = root.top_overlord
				}
			}
			
		}
		OR = {
			has_law = law_type:law_subjecthood
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural
		}
	}
	
	fail = {
		has_technology_researched = nationalism
		c:YLN = {
			NOT = {
				top_overlord = {
							THIS = root.top_overlord
				}
			}
		}
		
	}

    on_complete = {
		trigger_event = {
			id = dvg_australie.7
		}
	}
	
    on_fail = {
		trigger_event = {
			id = dvg_australie.9
		}
	}
	
	weight = 100
	should_be_pinned_by_default = yes
}
