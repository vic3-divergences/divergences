je_lotharian_house_of_cards = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_internal_affairs

	possible = {
		ig:ig_armed_forces = {
			is_in_government = yes
		}
	}

	immediate = {

		

	}

	on_monthly_pulse = {
		
		effect = {
			if = {
				limit = {
					ig:ig_armed_forces = {
						is_powerful = yes
						is_in_government = yes
					}
					any_interest_group = {
					count > 1
					ig_approval < -6
					}
				}
				change_variable = {
					name = aigle_var
					add = 1
				}
			}
			if = {
				limit = {
					any_interest_group = {
					count > 3
					ig_approval < -10
					}
					
				}
				change_variable = {
					name = aigle_var
					add = 20
				}
			}
			if = {
				limit = {
					ig:ig_armed_forces = {
						is_insurrectionary = yes
					}
					
				}
				change_variable = {
					name = aigle_var
					add = 20
				}
			}
			if = {
				limit = {
					any_interest_group = {
						is_insurrectionary = yes
						ig_clout >= 0.12
					}
					
				}
				change_variable = {
					name = aigle_var
					add = 15
				}
			}
			if = {
				limit = {
					ig:ig_armed_forces = {
						is_powerful = yes
						is_in_government = yes
					}
					root.average_sol > 10
				}
				change_variable = {
					name = aigle_var
					add = -1
				}
			}
			if = {
				limit = {
					ig:ig_armed_forces = {
						is_powerful = yes
						is_in_government = yes
					}
					root.average_sol > 11
				}
				change_variable = {
					name = aigle_var
					add = -2
				}
			}
			if = {
				limit = {
					has_variable = benevolent_dictatorship
				}
				change_variable = {
					name = aigle_var
					add = -2
				}
			}
			if = {
				limit = {
					this = { var:aigle_var >= 140 }
					NOT = {
						has_variable = aigle_criticized
					}
				}
				trigger_event = {
					id = dvg_lotharingia.15
				}
				set_variable = aigle_criticized
			}
			if = {
				limit = { 
					has_modifier = dvg_fragile_country
				}
				change_variable = {
					name = aigle_var
					add = 6
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		
		
		trigger_if = {
			limit = { exists = var:aigle_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		OR = {
			any_interest_group = {
				count >= 2
				ig_approval < -6
				ig_clout > 0.10
				}
			any_interest_group = {
					is_powerful = yes
					is_insurrectionary = yes
				}

		}

		
		
		has_law = law_type:law_autocracy
		root.average_sol <= 11
	}
	
	on_complete = {
		#set_variable = house_of_cards_complete
		trigger_event = {
			id = dvg_lotharingia.5
		}
	}
	
	fail = {
		OR = {
			ig:ig_armed_forces = {
				is_powerful = no
				is_in_government = no
			}
			AND = {
				var:aigle_var <= 0
				year >= 1846 
			}
			
			NOT = { has_law = law_type:law_presidential_republic
			        has_law = law_type:law_monarchy
			}
			NOT = { has_law = law_type:law_autocracy
			        has_law = law_type:law_monarchy
			}
		}
		
	}
	
	on_fail = {
		#set_variable = house_of_cards_complete
		trigger_event = {
			id = dvg_lotharingia.4
		}
		
	}

	
	
	timeout = 9125

	on_timeout = {
		trigger_event = {
			 id = dvg_lotharingia.2
		}
	}
	invalid = {
		NOT = {
			any_scope_character = {
				has_variable = chef_de_aigle_var
			}
		}
	}
	on_invalid = {
		trigger_event = {
			id = dvg_lotharingia.2
	   }
	}

	current_value = {
		value = root.var:aigle_var
	}

	goal_add_value = {
		value = 250
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
