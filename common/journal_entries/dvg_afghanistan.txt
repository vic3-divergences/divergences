je_panjabi_revolt = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		
	}

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = punjab_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:panjabi = {
						root = {
							cultural_acceptance_base = { target = prev value < acceptance_status_5 }
						}
					}	
				}
				change_variable = {
					name = punjab_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:panjabi = {
				culture_secession_progress = {
					target = ROOT
					value > 0.1
				}	
				}

				}
				change_variable = {
					name = punjab_var
					add = 1
				}
			}  
			if = {
				limit = {
					has_technology_researched = nationalism
				}
				change_variable = {
					name = punjab_var
					add = 1
				}
			}  
			if = {
				limit = {
					any_scope_pop = {
						standard_of_living < 6.0 
						religion = rel:sunni
					}
				}
				every_scope_pop = {
					limit = {
						standard_of_living < 6.0
						religion = rel:sunni
					}
					change_pop_religion = {
						target = rel:sikh
						value = 0.2
					}
				}
			}  
	}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		OR = {
		trigger_if = {
			limit = { exists = var:punjab_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		}

	}
	
	on_complete = {
		#set_variable = dvg_punjabi_revolt_complete
		trigger_event = {
			id = dvg_afghanistan.2
		}
	}
	
	fail = {
		OR = {
			has_law = law_type:law_total_separation 
			has_law = law_type:law_state_religion
		}
		has_technology_researched = egalitarianism
	}
	
	on_fail = {
		#set_variable = dvg_panjabi_accepted
		trigger_event = {
			id = dvg_afghanistan.3
		}
	}

	invalid = { 
	NOT = {
					has_state_in_state_region = STATE_PUNJAB
				}
	}
	
	current_value = {
		value = root.var:punjab_var
	}

	goal_add_value = {
		value = 300
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}

je_panjabi_revolt2 = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = s:STATE_PUNJAB
			
		}
	}

	possible = {
		cu:panjabi = {
			 root = {
				cultural_acceptance_base = { target = prev value < acceptance_status_5 }
			}
		}	
	}

	immediate = {
		set_variable = {
			name = punjab_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cu:panjabi = {
						root = {
							cultural_acceptance_base = { target = prev value < acceptance_status_5 }
						}
					}	
				}
				change_variable = {
					name = punjab_var
					add = 1
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:punjab_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		is_at_war = yes
		
		
		
	}
	
	on_complete = {
		#set_variable = dvg_punjabi_revolt_complete
		trigger_event = {
			id = dvg_persia.2
		}
	}
	
	fail = {
		NOT = { has_law = law_type:law_total_separation }
		OR = {
				NOT = {owns_entire_state_region = STATE_PUNJAB}
				cu:panjabi = {
					root = {
					cultural_acceptance_base = { target = prev value >= acceptance_status_5 }
				}
			}
		}
		

	}
	
	on_fail = {
		set_variable = dvg_sikh_accepted
		trigger_event = {
			id = dvg_persia.3
		}
	}

	
	current_value = {
		value = root.var:punjab_var
	}

	goal_add_value = {
		value = 48
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
