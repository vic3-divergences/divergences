je_dvg_greek_nationalism = {
	icon = "gfx/interface/icons/invention_icons/nationalism.dds"
	
	group = je_group_historical_content

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = {
				is_homeland = cu:greek
			}
		}
		#exists = c:VEN
		#c:VEN = ROOT
		any_scope_character = {
			has_variable = zimisces_var
		}
		
		NAND = {
			exists = c:GRE
			c:GRE = ROOT
		}
	}

	possible = {
		OR = {
			has_technology_researched = nationalism
			AND = {
				exists = c:GRE
				c:GRE = {
					has_technology_researched = nationalism
				}
			}
			

		}
	}

	immediate = {
		if = {
			limit = {
				any_scope_character = {
					has_variable = zimisces_var
				}
				exists = c:VEN
				c:VEN = ROOT
				
			}
			trigger_event = {
				id = dvg_greece.201
			}

		}
		set_variable = {
			name = greece_var
			value = 0
		}
	}
	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					exists = c:GRE
					
					c:GRE = {
						ruler = {
							NOT = {
								has_variable = zimisces_var
							}
						}
						any_scope_character = {
							has_variable = zimisces_var
						}
					}
				}
				
			}
		}
		
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cultural_acceptance_base = { target = cu:greek value < acceptance_status_4 }	
					exists = c:GRE
				}
				c:GRE = {
					add_liberty_desire = 0.10
				}
			}
			if = {
				limit = {
					cultural_acceptance_base = { target = cu:greek value >= acceptance_status_4 }
					exists = c:GRE
				}
				c:GRE = {
					add_liberty_desire = -0.20
				}
			}
			if = {
				limit = {
					country_has_voting_franchise = yes
					
					exists = c:GRE
				}
				c:GRE = {
					add_liberty_desire = -0.20
				}
			}
			
			if = {
				limit = {
					NOT = {
						exists = c:GRE
						has_variable = greece_to_liberate
					}
				}
				set_variable = greece_to_liberate

				
			}
			if = {
				limit = {
					has_variable = council_start
					NOT = {
						has_variable = greece_citizen_var
					}
				}
				trigger_event = {
					id = dvg_greece.210
				}
				
			}
			if = {
				limit = {
					has_variable = council_start
					NOT = {
						has_variable = greece_voting_var
					}
				}
				trigger_event = {
					id = dvg_greece.211
				}
			}
			if = {
			#Greek Integration

				limit = {
					has_variable = council_start
					has_variable= greece_citizen_var
					OR = {
						has_variable = greece_voting_var
						has_law = law_type:law_census_voting
					}	
					OR = {
						has_variable= greece_citizen_var
						cultural_acceptance_base = { target = cu:greek value >= acceptance_status_4 }
					}
					
				}
				trigger_event = {
					id = dvg_greece.212
				}
			}
			#Greek Revolt
			
		}
	}
	

	complete = {
		c:GRE = {
			liberty_desire > 90
		}
		has_variable = council_start
		has_variable = greece_voting_var
		has_variable= greece_citizen_var
		
	}

	on_complete = {
		if = {
			limit = {
				exists = c:GRE
				has_state_in_state_region = STATE_CYPRUS
			}
			trigger_event = {
				id = dvg_greece.203
			}
		}
	}
	fail = {
		NOT = {
			cultural_acceptance_base = { target = cu:greek value >= acceptance_status_4 }
			
			has_law = law_type:law_oligarchy
			has_law = law_type:law_autocracy
		}
		c:GRE = {
			liberty_desire < 25
		}

	}
	invalid = {
		OR = {
			has_variable = council_cancelled
			AND = {
				exists = c:GRE
				is_diplomatic_play_enemy_of = c:GRE
			}
			NOT = {
				any_scope_state = {
					state_region = {
						is_homeland = cu:greek
					}
				}
				any_subject_or_below = {
					country_has_primary_culture = cu:greek
				}
			}
		}
		
		
	}
	on_fail = {
		trigger_event = {
			id = dvg_greece.212
		}
	}

	weight = 500
	should_be_pinned_by_default = yes
	goal_add_value = {
		value = 48
	}
	timeout = 180
	on_timeout = {
		trigger_event = {
			id = dvg_greece.203
		}
	}
	
}
