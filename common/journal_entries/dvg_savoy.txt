je_lost_duke = {
	icon = "gfx\interface\icons\event_icons\event_empty_chair.dds"

	group = je_group_major_events


	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		scope:scope_luigi = { add_trait = lost_duke_reform remove_trait = lost_duke}
		scope:scope_amedeo = {kill_character = yes}
		add_loyalists = 0.20
	}

	fail = {
		scope:scope_luigi = { is_ruler_of_own_country = no }
		NOT = {
			OR = { 
			has_law = law_type:law_monarchy
			has_law = law_type:law_social_monarchy
			}
		}
	}
	
	on_fail = {
		add_radicals = 0.15
		scope:scope_luigi = {kill_character = yes}
	}
	current_value = {
		value = var:lost_duke_var
	}

	goal_add_value = {
		add = 3
	}

	progressbar = yes

	weight = 10000
	
	scripted_button = button_lost_duke_school

	should_be_pinned_by_default_uninvolved_or_context = yes
}
