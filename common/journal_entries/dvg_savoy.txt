je_lost_duke = {
	icon = "gfx\interface\icons\event_icons\event_empty_chair.dds"

	group = je_group_historical_content


	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		if = {
			limit = {
				ruler = {
					has_variable = is_luigi
				}
			}
			ruler = {
				add_trait = lost_duke_reform
				remove_trait = lost_duke
				remove_trait = grifter
				add_trait = ambitious
			}
		}
		if = {
			limit = {
				heir = {
					has_variable = is_amedeo
				}
			}
			heir = {kill_character = yes}
		}
		add_loyalists = { value = 0.20 }
	}

	fail = {
		ruler = {
			NOT = {
				has_variable = is_luigi
			}
		}
		NOR = {
			has_law = law_type:law_monarchy
			has_law = law_type:law_social_monarchy
		}
	}
	
	on_fail = {
		add_radicals = { value= 0.15 }
		if = {
			limit = {
				ruler = {
					has_variable = is_luigi
				}
			}
			ruler = { kill_character = yes }
		}
	}
	current_value = {
		value = var:lost_duke_var
	}

	goal_add_value = {
		add = 3
	}

	progressbar = yes

	weight = 10000
	
	scripted_button = button_lost_duke_school

	should_be_pinned_by_default_uninvolved_or_context = yes
}

je_radical_duke = {
	icon = "gfx\interface\icons\alert_icons\radicalized.dds"
	
	group = je_group_historical_content
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		if = {
			limit = {
				heir = {
					has_variable = is_amedeo
				}
			}
			heir = { set_character_as_ruler = yes }
		}
		any_character = {
			limit = {
				has_variable = is_amedeo
			}
			add_trait = radical_duke_reform
			add_trait = ambitious
			remove_trait = radical_duke
		}
		add_loyalists = { value = 0.20 }
	}

	fail = {
		NOR = {
			ruler = {
				has_variable = is_amedeo
			}
			heir = {
				has_variable = is_amedeo
			}
		}
	}
	
	on_fail = {
		add_radicals = { value= 0.15 }
		
		if = {
			limit = {
				heir = {
					has_variable = is_amedeo
				}
			}
			heir = { kill_character = yes }
		}
	}
	current_value = {
		value = var:radical_duke_var
	}

	goal_add_value = {
		add = 3
	}

	progressbar = yes

	weight = 10000
	
	scripted_button = button_radical_duke_aids
	scripted_button = button_radical_duke_kills

	should_be_pinned_by_default_uninvolved_or_context = yes
}
