je_armenian_revolt = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_internal_affairs

	possible = {
		
	}

	immediate = {
		set_variable = {
			name = armenia_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					cultural_acceptance_base = {
						target = cu:armenian
						value < acceptance_status_4
					 }
				}
				change_variable = {
					name = armenia_var
					add = 1
				}
			}
			if = {
				limit = {
					cu:armenian = {
				culture_secession_progress = {
					target = ROOT
					value > 0.5
				}	
				}
				
						}
				change_variable = {
					name = armenia_var
					add = 24
				}
			}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:armenia_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		is_at_war = yes
		
		
		
	}
	
	on_complete = {
		#set_variable = dvg_armenia_revolt_complete
		trigger_event = {
			id = dvg_persia.2
		}
	}
	
	fail = {
		OR = {
			NOT = {owns_entire_state_region = STATE_ARMENIA}
			cu:armenian = {
				root = { 
					cultural_acceptance_base = { target = prev value >= acceptance_status_4 }
				}
			}
		}
	}
	
	on_fail = {
		#set_variable = dvg_armenian_accepted
		
	}

	
	current_value = {
		value = root.var:armenia_var
	}

	goal_add_value = {
		value = 48
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
