je_amazonie_reconstruction = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	is_shown_when_inactive =
	{
	
		
	}

	possible = {
		
		
	}

	immediate = {
		set_variable = {
			name = reconcile_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				NOT = { cu:afro_amerigan = { culture_is_discriminated_in = ROOT}	}
				capital = {
					
					turmoil < 0.1
				}
				any_scope_state = {
					count >= 3
						is_incorporated = yes
						is_capital = no
						turmoil < 0.25
				 }
			}
			change_variable = {
					name = reconcile_var
					add = 1
				}
		}
	}

	complete = {
		# If-else clause to avoid errors because of missing variable before JE activates
		trigger_if = {
			limit = { exists = var:reconcile_var }
			scope:journal_entry = { is_goal_complete = yes }
		}
		
		NOT = { cu:afro_amerigan = { culture_is_discriminated_in = ROOT}	}
		capital = {
			
			devastation = 0
			turmoil < 0.1
		}
		any_scope_state = {
					count >= 3
						is_incorporated = yes
						is_capital = no
						turmoil < 0.25
					}
		
		has_law = law_type:law_slavery_banned
	}
	
	on_complete = {
		set_variable = reconcile_complete
		trigger_event = {
			id = dvg_amazonie.12
		}
	}
	
	fail = {
		
	}
	
	on_fail = {
		
	}

	current_value = {
		value = root.var:reconcile_var
	}

	goal_add_value = {
		value = 120
	}

	progressbar = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
