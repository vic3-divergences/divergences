je_skraeling_missions = {
	icon = "gfx/interface/icons/religion_icons/catholic.dds"

	group = je_group_historical_content

	scripted_button = dvg_ban_missionaries_button_skr

	is_shown_when_inactive = {
	
	}

	possible = {
		
	}

	immediate = {
		custom_tooltip = {
			text = missionary_activities_begin_tt
			trigger_event = {
				id = heavenly.1
			}
		}
		every_scope_state = {
			limit = {
				any_neighbouring_state = {
					is_treaty_port = yes
					owner = {
						religion = {
							has_discrimination_trait = heritage_christian
						}
					}
				}
			}
			set_variable = open_to_missionaries_state
		}
	}

	on_monthly_pulse = {
		effect = {
			
			if = {
				limit = {
					has_law = law_type:law_no_migration_controls
					any_scope_state = {
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
				}
				every_scope_state = {
					limit = {
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
					set_variable = open_to_missionaries_state
				}
			}
			else = {
				every_scope_state = {
					limit = {
						any_neighbouring_state = {
							is_treaty_port = yes
							owner = {
								religion = {
									has_discrimination_trait = heritage_christian
								}
							}
						}
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
					set_variable = open_to_missionaries_state
				}
			}
		}
		random_events = {
			20 = 0
			5 = dvg_skraeling.27
			5 = dvg_skraeling.28
		}
	}

	complete = {
		ig:ig_intelligentsia = {
			is_powerful = yes
			is_in_government = yes
		}
	}
	on_complete = {
		set_state_religion = rel:catholic
		ig:ig_devout = {
			set_interest_group_name = ig_catholic_church
		}
		trigger_event = {
			id = dvg_skraeling.31
		}
		change_relations = {
			country = c:PLA
			value = 30
		}
		change_relations = {
			country = c:CHE
			value = 30
		}
	}

	fail = {
		OR = {
			has_global_variable = skr_banned_missions
			ig:ig_intelligentsia = {
				is_powerful = yes
				OR = {
					is_insurrectionary = yes
					ig_approval < -15
				}
			}
		}
		
	}
	on_fail = {
		if = {
			limit = {
				NOT = {
					has_global_variable = skr_banned_missions
				}
			}
			trigger_event = {
				id = dvg_skraeling.14
			}
		}

		if = {
			limit = {
				has_global_variable = skr_banned_missions
			}
			trigger_event = {
				id = dvg_skraeling.31
			}
		}
	
	}

	weight = 10000
	should_be_pinned_by_default = yes
}
