je_consolidate_colonial_rule_bei = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	complete = {

		OR = {
		  country_or_subject_owns_entire_state_region = STATE_ACEH
		  country_or_subject_owns_entire_state_region = STATE_MINDANAO
		  country_or_subject_owns_entire_state_region = STATE_SUNDA_ISLANDS
		}
		
		any_state = {
			count >= 10
			
			filter = {
				region = sr:region_indonesia
			}
			OR = {
				owner = ROOT
				owner = {
					is_subject_of = ROOT
				}
				trigger_if = {
					limit = {
						exists = ROOT.top_overlord
					}
					OR = {
						owner = ROOT.top_overlord
						owner = {
							is_subject_of = ROOT.top_overlord
						}
					}
				}
			}
		}
		
		
		highest_secession_progress < 0.5
	}

	on_complete = {
		trigger_event = { id = east_indies.1 }
		set_variable = consolidated_colonial_rule
	}

	fail = {
		OR = {
			NOT = {
				exists = c:BRG
			}
			trigger_if = {
				limit = {
					exists = c:BRG
				}
				NOT = {
					is_subject_of = c:BRG
				}
			}
			is_subject_type = subject_type_colony
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					highest_secession_progress < 0.3
				}
				change_variable = {
					name = east_indies_revolt_var
					add = 1
				}
				clamp_variable = { name = east_indies_revolt_var max = 120 min = 0 }
			}
		}
	}

	invalid = {
		has_variable = bic_collapse_var
	}

	current_value = {
		value = root.var:east_indies_revolt_var
	}

	goal_add_value = {
		add = 120
	}

	should_be_pinned_by_default = yes

	weight = 100
}

je_reform_indies = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	group = je_group_internal_affairs

	is_shown_when_inactive = {
		AND = {
			exists = c:BEI
			c:BEI = THIS
		}
	}

	possible = {
		OR = {
			has_law = law_type:law_free_trade
			has_law = law_type:law_mercantilism
			NOT = { has_law = law_type:law_land_based_taxation }
			NOT = { has_law = law_type:law_serfdom }
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		OR = {
			has_law = law_type:law_free_trade
			has_law = law_type:law_mercantilism
		}
		NOT = { has_law = law_type:law_land_based_taxation }
		NOT = { has_law = law_type:law_serfdom }
	}

	on_complete = {
		trigger_event = { id = east_indies.2 }
		show_as_tooltip = {
			add_journal_entry = {
				type = je_ethical_policy
			}
		}
	}

	invalid = {
		has_variable = bic_collapse_var
	}


	current_value = {
		value = root.gdp
	}

	goal_add_value = {
		value = root.gdp
		multiply = 0.5
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}
