je_white_african_national_identity_emergence = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		capital = {
			OR = {	
				is_in_geographic_region = geographic_region_north_africa
				is_in_geographic_region = geographic_region_west_africa
			}
		}
		any_primary_culture = {
			OR = {
				AND = {
					has_discrimination_trait = heritage_iberian
					OR = {
						has_discrimination_trait = language_lusophone
						has_discrimination_trait = language_occitano_romance
					}
				}
				has_discrimination_trait = heritage_italic
			}
		}
		hidden_trigger = {
			NOT = { has_variable = completed_culture_genesis }
		}
		hidden_trigger = {
			NOR = {
				c:SPA ?= this
				c:IBE ?= this
				c:ARI ?= this
				c:ITA ?= this
			}
		}
		is_revolutionary = no
	}

	possible = {
		has_technology_researched = nationalism
		hidden_trigger = {
			NOT = { has_variable = completed_culture_genesis }
		}
	}

	immediate = {
		set_national_identity_vars = yes
		set_variable = { name = national_identity_progress_var value = 0 }
		calculate_national_identity_progress = yes
		if = {
			limit = {
				OR = {
						c:TDR ?= this

						AND = {
							any_primary_culture = {
								has_discrimination_trait = heritage_italic
							}
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
				}
			}
			cu:venetian = {
				save_scope_as = old_national_culture
			}
			cu:orano = {
				save_scope_as = new_national_culture
			}
		}

		if = {
			limit = {
				OR = {
						c:NMD ?= this
						AND = {
							any_primary_culture = {
								has_discrimination_trait = heritage_iberian
								has_discrimination_trait = language_occitano_romance
							}
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
				}
			}
			cu:catalan = {
				save_scope_as = old_national_culture
			}
			cu:peuvermhels = {
				save_scope_as = new_national_culture
			}
		}

		if = {
			limit = {
				OR = {
					c:MAU ?= this
					c:NCZ ?= this
					AND = {
							any_primary_culture = {
								has_discrimination_trait = heritage_iberian
								has_discrimination_trait = language_lusophone
							}
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
					}
			}
			cu:portuguese = {
				save_scope_as = old_national_culture
			}
			cu:mauretanian = {
				save_scope_as = new_national_culture
			}
		}
		

		trigger_event = { id = culture_south_america.4 popup = yes }
	}

	on_weekly_pulse = {
		effect = {
			calculate_national_identity_progress = yes
		}
	}

	on_monthly_pulse = {
		effect = {
			add_national_identity_progress = yes
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		clear_national_identity_vars = yes
		trigger_event = {
			id = culture_south_america.1 popup = yes
		}
		show_as_tooltip = {
			if = {
				limit = {
					OR = {
						c:TDR ?= this

						AND = {
							any_primary_culture = {
								has_discrimination_trait = heritage_italic
							}
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
					}
				}
				add_primary_culture = cu:orano
				remove_primary_culture = cu:venetian
				if = {
					limit = {
						 cu:orano ?= {
							culture_national_awakening_occurred = no
						}
					}
					start_national_awakening = {
						culture =  cu:orano
						months = 120
						state_region = capital.state_region
					}
				}
			}
			if = {
				limit = {
					OR = {
					c:MAU ?= this
					c:NCZ ?= this
					AND = {
							any_primary_culture = {
								has_discrimination_trait = heritage_iberian
								has_discrimination_trait = language_lusophone
							}
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
					}
				}
				add_primary_culture = cu:mauretanian
				remove_primary_culture = cu:portuguese
				if = {
					limit = {
						 cu:mauretanian ?= {
							culture_national_awakening_occurred = no
						}
					}
					start_national_awakening = {
						culture =  cu:mauretanian
						months = 120
						state_region = capital.state_region
					}
				}
			}
			
			
			
			if = {
				limit = {
					OR = {
						c:NMD ?= this
						AND = {
							any_primary_culture = {
								has_discrimination_trait = heritage_iberian
								has_discrimination_trait = language_occitano_romance
							}
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
					}
				}
				add_primary_culture = cu:peuvermhels
				remove_primary_culture = cu:catalan
				if = {
					limit = {
						 cu:peuvermhels ?= {
							culture_national_awakening_occurred = no
						}
					}
					start_national_awakening = {
						culture =  cu:peuvermhels
						months = 120
						state_region = capital.state_region
					}
				}
			}
		}
		hidden_effect = {
			set_variable = completed_culture_genesis
		}
	}

	fail = {
		custom_tooltip = {
			text = completed_pan_national_project_tt
			OR = {
				c:SPA ?= this
				c:ITA ?= this
				c:IBE ?= this
				c:ARI ?= this
			}
		}
	}

	on_fail = {
		hidden_effect = {
			set_variable = completed_culture_genesis
		}
	}

	current_value = {
		if = {
			limit = {
				has_variable = national_identity_progress_var
			}
			value = root.var:national_identity_progress_var
		}
	}

	goal_add_value = {
		add = 100
	}

	weight = 100

	progressbar = yes

	should_be_pinned_by_default = yes
}
