je_south_american_national_identity = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_internal_affairs

	# Now activated by event

	modifiers_while_active = {
		modifier_assimilation_nnc
	}

	immediate = {
		if = {
			limit = {
				c:TDR ?= this
			}
			cu:venetian = {
				save_scope_as = old_national_culture
			}
			cu:orano = {
				save_scope_as = new_national_culture
			}
		}
		if = {
			limit = {
				c:MAU ?= this
			}
			cu:portuguese = {
				save_scope_as = old_national_culture
			}
			cu:mauretanian = {
				save_scope_as = new_national_culture
			}
		}
		if = {
			limit = {
				c:NCZ ?= this
			}
			cu:portuguese = {
				save_scope_as = old_national_culture
			}
			cu:mauretanian = {
				save_scope_as = new_national_culture
			}
		}
		if = {
			limit = {
				c:NMD ?= this
			}
			cu:catalan = {
				save_scope_as = old_national_culture
			}
			cu:peuvermhels = {
				save_scope_as = new_national_culture
			}
		}
		
	}

	complete = {
		NOR = {
			trigger_if = {
				limit = {
					OR = {
						c:TDR ?= this
						AND = {
							country_has_primary_culture = cu:venetian
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
					}
					}
					
				}
				any_scope_state = {
					state_region = {
						is_homeland = cu:venetian
					}
				}
			}
			trigger_if = {
				limit = {
					OR = {
						c:MAU ?= this
						c:NCZ ?= this
						AND = {
							country_has_primary_culture = cu:portuguese
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
						}
					}
				}
				any_scope_state = {
					state_region = {
						is_homeland = cu:portuguese
					}
				}
			}
			trigger_if = {
				limit = {
					OR = {
						c:NMD ?= this
						AND = {
							country_has_primary_culture = cu:catalan
							OR = {
								is_in_geographic_region = geographic_region_north_africa
								is_in_geographic_region = geographic_region_west_africa
							}
					}
					}
				}
				any_scope_state = {
					state_region = {
						is_homeland = cu:catalan
					}
				}
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = culture_south_america.3
		}
		add_loyalists = {
			value = large_radicals
			culture = scope:new_national_culture
		}
	}

	scripted_button = je_south_american_national_identity_integrate_a_state

	weight = 100

	should_be_pinned_by_default = yes
}
