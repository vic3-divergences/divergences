je_amazon = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		is_player = yes
		any_scope_state = {
			state_region = s:STATE_AMAZONIE_PROFONDE #DVG
			market = {
				owner = ROOT
			}
		}
	}

	possible = {
		has_technology_researched = rubber_mastication
		has_technology_researched = railways
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_AMAZONIE_PROFONDE #DVG
				market = {
					owner = ROOT
				}
			}
			save_scope_as = amazon_state
		}
	}

	complete = {
		scope:amazon_state = {
			infrastructure > 50
			total_urbanization > 30
			#state_population = 100000
			any_scope_building = {
				is_building_type = building_rubber_plantation
				level >= 10
			}
		}
	}

	on_complete = {
		trigger_event = { id = amazon.3 }
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0 
			50 = amazon.1 # devil's railroad
			10 = amazon.2 # hell on earth
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_AMAZONIE_PROFONDE #DVG
				market = {
					owner = ROOT
				}
			}
		}
	}

	weight = 10
}
