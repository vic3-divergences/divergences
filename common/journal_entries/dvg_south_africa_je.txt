	}

			NOT = { has_law = law_type:law_autocracy }

				has_state_in_state_region = STATE_NORTHERN_CAPE
			}

		}
	}

		value = root.var:aloysius_legitimacy
	}

	goal_add_value = {
		add = 240
	}
	invalid = {
		OR = {
			NOT = {

				}

		}
	}

	}

	weight = 1000
	should_be_pinned_by_default = yes
