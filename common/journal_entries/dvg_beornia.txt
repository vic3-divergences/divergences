						owner = ROOT

						limit = { culture = cu:aimara }

					}
					
				}
			}
			if = {
				limit = { exists = c:TWT }
				
				every_scope_state = {
					limit = {
						owner = ROOT
					}
					random_scope_pop = {
						limit = { culture = cu:quechua }
						move_pop = s:STATE_CUZCO.region_state:TWT
					}
					
				}
			}
			if = {
				limit = { has_journal_entry = je_new_mercia_independent }
				
				every_scope_state = {
					limit = {
						owner = c:TWT
					}
					random_scope_pop = {
