je_reform_indies = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	group = je_group_internal_affairs

	is_shown_when_inactive = {
		OR = {
			AND = {
				exists = c:BEI
				c:BEI = THIS
			}
			AND = {
				exists = c:CHR
				c:CHR = THIS
			}
		}
	}

	possible = {
		OR = {
			has_law = law_type:law_free_trade
			has_law = law_type:law_mercantilism
			NOT = { has_law = law_type:law_land_based_taxation }
			NOT = { has_law = law_type:law_serfdom }
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		OR = {
			has_law = law_type:law_free_trade
			has_law = law_type:law_mercantilism
		}
		NOT = { has_law = law_type:law_land_based_taxation }
		NOT = { has_law = law_type:law_serfdom }
		has_law = law_type:law_slavery_banned
	}

	on_complete = {
		trigger_event = { id = east_indies.2 }
		show_as_tooltip = {
			add_journal_entry = {
				type = je_ethical_policy
			}
		}
		#set_variable = reformed_indies
	}

	fail = {
	}
	on_fail = {
	}

	current_value = {
		value = root.gdp
	}

	goal_add_value = {
		value = root.gdp
		multiply = 0.5
	}

	progressbar = yes

	weight = 100
}

je_lanfang_migration = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_foreign_affairs

	is_shown_when_inactive = {

		exists = c:LAN

		c:LAN = ROOT
	}

	possible = {
		has_technology_researched = civilizing_mission
		country_rank > rank_value:insignificant_power
		average_sol >= 10
		is_at_war = no
		is_subject = no
		OR = {
			has_law = law_type:law_racial_segregation
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural
		}
		NOR = {
			has_law = law_type:law_closed_borders
			has_law = law_type:law_isolationism
		}
	}

	immediate = {
		set_variable = {
			name = zhourao_incorporated_population
			value = 0
		}
		set_variable = {
			name = zhourao_incorporated_states
			value = 0
		}
		every_scope_state = {
			limit = {
				is_incorporated = yes
				owner = ROOT
			}
			ROOT = {
				change_variable = {
					name = zhourao_incorporated_population
					add = PREV.state_population
				}
				change_variable = {
					name = zhourao_incorporated_states
					add = 1
				}
			}
		}
		set_variable = {
			name = zhourao_migration_goal
			value = var:zhourao_incorporated_states
		}
		change_variable = {
			name = zhourao_migration_goal
			multiply = 1500000
		}
		trigger_event = { id = zhr_migration.2 popup = yes }
	}

	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = zhourao_incorporated_population
				value = 0
			}
			every_scope_state = {
				limit = {
					is_incorporated = yes
					
				}
				ROOT = {
					change_variable = {
						name = zhourao_incorporated_population
						add = PREV.state_population
					}
				}
			}
			set_variable = {
				name = zhourao_incorporated_states
				value = 0
			}
			every_scope_state = {
				limit = {
					is_incorporated = yes
				}
				ROOT = {
					change_variable = {
						name = zhourao_incorporated_states
						add = 1
					}
				}
			}
			set_variable = {
				name = zhourao_migration_goal
				value = var:zhourao_incorporated_states
			}
			change_variable = {
				name = zhourao_migration_goal
				multiply = 1500000
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = zhourao_migration_completion_tt
			var:zhourao_incorporated_population >= var:zhourao_migration_goal
			var:zhourao_incorporated_population > 0
			var:zhourao_migration_goal > 0
		}
	}

	on_complete = {
		trigger_event = { id = zhr_migration.3 popup = yes }
		show_as_tooltip = {
			add_modifier = {
				name = populated_southamerica
			}
		}
	}

	invalid = {
		#has_variable = cancel_sa_migration_var
	}

	scripted_button = zhourao_migration_button

	weight = 100

	should_be_pinned_by_default = yes
}

				save_scope_as = new_national_culture
			}
			set_variable = {
				name = lanfang_var

		}

	}
	possible = {
		is_subject = no
		has_technology_researched = nationalism
		NOT = {
			has_journal_entry = je_tionghoa_culture
		}
	}
	timeout = 600

	on_timeout = {

	weight = 100

	should_be_pinned_by_default = yes
