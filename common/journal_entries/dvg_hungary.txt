je_hungarian_crisis_old = {
	icon = "gfx/interface/icons/invention_icons/pan_nationalism.dds"
	should_be_pinned_by_default = yes

	group = je_group_crises

	immediate = {
		set_variable = {
			name = dvg_hungarian_crisis_old_hun_power
			value = 0
		}

		set_variable = {
			name = dvg_hungarian_crisis_old_pom_power
			value = 0
		}

	}

	complete = {
		custom_tooltip = {
			text = je_hungarian_crisis_old_complete_tooltip
			has_global_variable = dvg_hungarian_crisis_old_complete
		}
	}

	fail = {
		custom_tooltip = {
			text = je_hungarian_crisis_old_fail_tooltip
			has_global_variable = dvg_hungarian_crisis_old_fail
		}
	}

	on_complete = {
		custom_tooltip = je_hungarian_crisis_old_on_complete_tooltip
		hidden_effect = {
			if = {
				limit = {
					this = c:BOH
					var:dvg_hungarian_crisis_old_pom_power > 8 }
	
				every_country = {
					limit = {
						has_variable = dvg_attend_hungarian_conference
					}
					trigger_event = { id = dvg_hungary.61 days = 10 }
					
				}
	
			} else_if = {
				
				limit = {
					this = c:BOH
				}
				every_country = {
					limit = {
						has_variable = dvg_attend_hungarian_conference
					}
					trigger_event = { id = dvg_hungary.62 days = 10 }
					
				}

				c:BOH = {
					trigger_event = { id = dvg_bohemia.10 days = 72 }
					
				}
	
			}
	
			if = {
				limit = { 
					this = c:BOH
					var:dvg_hungarian_crisis_old_hun_power > 8 }
	
				every_country = {
					limit = {
						has_variable = dvg_attend_hungarian_conference
					}
					trigger_event = { id = dvg_hungary.71 days = 10 }
					
				}
	
			} else_if = {

				limit = {
					this = c:BOH
				}
	
				every_country = {
					limit = {
						has_variable = dvg_attend_hungarian_conference
					}
					trigger_event = { id = dvg_hungary.72 days = 10 }
					
				}
			}
		}
	}
}
