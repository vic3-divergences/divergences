je_nogai = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
        
    }
	immediate = {
		
	}

	possible = {
		
    }

	complete = {
		is_country_type = recognized
		is_diplomatic_play_committed_participant = no
		legitimacy > 80
		any_neighbouring_state = {
			state_region = {
				is_homeland = cu:xacitarxani
			}
		}
		has_technology_researched = nationalism
	}

    on_complete = {
		every_scope_state = {
			limit = {
				is_incorporated = yes
				culture_percent_state = {
					target = cu:kazak
					value > 0.25
				}
				OR = {
					is_in_geographic_region = geographic_region_europe
					is_in_geographic_region = geographic_region_central_asia
					is_in_geographic_region = geographic_region_middle_east
				}
				is_isolated_from_market = no
			}
			state_region = {
				add_homeland = cu:xacitarxani
			}
			
		}
	}
	invalid = {
		NOT = {
			country_has_primary_culture = cu:xacitarxani
		}
	}
	transferable = yes
	weight = 1000
	should_be_pinned_by_default = yes
}
