je_prussian_independence = {
	icon = "gfx/interface/icons/event_icons/event_prussia_independence.dds"
	
	group = je_group_historical_content
	
	is_shown_when_inactive = {
		AND = {
			exists = c:PRU
			this = c:PRU
		}
	}

	possible = {
		has_technology_researched = nationalism
	}
	immediate = {
		s:STATE_WEST_PRUSSIA = {
            add_claim = c:PRU
		}
		s:STATE_EAST_PRUSSIA = {
			add_claim = c:PRU
		}
	}
	complete = {
		AND = {
			country_or_subject_owns_entire_state_region = STATE_EAST_PRUSSIA
			country_or_subject_owns_entire_state_region = STATE_WEST_PRUSSIA
			is_subject = no
		}
	}

	on_complete = {
		trigger_event = { id = dvg_prussia.2 }
	}
	
	should_be_pinned_by_default = yes
	scripted_button = buy_indepence
	scripted_button = scandi_backing
	scripted_button = bohemian_backing
	scripted_button = prussian_nationalism
	scripted_button = prussian_loyalty
	weight = 1000
}

		AND = {
			country_or_subject_owns_entire_state_region = STATE_LATVIA
			country_or_subject_owns_entire_state_region = STATE_ESTONIA
			country_or_subject_owns_entire_state_region = STATE_COURLAND
			is_subject = no
		}
	}

	on_complete = {
		if = {
			limit = {
				c:PRU = {
					country_or_subject_owns_entire_state_region = STATE_LITHUANIA

				add_homeland = cu:balters
			}
		}
		add_modifier = {
		name = dvg_drang_nach_osten
		days = 1800
		}
	}
	
	should_be_pinned_by_default = yes
	scripted_button = courland_unity
	weight = 1000
}
je_prussia_west = {
	icon = "gfx/interface/icons/event_icons/event_prussia_destiny_west.dds"
	
	group = je_group_historical_content
	
	complete = {

			country_or_subject_owns_entire_state_region = STATE_POMERANIA

			country_or_subject_owns_entire_state_region = STATE_ANHALT

			is_subject = no

					country_or_subject_owns_entire_state_region = STATE_MECKLENBURG

					country_or_subject_owns_entire_state_region = STATE_UPPER_SILESIA
					country_or_subject_owns_entire_state_region = STATE_SILESIA
				}
			}
		}		
	}

	on_complete = {
		if = {

				add_homeland = cu:balters
			}
		}

			limit = {
				c:PRU = {
					country_or_subject_owns_entire_state_region = STATE_UPPER_SILESIA
				}
			}
			s:STATE_UPPER_SILESIA= { 
				add_homeland = cu:balters
			}
		}

					country_or_subject_owns_entire_state_region = STATE_SILESIA
				}
			}
			s:STATE_SILESIA= { 
				add_homeland = cu:balters
			}
		}
		add_primary_culture = cu:north_german
		add_modifier = {
		name = dvg_drang_nach_westen
		days = 1800
		}
	}
	
	should_be_pinned_by_default = yes
	weight = 1000
}
