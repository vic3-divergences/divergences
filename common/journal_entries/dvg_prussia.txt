je_prussian_independence = {
	icon = "gfx/interface/icons/event_icons/event_prussia_independence.dds"
	
	group = je_group_historical_content
	
	is_shown_when_inactive = {
		AND = {
			exists = c:PRU
			this = c:PRU
		}
	}

	possible = {
		has_technology_researched = nationalism
	}
	immediate = {
		s:STATE_WEST_PRUSSIA = {
            add_claim = c:PRU
		}
		s:STATE_EAST_PRUSSIA = {
			add_claim = c:PRU
		}
	}
	complete = {
		AND = {
			country_or_subject_owns_entire_state_region = STATE_EAST_PRUSSIA
			country_or_subject_owns_entire_state_region = STATE_WEST_PRUSSIA
			is_subject = no
		}
	}

	on_complete = {
		trigger_event = { id = dvg_prussia.2 }
	}
	
	should_be_pinned_by_default = yes
	scripted_button = buy_indepence
	scripted_button = scandi_backing
	scripted_button = bohemian_backing
	scripted_button = prussian_nationalism
	scripted_button = prussian_loyalty
	weight = 1000
}

je_prussian_poles = {
	icon = "gfx/interface/icons/generic_icons/national_awakening.dds"
	group = je_group_historical_content
	should_be_pinned_by_default = yes
	
	immediate = {
		law_national_supremacy = {
			save_scope_as = racist_scope
		}
		amendment_polish_autonomy = {
			save_scope_as = scope_polska
		}
	}
	complete = {
		radical_fraction = {
			value < 0.20
			culture = cu:polish
		}
		NOT = {
			c:PRU = {
				OR = {
					has_law = law_type:law_ethnostate
					has_law = law_type:law_national_supremacy
				}
			}
		}
	}
	on_complete = {
		c:PRU = { add_primary_culture = cu:polish }
		add_loyalists = { value = 0.1 culture =  cu:polish}
		custom_tooltip = {
			text = poles_happy_tt
			set_global_variable = pru_poles_happy
		}
	}
	fail = {
		OR = {
			AND = {
				has_law = law_type:law_national_supremacy
				NOT = { 
					active_law:lawgroup_citizenship = {
						has_amendment = amendment_type:amendment_polish_autonomy
					}
				}
			}
			has_law = law_type:law_ethnostate
		}
	}
	on_fail = {
	add_radicals = {value = 0.25 culture = cu:polish}
	}
}
