je_lou_insurgency = {
	group = je_group_internal_affairs

	is_shown_when_inactive = {
		any_scope_state = {
			state_region = s:STATE_LOUISIANE_ROYALE
		}
		exists = c:PLA
		c:PLA = ROOT
	}
	possible = {
		has_global_variable = lou_to_liberate
	}
	immediate = {
		set_variable = {
			name = lou_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_LOUISIANE_ROYALE
						turmoil > 0.33
						radical_fraction = {
						value > 0.33
					}
					}
					var:lou_var < 20
				}
				change_variable = {
					name = lou_var
					add = 1
				}
			}
			if = {
				limit = {
					has_variable = diverted_troops_1
					var:lou_var < 20

				}
				change_variable = {
					name = lou_var
					add = 1
				}
			}
			if = {
				limit = {
					has_variable = diverted_troops_2
					var:lou_var < 20

				}
				change_variable = {
					name = lou_var
					add = 2
				}
			}
			if = {
				limit = {
					institution_investment_level = {
						institution = institution_police
						value > 0
					}
					var:lou_var < 20

				}
				change_variable = {
					name = lou_var
					subtract = 1
				}
			}
			if = {
				limit = {
					institution_investment_level = {
						institution = institution_home_affairs
						value > 0
					}
					var:lou_var > 0

				}
				change_variable = {
					name = lou_var
					subtract = 1
				}
			}
			if = {
				limit = {
					is_ai = yes

				}
				change_variable = {
					name = lou_var
					add = 2
				}
			}
		}
		
	}
	complete = {
		OR = {
			scope:journal_entry = {
				is_goal_complete = yes
			}
			is_diplomatic_play_committed_participant = yes
		}
		OR = {
			has_variable = diverted_troops_1
			has_variable = diverted_troops_2
		}
	}
	on_complete = {
		if = {
			limit = {
				has_variable = diverted_troops_1
			}

			trigger_event = {
				id = dvg_belgium.22
			}
			remove_variable = diverted_troops_1
		}
		if = {
			limit = {
				has_variable = diverted_troops_2
			}

			trigger_event = {
				id = dvg_belgium.23
			}
			remove_variable = diverted_troops_2

		}
		
	}

	fail = {
		OR = {
			AND = {
				is_diplomatic_play_enemy_of = c:BLC
				NOT = {
					has_variable = diverted_troops_1
				}
				has_variable = diverted_troops_2
			}
			AND = {
				has_law = law_type:law_secret_police
				institution_investment_level = {
					institution = institution_home_affairs
					value > 1
				}
			}
			AND = {
				NOT = {
					has_law = law_type:law_no_police
				}
				institution_investment_level = {
					institution = institution_police
					value > 3
				}
			}
		}
	}
	on_fail = {
		trigger_event = {
			id = dvg_belgium.20
		}
	}

	timeout = 2555

	progressbar = yes

	current_value = {
		value = root.var:lou_var
	}
	goal_add_value = 20

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_LOUISIANE_ROYALE
			}
		}
	}

}
