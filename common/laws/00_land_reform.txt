law_serfdom = {
	group = lawgroup_land_reform

	icon = "gfx/interface/icons/law_icons/serfdom.dds"

	progressiveness = -50

	is_visible = {
		NOR = {
			c:AUS ?= this # Austria and the Danubian Principalities have manorialism instead
			AND = {
				exists = c:AUS
				is_subject_of = c:AUS
				c:AUS ?= { has_law = law_type:law_manorialism }
			}
			country_has_primary_culture = cu:romanian
		}
	}

	disallowing_laws = {
		law_interventionism
		law_laissez_faire
		law_command_economy
	}

	on_enact = {
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_commercialized_agriculture
					has_law_or_variant = law_type:law_homesteading
				}
			}
			custom_tooltip = farmers_pb_ig_shift_effect_negative_tt
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_commercialized_agriculture
					has_law_or_variant = law_type:law_collectivized_agriculture
				}
			}
			custom_tooltip = rural_laborers_tu_ig_shift_effect_negative_tt
		}
	}

	modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.5
		state_peasants_internal_migration_disallowed_bool = yes
		state_peasants_mass_migration_disallowed_bool = yes
	}

	# AI should never try to bring back serfdom
	ai_will_do = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
