# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law

law_autocracy = {
	group = lawgroup_distribution_of_power

	icon = "gfx/interface/icons/law_icons/autocracy.dds"

	progressiveness = -100

	# Has major legitimacy issues with parliamentary republic, they should be oligarchy instead
	disallowing_laws = {
		law_parliamentary_republic
	}

	is_visible = {
		NOR = { # Replaced by Neo-Absolutism and Bakufu
			has_law = law_type:law_neo_absolutism
			has_law = law_type:law_bakufu
		}
	}

	on_activate = {
		scripted_effect_parties_disappearence = yes
		cleanup_voting_strength_modifiers = yes
	}

	modifier = {
		country_legitimacy_headofstate_add = 30
		country_legitimacy_govt_total_clout_add = 120
		country_legitimacy_ideological_incoherence_mult = 0.5
		country_aristocrats_pol_str_mult = 0.5
		country_officers_pol_str_mult = 0.5
		country_authority_add = 200
	}

	can_enact = {}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				has_law = law_type:law_autocracy
				OR = {
					AND = {
						power_bloc ?= { has_identity = identity:identity_ideological_union }
						has_strategy = ai_strategy_conservative_agenda
					}
					has_strategy = ai_strategy_reactionary_agenda
				}
				scope:target_country = {
					NOR = { # Is their law different enough from what we want to impose?
						has_law_or_variant = law_type:law_oligarchy
						has_law_or_variant = law_type:law_landed_voting
					}
				}
			}

			add = base_impose_law_weight
		}
	}
}

law_bakufu = {
	group = lawgroup_distribution_of_power
	parent = law_autocracy

	icon = "gfx/interface/icons/law_icons/bakufu.dds"

	progressiveness = -100

	unlocking_laws = {
		law_monarchy
		law_social_monarchy
	}

	is_visible = {
		c:JAP ?= this
		NOT = {
			has_law = law_type:law_autocracy
		}
	}

	on_activate = {
		scripted_effect_parties_disappearence = yes
		if = {
			limit = {
				country_has_monarchy_law = yes
				game_date > 1836.1.1
				c:JAP ?= this
				NOR = { # Revolter tags are handled a different way
					is_revolutionary = yes 
					is_secessionist = yes
				}
			}
			custom_tooltip = {
				text = shogun_returns_to_throne_tt
				country_generate_japanese_shogun = yes
			}
		}
		hidden_effect = {
			country_assess_daimyo_alignments = yes
		}
	}

	on_deactivate = {
		if = {
			limit = {
				country_has_monarchy_law = yes
				country_has_non_monarchy_law = no # To handle an execution order issue
			}
			custom_tooltip = {
				text = emperor_returns_to_throne_tt
				country_restore_japanese_emperor = yes
			}
			hidden_effect = {
				country_assess_daimyo_alignments = yes
			}
		}
	}

	modifier = {
		country_legitimacy_headofstate_add = 30
		country_legitimacy_govt_total_clout_add = 120
		country_legitimacy_ideological_incoherence_mult = 0.5
		country_legitimacy_govt_size_add = 1
		country_aristocrats_pol_str_mult = 0.25
		country_officers_pol_str_mult = 0.5
		country_soldiers_pol_str_mult = 0.25
		country_authority_add = 200
	}

	can_impose = {
		always = no
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
	}

	ai_impose_chance = {
		value = 0
	}
}

	can_enact = {
		OR = {

							has_party = yes
						}
					}
				}
			}
			custom_tooltip = {
				text = sps_no_party_tt
				country_has_voting_franchise = no
			}
		}

		interest_group_in_government_attraction_mult = 0.20
		country_electoral_confidence_impact_mult = -1.0
	}

	on_activate = {
		set_only_legal_party_from_ig = ruler.interest_group
		custom_tooltip = {
			text = sps_government_reshuffle_tt
			every_interest_group = {
				limit = {
					is_in_government = yes
					NOT = { #Removes all IGs from government that aren't in your party
						has_party = yes
					}
				}
				remove_ruling_interest_group = yes
			}
			every_interest_group = {
				limit = { #Adds all IGs that are in your party to the government
					is_in_government = no
					has_party = yes
				}
				add_ruling_interest_group = yes
			}

			limit = {
				has_law = law_type:law_monarchy

			remove_variable = fatherland_front_var
		}
	}

		exists = ruler
		ruler = {
			interest_group ?= {
				has_party = yes
			}
			law_stance = {
				law = law_type:law_single_party_state
				value > neutral
			}
		}
	}
	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			AND = { # modification of the standard trigger, only other single-party states countries can impose single-party states on their subjects
				has_law = scope:law
				can_impose_law_default = yes
			}
		}
	}

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
