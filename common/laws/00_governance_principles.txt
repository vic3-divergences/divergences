law_monarchy = {
	group = lawgroup_governance_principles
	
	icon = "gfx/interface/icons/law_icons/monarchy.dds"
	
	progressiveness = 0
	
	can_enact = {
		NOT = { is_subject_type = subject_type_puppet }
		NOT = { is_subject_type = subject_type_dominion }
		NOT = { is_subject_type = subject_type_vassal }
	}

	disallowing_laws = {
		law_anarchy
	}

	modifier = {
		country_legitimacy_headofstate_add = 10
		interest_group_ig_landowners_pol_str_mult = 0.25
		country_authority_add = 100	
	}
	
	possible_political_movements = {
		law_republic
		law_theocracy
	}
	
	pop_support = {
		value = 0	
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = {
					OR = {
						is_pop_type = aristocrats
						is_pop_type = peasants
					}
				}
				value = 0.33
			}
		}
	}
	
	ai_will_do = {
		exists = ruler
		ruler = {
			has_monarchist_ideology = yes
		}
	}
}

	progressiveness = -50
	
	can_enact = {
		NOT = { is_subject_type = subject_type_puppet }
		NOT = { is_subject_type = subject_type_dominion }
		NOT = { is_subject_type = subject_type_vassal }
	}
	
	disallowing_laws = {
		law_anarchy
	}

	modifier = {
		country_legitimacy_headofstate_add = 10
		interest_group_ig_devout_pol_str_mult = 0.25
		country_authority_add = 100	
	}
	
	possible_political_movements = {
		law_monarchy
		law_republic
	}
	
	on_activate = {
		activate_law = law_type:law_autocracy
		activate_law = law_type:law_independent_executive
		activate_law = law_type:law_no_legislature
		activate_law = law_type:law_no_voting

	pop_support = {

		add = {
			desc = "POP_CLERGYMEN"
			if = {
				limit = { 
					is_pop_type = clergymen
				}
				value = 0.33
			}
		}
	}
	
	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_theocrat
		}
	}
}

law_republic = {
	group = lawgroup_governance_principles
	
	icon = "gfx/interface/icons/law_icons/presidential_republic.dds"
	
	progressiveness = 100
	
	can_enact = {
		NOT = { is_subject_type = subject_type_puppet }
		NOT = { is_subject_type = subject_type_dominion }
		NOT = { is_subject_type = subject_type_vassal }
	}

	unlocking_technologies = {
		democracy
	}
	
	modifier = {
		country_legitimacy_govt_total_votes_add = 10
		#country_legitimacy_ideological_incoherence_mult = -0.1
		state_migration_pull_mult = 0.1
	}
	
	possible_political_movements = {
		law_monarchy
		law_theocracy
	}
	
	on_activate = {
		#hidden_effect = {
			if = {
				limit = {
					any_scope_character = {
						is_heir = yes
					}
				}
				random_scope_character = {
					limit = {
						is_heir = yes
					}
					kill_character = yes
				}
			}
			if = {
				limit = {
					any_scope_character = {
						is_ruler = yes
					}
				}
				random_scope_character = {
					limit = {
						is_ruler = yes
					}
					kill_character = yes
				}
			}
		#}
	}
	
	pop_support = {
		value = 0
		add = {
			desc = "POP_MIDDLE_STRATA"
			if = {
				limit = { 
					strata = middle
				}
				value = 0.05
			}
		}
	}
	
	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_republican_leader
		}
	}
}
