	}

	}

	potential = {
		country_has_primary_culture = cu:russian
        country_has_primary_culture = cu:north_russian
        country_has_primary_culture = cu:byelorussian
	}

	possible = {
		NOT = { has_law = law_type:law_laissez_faire }
		any_scope_state = {
			is_incorporated = yes

				is_building_type = building_food_industry
				level >= 5
			}
		}
	}

	prosperity_modifier = {
		state_radicals_from_political_movements_mult = -0.15
	}

	ai_will_do = {
		NOT = { has_law = law_type:law_laissez_faire }

			OR = {
				state_region = s:STATE_DVG_INGRIA
				state_region = s:STATE_DVG_MOSCOW
                state_region = s:STATE_DVG_NOVGOROD
			}
		}
	}

	ai_construction_targets = {
		building_food_industry = {
			level = 5
			state_trigger = {
				OR = {
				state_region = s:STATE_DVG_INGRIA
				state_region = s:STATE_DVG_MOSCOW

				}
			}
		}
	}

	ai_weight = {
		value = 3 # Higher base value for flavored companies
	}
}
