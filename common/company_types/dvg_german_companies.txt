dvg_company_schlesag = {
    icon = "gfx/interface/icons/company_icons/dvg_icons/schlesag.dds"
	background = "gfx/interface/icons/company_icons/company_backgrounds/comp_illu_mining.dds"

    flavored_company = yes

	preferred_headquarters = { STATE_SILESIA }

    building_types = {
		building_coal_mine
        building_iron_mine
        building_sulfur_mine
	}

    extension_building_types = {
		
        building_lead_mine
	}

    attainable = {
		hidden_trigger = { # Hide attainable if it's always fulfilled by possible
			any_scope_state = {
				OR = {
					state_region = s:STATE_SILESIA
					state_region = s:STATE_LOWER_SILESIA
				}
			}
		}
	}

    possible = {
		any_scope_state = {
			OR = {
				state_region = s:STATE_SILESIA
				state_region = s:STATE_LOWER_SILESIA
			}
			is_incorporated = yes
			any_scope_building = {
				OR = {
					is_building_type = building_coal_mine
					is_building_type = building_iron_mine
					is_building_type = building_sulfur_mine
					is_building_type = building_lead_mine
				}
				level >= 10
			}
		}
	}


    prosperity_modifier = {
		building_steel_mills_throughput_add = 0.1
        interest_group_ig_industrialists_pop_attraction_mult = 0.1
	}

    ai_will_do = {
		any_scope_state = {
			state_region = s:STATE_SILESIA
			is_incorporated = yes
		}
	}

	ai_construction_targets = {
		building_coal_mine = {
			level = 20
			state_trigger = {
				state_region = s:STATE_SILESIA
			}
		}
		building_iron_mine = {
			level = 20
			state_trigger = {
				state_region = s:STATE_SILESIA
			}
		}
		building_sulfur_mine = {
			level = 5
			state_trigger = {
				state_region = s:STATE_SILESIA
			}
		}
        building_lead_mine = {
			level = 5
			state_trigger = {
				state_region = s:STATE_SILESIA
			}
		}
	}

	ai_weight = {
		value = 10 # Higher base value for flavored companies
	}
}
