# Generic
ai_strategy_maintain_power_balance = {
	icon = "gfx/interface/icons/ai_strategy_icons/maintain_power_balance.dds"

	type = diplomatic

	will_form_power_bloc = {
		global_country_ranking = 1
	}

	undesirable_infamy_level = {
		value = 25

		if = {
			limit = {
				has_technology_researched = pan-nationalism
			}
			multiply = 1.5
		}
		else_if = {
			limit = {
				has_technology_researched = nationalism
			}
			multiply = 1.25
		}
	}

	unacceptable_infamy_level = {
		value = 50

		if = {
			limit = {
				has_technology_researched = pan-nationalism
			}
			multiply = 1.5
		}
		else_if = {
			limit = {
				has_technology_researched = nationalism
			}
			multiply = 1.25
		}
	}

	ideological_opinion_effect_mult = {
		value = 1.5
	}

	wargoal_maneuvers_fraction = {
		value = -0.1
	}

	diplomatic_play_neutrality = {
		value = 0
	}

	diplomatic_play_boldness = {
		value = -25
	}

	recklessness = {
		value = -0.25
	}

	aggression = {
		value = 0.25

		if = {
			limit = {
				NOT = { is_country_type = unrecognized }
				scope:target_country = { is_country_type = unrecognized }
				has_technology_researched = colonization
				institution_investment_level = {
					institution = institution_colonial_affairs
					value >= 1
				}
			}
			add = 0.25

			if = {
				limit = {
					power_bloc ?= { has_principle = principle:principle_colonial_offices_3 }
				}
				add = 0.5
			}

			if = {
				limit = { has_technology_researched = civilizing_mission }
				add = 0.5

				if = {
					limit = { has_strategic_adjacency = scope:target_country }
					add = 0.5
				}
			}
		}

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
			}
			multiply = 0.1
		}

		# More aggressive against ideological enemies, a bit less aggressive against ideological allies
		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = -0.1
			add = 1
			max = 2.0
			min = 0.5
		}
	}

	building_group_weights = {
		bg_technology = 1.25
	}

	secret_goal_weights = {
		protect = 1.5
		dominate = 1.5
		reconcile = 1.5
	}

	wargoal_weights = {
		regime_change = 3.0
		humiliation = 2.0
		liberate_subject = 1.5
		liberate_country = 1.5
		force_nationalization = 0.25
	}

	possible = {
		is_subject = no
	}

	weight = {
		value = 10

		if = {
			limit = { country_rank < rank_value:major_power }
			add = 10
		}
		# dvg
		# if = { # Metternich System

ai_strategy_territorial_expansion = {
	icon = "gfx/interface/icons/ai_strategy_icons/territorial_expansion.dds"

	type = diplomatic

	will_form_power_bloc = {
		OR = {
			global_country_ranking <= 3
			AND = {
				country_rank >= rank_value:great_power
				has_technology_researched = nationalism
			}
		}
	}

	undesirable_infamy_level = {
		value = 50

		if = {
			limit = {
				has_technology_researched = pan-nationalism
			}
			multiply = 1.5
		}
		else_if = {
			limit = {
				has_technology_researched = nationalism
			}
			multiply = 1.25
		}
	}

	unacceptable_infamy_level = {
		value = 100
	}

	ideological_opinion_effect_mult = {
		value = 0.75
	}

	wargoal_maneuvers_fraction = {
		value = 0.15
	}

	diplomatic_play_neutrality = {
		value = 25
	}

	diplomatic_play_boldness = {
		value = 25
	}

	recklessness = {
		value = 0.25
	}

	aggression = {
		value = 1

		if = {
			limit = {
				NOT = { is_country_type = unrecognized }
				scope:target_country = { is_country_type = unrecognized }
			}
			add = 1
		}

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
			}
			multiply = 0.25
		}

		# More aggressive against ideological enemies, a bit less aggressive against ideological allies
		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = -0.1
			add = 1
			max = 2.0
			min = 0.5
		}
	}

	building_group_weights = {
		bg_army = 1.5
	}

	secret_goal_weights = {
		protect = 0.5
		conquer = 1.5
	}

	wargoal_weights = {
		conquer_state = 1.5
		return_state = 1.5
		force_nationalization = 0.25
	}

	secret_goal_scores = {
		conquer = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				can_reach_target_country = yes
				has_strategic_adjacency = scope:target_country
				scope:target_country.country_rank <= root.country_rank
			}
			score = {
				if = {
					limit = {
						NOT = { is_country_type = unrecognized }
						scope:target_country = { NOT = { is_country_type = unrecognized } }
					}
					add = 25
				}
				else_if = {
					limit = {
						scope:target_country = { is_country_type = unrecognized }
					}
					add = 50
				}
			}
		}
	}

	possible = {
		is_subject = no
	}

	weight = {
		value = 10
		# dvg
		# if = {
		# 	limit = {
		# 		exists = c:FRA
		# 		c:FRA = this
		# 		ruler = {
		# 			has_variable = house_bonaparte
		# 		}
		# 	}
		# 	add = 5
		# }

		# AI implications for Power Blocs: 'Territorial Expansion' strategy is more likely for Blocs with Aggressive Coordination
		if = {
			limit = {
				power_bloc ?= {
					has_principle_group = principle_group:principle_group_aggressive_coordination
				}
			}
			add = 5
		}

		# High-fervour AI is inclined to take its homelands
		if = {
			limit = {
				country_fervor_primary_culture > 10
			}
			add = {
				value = country_fervor_primary_culture
				divide = 2
			}
		}

		if = {
			limit = { ig:ig_armed_forces ?= { is_powerful = yes } }
			multiply = 1.5
		}

		if = {
			limit = { is_in_geographic_region = geographic_region_europe }
			multiply = 0.25
		}

		if = {
			limit = { has_radical_leftist_government = yes }
			multiply = 0.5
		}
	}
}

						is_in_same_power_bloc = root
					}
					power_bloc ?= {
						has_principle_group = principle_group:principle_group_exploit_members
					}
				}
			}
			multiply = 5
		}

		# More aggressive against ideological enemies, a bit less aggressive against ideological allies
		multiply = {
			value = "ai_ideological_opinion(scope:target_country)"
			multiply = -0.1
			add = 1
			max = 2.0
			min = 0.5
		}
	}

	building_group_weights = {
		bg_technology = 1.25
	}

	secret_goal_scores = {
		protect = {
			score = {
				if = {
					limit = {
						scope:target_country = { has_radical_leftist_government = yes }
					}
					add = 50
				}
				if = {
					limit = {
						scope:target_country = { is_country_type = unrecognized }
					}
					add = 50
				}
			}
		}
		befriend = {
			score = {
				if = {
					limit = {
						NOT = { scope:target_country = { is_subject_of = root } }
						scope:target_country = { has_radical_leftist_government = yes }
					}
					add = 100
				}
				if = {
					limit = {
						NOT = { scope:target_country = { is_subject_of = root } }
						scope:target_country = { is_country_type = unrecognized }
					}
					add = 50
				}
			}
		}
		antagonize = {
			score = {
				if = {
					limit = {
						NOT = { scope:target_country = { is_subject_of = root } }
						has_strategic_adjacency = scope:target_country
						scope:target_country = {
							OR = {
								has_strategy = ai_strategy_economic_imperialism
								has_strategy = ai_strategy_territorial_expansion
							}
						}
					}
					add = 50
				}
				if = {
					limit = {
						NOT = { scope:target_country = { is_subject_of = root } }
						scope:target_country ?= { country_is_colonial_or_company = yes }
					}
					add = 25
				}
			}
		}
	}

	wargoal_weights = {
		regime_change = 1.5
		humiliation = 1.5
		liberate_subject = 5.0
		liberate_country = 3.0
		force_nationalization = 2.0
	}

	possible = {
		is_subject = no
		has_radical_leftist_government = yes
	}

	weight = {
		value = 10

		if = {
			limit = { country_rank >= rank_value:major_power }
			add = 25
		}
	}
}
