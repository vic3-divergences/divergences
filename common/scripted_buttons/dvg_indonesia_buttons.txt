indo_cancel_pirates_button = {
	name = "indo_cancel_pirates_button"
	desc = "indo_cancel_pirates_button_desc"

	visible = {
		has_journal_entry = je_indopiracy_send	
		has_variable = pirate_sender	

	}

	possible = {
		navy_size >=2
	}
	effect = {
		set_variable = pirates_cancelled
		add_radicals = {
			value = 0.20
			pop_type = soldiers
		}
		add_radicals = {
			value = 0.20
			pop_type = officers
		}
	}
}
