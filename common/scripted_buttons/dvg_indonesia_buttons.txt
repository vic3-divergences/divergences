sumatra_presidency_button = {
	name = "sumatra_presidency_button"
	desc = "sumatra_presidency_button_desc"

	visible = { 
		any_scope_state = {
			state_region = {
				is_homeland = cu:batak
			}
		}
		NOT = {
			exists = c:SMT
		}
	}

	possible = {
		top_overlord = {
			relations:root >= relations_threshold:amicable
		}
		any_scope_state = {
			state_region = s:STATE_SOUTH_SUMATRA
		}
		any_scope_state = {
			state_region = s:STATE_NORTH_SUMATRA
		}
	}

	effect = {
		change_tag = SMT
		top_overlord = {
			create_diplomatic_pact = {
			country = ROOT
			type = dominion
			}
		}
		if = {
			limit = {
				NOT = {
					owns_entire_state_region = STATE_SOUTH_SUMATRA
				}
			}
			top_overlord = {
			
			every_subject_or_below = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_SOUTH_SUMATRA
						is_capital = no
					}
				}
				random_scope_state = {
					limit = {
						state_region = s:STATE_SOUTH_SUMATRA
						is_capital = no
					}
					set_state_owner = ROOT
				}
			}
		}
		}
		
	}

	ai_chance = {
		value = 100
		 		 
	 }	
}
