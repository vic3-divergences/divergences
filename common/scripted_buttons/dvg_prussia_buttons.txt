
buy_indepence = {
	name = "buy_indepence"
	desc = "buy_indepence_desc"

	visible = { always = yes }

	possible = {
		is_diplomatic_play_committed_participant = no
		is_at_war = no
		is_subject_of = c:POL
		gold_reserves >= 200000
		liberty_desire < 10
		c:POL = {
			has_attitude = {
				who = c:PRU
				attitude = protective
			}
		}
	}

	ai_chance = {
		value = 0
	}
	effect = {
		c:PRU = {
			make_independent = yes
			add_treasury = -200000
			set_owes_obligation_to = {
				country = c:POL
				setting = yes
			}
		}
		c:POL = {
			add_treasury = 200000
			every_scope_state = {
				limit = {
					state_region = {
						is_homeland = cu:balters
					}
				}
				set_state_owner = c:PRU
			}
		}
	}
}
