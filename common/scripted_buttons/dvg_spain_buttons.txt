button_integrate_new_spain = {
	name = "button_integrate_new_spain"
	desc = "button_integrate_new_spain_desc"

	visible = {
		NOT = {
			has_variable = button_integrate_new_spain_var
		}
	}

	possible = {
		country_or_subject_owns_entire_state_region = STATE_MEXICO
		sr:region_mexico = {
			any_scope_state = {
				OR = {
					owner = root
					owner = {
						is_subject_of = root
						liberty_desire <= 10
					}
					owner = {
						is_country_type = decentralized
					}
				}
				percent >= 0.7
			}
		}
		custom_tooltip = {
			text = new_world_colonies_cooldown
			NOR = {
				has_variable = integrating_new_spain_var
				has_variable = integrating_new_grenada_var
				has_variable = integrating_new_toledo_var
				has_variable = integrating_new_portugal_var
				has_variable = integrating_san_salvador_var
				has_variable = integrating_caribbean_var
			}
		}
	}

	effect = {
		set_variable = button_integrate_new_spain_var
		set_variable = {
			name = integrating_new_spain_var
			days = very_short_modifier_time
		}
		if = {
			limit = { exists ?= c:GCO }
			annex_with_incorporation = c:GCO
		}
		else_if = {
			limit = { exists ?= c:MEX }
			annex_with_incorporation = c:MEX
		}
		add_primary_culture = cu:mexican
		cu:mexican = {
			remove_tradition_trait = gran_colombian
		}
		if = {
			limit = {
				NOT = { exists ?= c:LUS }
			}
			add_primary_culture = cu:lusitanian
			cu:lusitanian = {
				remove_tradition_trait = gran_colombian
			}
		}
	}

	ai_chance = {
		value = 10
	}
}
