je_hre_unify_empire = {
    name = "je_hre_unify_empire_title"
    desc = "je_hre_unify_empire_desc"


    visible = { this = c:BOH }

    possible = {
        custom_tooltip = {
            text = dvg_hre_stability_full
            var:je_holy_roman_empire_stability_var >=10
        }
        country_or_subject_owns_entire_state_region = STATE_MECKLENBURG
        country_or_subject_owns_entire_state_region = STATE_POMERANIA

        any_country = {
            filter = {
                OR = {
                    country_has_primary_culture = cu:north_german
                    country_has_primary_culture = cu:south_german
                    country_has_primary_culture = cu:rheinish
                }
            }

            is_in_same_power_bloc = c:BOH
            percent >= 1
        }

        custom_tooltip = {
            text = je_hre_unify_empire_no_BRG_controlled_states
            any_state_region = {
                filter = {
                    OR = {
                        is_homeland = cu:north_german
                        is_homeland = cu:south_german
                        is_homeland = cu:rheinish
                    }
                }


                NOT = {
                    any_scope_state = {
                        owner = c:BRG

                    }
                }
                percent >= 1
            }
        }




    }

    effect = {


        every_country = {
            limit = {
                is_in_same_power_bloc = c:BOH
                NOT = {
                    this = c:BOH
                }
            }

            c:BOH = { annex_with_incorporation = prev }
        }

        change_tag = HLR
    }

    ai_chance = {
        base = 0
    }
}
