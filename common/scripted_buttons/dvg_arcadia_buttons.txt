
golden_law_button = {
	name = "golden_law_button"
	desc = "golden_law_button_pla__desc"

	visible = { always = yes }

	possible = {
		has_law = law_type:law_legacy_slavery
		c:PLA ?= {
			OR = {
				AND = {
					any_scope_character = {
						OR = {
							is_heir = yes
							is_ruler = yes
						}
						OR = {
							has_ideology = ideology:ideology_abolitionist
							has_ideology = ideology:ideology_humanitarian
							has_ideology = ideology:ideology_humanitarian_royalist
						}
						age >= 20
					}
				}
				ig:ig_landowners = {
					is_in_government = no
					ig_clout < 0.25
				}
			}
			is_enacting_law = law_type:law_slavery_banned
			custom_tooltip = {
				text = golden_law_button_tt_2
				trigger_if = {
					limit = {
						is_enacting_law = law_type:law_slavery_banned
					}
					enactment_phase >= 1
				}
				trigger_else = {
					always = no
				}
			}
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = { 
				ig:ig_landowners = {
					is_in_government = no
				}
			}
			add = 2
		}
		if = {
			limit = { 
				ig:ig_landowners = {
					is_in_government = no
					is_powerful = no
				}
			}
			add = 4
		}
		if = {
			limit = { 
				any_interest_group = {
					is_in_government = yes
					OR = {
						leader = {
							OR = {
								has_ideology = ideology:ideology_abolitionist
								has_ideology = ideology:ideology_humanitarian
								has_ideology = ideology:ideology_humanitarian_royalist
							}
						}
						OR = {
							law_stance = {
								law = law_type:law_slave_trade
								value = strongly_disapprove
							}
							law_stance = {
								law = law_type:law_legacy_slavery
								value = strongly_disapprove
							}
						}
					}
				}
			}
			add = 7
		}
		if = {
			limit = { 
				OR = {
					is_at_war = yes
					any_revolution_ongoing = yes
				}
			}
			add = -1000
		}
	}

	effect = {
		
		activate_law = law_type:law_slavery_banned
		ig:ig_landowners = {
			add_modifier = {
				name = outraged_over_golden_law_modifier
				days = normal_modifier_time
			}
		}
		
		
	}
}

			}

		add_modifier = {
			name = restricted_slave_trade_modifier

	}
