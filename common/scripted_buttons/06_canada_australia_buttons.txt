

je_australia_unite_aus_button = {
	name = "je_australia_unite_aus_button"
	desc = "je_australia_unite_aus_button_desc"

	visible = {
		always = yes
	}

	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		NOT = { has_modifier = australia_test_modifier }
		AND = {
				top_overlord ?= {
				relations:root >= relations_threshold:amicable
				}
				custom_tooltip = {
				text = australie_confederation_tt

				any_country = {
				is_at_war = no
				is_player = no
				is_adjacent_to_country = ROOT
				country_has_primary_culture = cu:australian
				this.gdp < ROOT.gdp
				relations:ROOT >= relations_threshold:amicable
				OR = {
					is_subject = no
					is_subject_of = ROOT
					AND = {
						ROOT = { is_subject = yes }
						is_subject_of = ROOT.top_overlord
					}
				}
			}
				}
		}
			
			
		
		
	}

	cooldown = { days = short_modifier_time }

	effect = {
		custom_tooltip = unite_australia_tt

		trigger_event = { id = can_aus.3 popup = yes }


		
		add_modifier = {
			name = australia_test_modifier
			days = short_modifier_time
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				has_technology_researched = nationalism
			}
			add = 100
		}
	}
}

			}
		}
		
	}

	cooldown = {
		months = 6
	}
	effect = {
		
		trigger_event = {
			id = dvg_australie.20
		}
	}
	ai_chance = {
		value = 100
	}
}

enfranchise_indo_button = {
	name = enfranchise_indo_button
	desc = enfranchise_indo_button_desc

	visible = {
		NOT = {
			country_has_primary_culture = cu:indo_australien
		}
		any_scope_pop = {
			has_culture = cu:indo_australien
		}
	}

	possible = {
		country_average_cultural_acceptance = {
			target = cu:indo_australien
			value >= acceptance_status_3
		}
		any_scope_state = {
			has_cultural_community = cu:indo_australien
		}
		has_technology_researched = pan-nationalism
	}

	selected = {
	}

	effect = {
		add_primary_culture = cu:indo_australien
		custom_tooltip = saf_struggle_for_the_highveld_griqua_confederation_tt

		every_country = {
			limit = {
				NOT = {
					ROOT = this
					is_country_type = decentralized
					country_has_primary_culture = cu:australian
				}
				country_has_primary_culture = cu:australian
			}
			change_relations = {
				country = ROOT
				value = -30
			}
		}
		cu:australian = {
			add_fervor = 5
		}
		every_scope_state = {
				limit = {
				has_cultural_community = cu:indo_australien
				NOT = {
					any_scope_state = {
						state_region = {
							is_homeland = cu:indo_australien
						}
					}
				}
				}
				add_homeland = cu:indo_australien
		}

	}

	ai_chance = {
		value = 1
	}
}

pyard_treaty_button = {
	name = pyard_treaty_button
	desc = pyard_treaty_button_desc

	visible = {

		exists = c:NJY
		exists = c:YLN
	}

	possible = {
		is_subject = yes
		OR = {
			AND = {
				c:NJY = this
				c:YLN = {
				is_ai = yes
				relations:root >= relations_threshold:amicable
				top_overlord = {
							THIS = root.top_overlord
						}
				this.gdp < root.gdp
				OR = {
					has_attitude = {
					who = ROOT
					attitude = cooperative
					}
					liberty_desire < 25
				}
				top_overlord = {
							THIS = root.top_overlord
				}
				}
			}
			AND = {
				c:YLN = this
				c:NJY = {
				is_ai = yes
				relations:root >= relations_threshold:amicable
				top_overlord = {
							THIS = root.top_overlord
						}
				this.gdp < root.gdp
				OR = {
					has_attitude = {
					who = ROOT
					attitude = cooperative
					}
					liberty_desire < 25
				}
				top_overlord = {
							THIS = root.top_overlord
				}
				}
			}
		}
		
	}

	selected = {

	}

	effect = {
		if = {
			limit = {
				country_average_cultural_acceptance = {
					target = cu:aborigine_french
					value >= acceptance_status_3
					}
			}
			add_primary_culture = cu:aborigine_french
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_racial_segregation
					has_law = law_type:law_ethnostate
				}	
			}
			activate_law = law_type:law_subjecthood
		}

		if = {
			limit = {
				has_law = law_type:law_national_supremacy
			}
			activate_law = law_type:law_cultural_exclusion
		}
		if = {
			limit = {
				c:NJY = this
			}
			change_relations = {
			country = c:YLN
			value = 30
			}
		}
		if = {
			limit = {
				c:YLN = this
			}
			change_relations = {
			country = c:NJY
			value = 30
			}
		}
		
	}
	ai_chance = {
		value = 1
	}
}

	}

				}
			}
		}

	}
