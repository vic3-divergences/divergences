			je:je_amazonie_debt ?= { remove_modifier = modifier_amazonie_independence_payments_3 }
		}
		if = {
			limit = {
				je:je_amazonie_debt ?= { has_modifier = modifier_amazonie_independence_payments_4 }
			}
			je:je_amazonie_debt ?= { remove_modifier = modifier_amazonie_independence_payments_4 }
		}
		c:DUA = {

				limit = {
					has_modifier = modifier_dua_independence_payments_3
				}
				remove_modifier = modifier_dua_independence_payments_3
			}
			if = {
				limit = {
					has_modifier = modifier_dua_independence_payments_4
				}
				remove_modifier = modifier_dua_independence_payments_4
			}
			change_relations = {
				country = c:AMA
				value = -70
			}
		}
		change_infamy = 45
		every_state_region = {
			limit = {
				OR = {
					is_homeland = cu:amazonien
					is_homeland = cu:antarctican

			}

						in_default = yes
						has_revolution = yes
					}
				}
			}
			add = 10
		}
		if = {
			limit = {
				OR = {
					AND = {
						army_size > c:DUA.army_size
						navy_size > c:DUA.navy_size
					}
					any_scope_ally = {
						army_size > c:DUA.army_size
						navy_size > c:DUA.navy_size
					}
				}
			}
			add = 10
		}

		if = {
			limit = {
				country_rank >= rank_value:major_power

			add = 2
		}

		if = {
			limit = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}
			multiply = 0
		}
	}

je_ploumanao_ama_raid_back = {
	name = "je_ploumanao_ama_raid_back"
	desc = "je_ploumanao_ama_raid_back_desc"

	visible = {
		is_player = yes
		has_journal_entry = je_ama_raided
		c:PLU ?= {
			is_country_alive = yes
			is_player = no
			is_subject = no
		}
	}

	ai_chance = { #Mon event handles balance. This event is only for players.
		value = 0
	}

	cooldown = { years = 3 }

	possible = {
		hidden_trigger = {
			is_player = yes
		}
		NOR = {
			custom_tooltip = {
				text = counterraided_recently_tt
				has_variable = counterraided_recently
			}
			is_at_war = yes
			is_diplomatic_play_committed_participant = yes
		}
		c:PLU ?= {
			NOR = {
				is_at_war = yes
				is_diplomatic_play_committed_participant = yes
				is_diplomatic_play_undecided_participant = yes
				is_player = yes #if both are players, event handles balance.
			}
		}
		custom_tooltip = {
			text = je_ploumanao_ama_raid_back_condition_tt
			c:PLU ?= {
				is_country_alive = yes
				raiding_income > 0
			}
		}
	}

	effect = {
		if = {
			limit = {
				exists = c:EQX
			}
			create_diplomatic_catalyst = {
			type = catalyst_event_negative
			target = c:EQX
			}
			change_relations = {
				country = c:EQX
				value = -30
			}
		}
		change_infamy = 7
		custom_tooltip = {
			text = je_ploumanao_ama_raid_back_effects_tt
			c:PLU ?= {
				destroy_barracks = yes
			}
		}
		every_scope_state = {
			limit = {
				has_modifier = raid_target
			}
			remove_modifier = raid_target
			remove_global_variable = mon_raid_state
		}
		hidden_effect = {
			set_variable = {
				name = counterraided_recently
				years = 3
			}
		}
	}
}
