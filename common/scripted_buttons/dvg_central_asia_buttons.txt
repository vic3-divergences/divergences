je_unite_georgians_button = {
	name = "je_unite_georgians_button"
	desc = "je_unite_georgians_button_desc"

	visible = { always = yes }

	ai_chance = {
		value = 100
	}

	possible = {
        NOT = {
		        any_country = {
		            country_rank >= rank_value:unrecognized_regional_power
		            NOT = { has_war_with = ROOT }
		            has_interest_marker_in_region = ROOT.capital.region
		            relations:root < 0
		        }
	    }
        custom_tooltip = {
            text = je_unite_the_georgians_amicable_or_better_relations_tt
            NOR = {
            	has_variable = georgia_unification_in_motion
            	has_variable = ROM_united_principality
            }
            OR = {
				is_subject_type = subject_type_protectorate
				is_subject_type = subject_type_dominion
				is_subject_type = subject_type_tributary
				is_subject = no
			}
	        any_country = {
	        	NOT = { this = ROOT }
	        	country_has_primary_culture = cu:georgian
	            relations:root >= 25
	            OR = {
	            	AND = {
	            		exists = top_overlord
						OR = {
							is_subject_type = subject_type_protectorate
							is_subject_type = subject_type_dominion
							is_subject_type = subject_type_tributary
						}
						top_overlord = {
							THIS = root.top_overlord
						}
					}
					is_subject = no
	            }
				NOR = {
					c:AZB ?= this
					c:CAU ?= this
				}
	        }
        }
	}

	effect = {
		if = {
			limit = {
				is_subject = yes
				exists = top_overlord
			}
			set_variable = georgia_subject
        	top_overlord = {
        		trigger_event = {
	        		id = dvg_persia.22
	        	}
	        }
	    }
	    trigger_event = { id = dvg_persia.23 }
	}
}

je_kazakh_unite_button = {
	name = "je_kazakh_unite_button"
	desc = "je_kazakh_unite_button_desc"

	visible = {
		always = yes
	}

	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		NOT = { has_modifier = australia_test_modifier }
		trigger_if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:kazak
					country_has_primary_culture = cu:xacitarxani
					country_has_primary_culture = cu:sibir
				}
			}
			top_overlord ?= {
				relations:root >= relations_threshold:amicable
			}
			custom_tooltip = {
				text = unite_kazakh_je_kaz_tt
				any_neighbouring_state = {
					owner = {
						is_ai = yes
						relations:root >= relations_threshold:amicable
						OR = {
							country_has_primary_culture = cu:kazak
							country_has_primary_culture = cu:xacitarxani
							country_has_primary_culture = cu:sibir
						}
						is_subject = yes
						NOT = { THIS = ROOT }
						top_overlord = {
							THIS = root.top_overlord
						}
						this.gdp < root.gdp
					}
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = gbr_aus_relations_tt
				any_subject_or_below = {
					OR = {
						country_has_primary_culture = cu:kazak
						country_has_primary_culture = cu:xacitarxani
						country_has_primary_culture = cu:sibir
					}
					is_ai = yes
					relations:root >= relations_threshold:amicable
					count >= 2
				}
			}
		}
	}

	cooldown = { days = short_modifier_time }

	effect = {
		custom_tooltip = unite_australia_tt
		if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:australian
					country_has_primary_culture = cu:maori
					country_has_primary_culture = cu:aborigine
				}
			}
			hidden_effect = {
				trigger_event = { id = can_aus.3 popup = yes }
			}
		}
		else = {
			hidden_effect = {
				trigger_event = { id = can_aus.4 popup = yes }
			}
		}
		add_modifier = {
			name = australia_test_modifier
			days = short_modifier_time
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				has_technology_researched = pan-nationalism
			}
			add = 100
		}
	}
}
