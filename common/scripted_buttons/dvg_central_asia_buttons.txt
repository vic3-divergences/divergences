je_unite_georgians_button = {
	name = "je_unite_georgians_button"
	desc = "je_unite_georgians_button_desc"

	visible = { always = yes }

	ai_chance = {
		value = 100
	}

	possible = {
        NOT = {
		        any_country = {
		            country_rank >= rank_value:unrecognized_regional_power
		            NOT = { has_war_with = ROOT }
		            has_interest_marker_in_region = ROOT.capital.region
		            relations:root < 0
		        }
	    }
        custom_tooltip = {
            text = je_unite_the_georgians_amicable_or_better_relations_tt
            NOR = {
            	has_variable = georgia_unification_in_motion
            	has_variable = ROM_united_principality
            }
            OR = {
				is_subject_type = subject_type_protectorate
				is_subject_type = subject_type_dominion
				is_subject_type = subject_type_tributary
				is_subject = no
			}
	        any_country = {
	        	NOT = { this = ROOT }
	        	country_has_primary_culture = cu:georgian
	            relations:root >= 25
	            OR = {
	            	AND = {
	            		exists = top_overlord
						OR = {
							is_subject_type = subject_type_protectorate
							is_subject_type = subject_type_dominion
							is_subject_type = subject_type_tributary
						}
						top_overlord = {
							THIS = root.top_overlord
						}
					}
					is_subject = no
	            }
				NOR = {
					c:AZB ?= this
					c:CAU ?= this
				}
	        }
        }
	}

	effect = {
		if = {
			limit = {
				is_subject = yes
				exists = top_overlord
			}
			set_variable = georgia_subject
        	top_overlord = {
        		trigger_event = {
	        		id = dvg_persia.22
	        	}
	        }
	    }
	    trigger_event = { id = dvg_persia.23 }
	}
}
