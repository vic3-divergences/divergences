dvg_east_indies.6 = {
	type = country_event
	placement = ROOT

	title = dvg_east_indies.6.t
	desc = dvg_east_indies.6.d
	flavor = dvg_east_indies.6.f
	
	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	immediate = {
		
	}
	
	# It shall aide us immensely!
	option = {
		name = dvg_east_indies.6.a
		default_option = yes
		trigger = {
			exists = c:JAV
		}
		play_as = c:JAV
		c:JAV = {
			annex = ROOT
		}
		if = {
			limit ={
				exists = s:STATE_NOONGAR.region_state:BRG
			}
			every_scope_pop = {
				limit = {
					culture = cu:burgundian
				}
				move_pop = s:STATE_NOONGAR.region_state:BRG
			}
			every_scope_pop = {
				limit = {
					culture = cu:dutch
				}
				move_pop = s:STATE_NOONGAR.region_state:BRG
			}
		}
	}
	option = {
		name = dvg_east_indies.6.b
		trigger = {
			NOT = {
				exists = c:JAV
			}
		}
		add_loyalists = {
			value = 0.20
			culture = cu:sundanese
		}

		
		remove_primary_culture = cu:dutch
		remove_primary_culture = cu:burgundian
		if = {
			limit ={
				exists = s:STATE_NOONGAR.region_state:BRG
			}
			every_scope_pop = {
				limit = {
					culture = cu:burgundian
				}
				move_pop = s:STATE_NOONGAR.region_state:BRG
			}
			every_scope_pop = {
				limit = {
					culture = cu:dutch
				}
				move_pop = s:STATE_NOONGAR.region_state:BRG
			}
		}
		change_tag = BTN
		
	}
	option = {
		name = dvg_east_indies.6.c
		
		add_radicals = {
			value = 0.35
			culture = cu:javan
		}
		
	}
}

#Sumatra Christianzed
dvg_east_indies.7 = {
	type = country_event
	placement = ROOT

	title = dvg_east_indies.7.t
	desc = dvg_east_indies.7.d
	flavor = dvg_east_indies.7.f
	
	event_image = {
		video = southamerica_christianity
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	
	# We saved Sumatra!
	option = {
		name = dvg_east_indies.7.a
		add_loyalists = {
			value = 0.10
		}
		add_modifier = {
			name = dvg_assimilate
			months = 305
			is_decaying = yes
		}
		add_modifier = {
			name = malay_cultural_acceptance_modifier_positive
		}
		add_modifier = {
			name = batak_cultural_acceptance_modifier_positive
		}
		add_modifier = {
			name = tionghoa_cultural_acceptance_modifier_positive
		}
		if = {
			limit = {
				NOT = {
					has_technology_researched = nationalism
				}
			}
			add_technology_researched = nationalism
		}
		if = {
			limit = {
				NOT = {
					has_technology_researched = pan-nationalism
				}
				has_technology_researched = nationalism
			}
			add_technology_researched = pan-nationalism
		}
		if = {
			limit = {
				any_scope_pop = {
					has_culture = cu:indo_australien
				}
				OR = {
					has_technology_researched = pan-nationalism
					any_scope_state = {
						is_homeland = cu:indo_australien
					}
					cultural_acceptance_base = {
						target = cu:indo_australien
						value >= acceptance_status_3
					}
				}
			}
			add_primary_culture = cu:indo_australien
		}
		
		ai_chance = {
			base = 1
		}
	}

	option = {
		name = dvg_east_indies.7.b

		add_loyalists = {
			value = 0.10
		}

		add_primary_culture = cu:batak
		add_primary_culture = cu:malay
		add_primary_culture = cu:tionghoa

		if = {
			limit = {
				NOT = {
					has_technology_researched = nationalism
				}
			}
			add_technology_researched = nationalism
		}
		if = {
			limit = {
				NOT = {
					has_technology_researched = pan-nationalism
				}
				has_technology_researched = nationalism
			}
			add_technology_researched = pan-nationalism
		}
		
		ai_chance = {
			base = 1
		}
	}
}
