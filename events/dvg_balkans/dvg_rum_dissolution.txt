	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"
	duration = 1

	type = country_event
	placement = ROOT
	
	title = dvg_rum_dissolution.1.t
	desc = dvg_rum_dissolution.1.d
	flavor = dvg_rum_dissolution.1.f

	immediate = {
		change_relations = {
			country = ROOT
			value = -20
		}
		if = {
			limit = {
				NOT = {
					has_technology_researched = nationalism
				}
			}
			add_technology_researched = nationalism
		}
		if = {
			limit = {
				exists = c:SER
			}
			s:STATE_SERBIA = { 
				add_claim = c:SER
			}
		}
		if = {
			limit = {
				exists = c:CRO
			}
			every_state_region = {
				limit = {
					is_homeland = cu:croat
				}
				add_claim = c:CRO
			}
		}
		if = {
			limit = {
				exists = c:GRE
			}
			every_state_region = {
				limit = {
					is_homeland = cu:greek
				}
				add_claim = c:GRE
			}
		}
		if = {
			limit = {
				exists = c:WAL
			}
			s:STATE_WALLACHIA = {
				add_claim = c:WAL
			}
			s:STATE_DOBRUDJA = {
				add_claim = c:WAL
			}
		}
		if = {
			limit = {
				exists = c:MOL
			}
			s:STATE_MOLDAVIA = {
				add_claim = c:MOL
			}
			s:STATE_BESSARABIA = {
				add_claim = c:MOL
			}
		}
		if = {
			limit = {
				exists = c:HUN
			}
			every_state_region = {
				limit = {
					is_homeland = cu:hungarian
				}
				add_claim = c:HUN
			}
		}
		if = {
			limit = {
				exists = c:EGY
			}
			every_state_region = {
				limit = {
					is_homeland = cu:misri
				}
				add_claim = c:EGY
			}
		}
		if = {
			limit = {
				exists = c:CRI
			}
			every_state_region = {
				limit = {
					is_homeland = cu:kirim
				}
				add_claim = c:CRI
			}
		}
		if = {
			limit = {
				exists = c:MCD
			}
			s:STATE_MACEDONIA = {
				add_claim = c:MCD
			}
		}
		if = {
			limit = {
				exists = c:ALB
			}
			every_state_region = {
				limit = {
					is_homeland = cu:albanian
				}
				add_claim = c:ALB
			}
		}
		
	}

	option = {
		name = dvg_rum_dissolution.1.a
		default_option = yes


		every_subject_or_below = {
			set_country_type = unrecognized
		}


		add_modifier = {
			name = modifier_army_dissaproves
			months = 30
			is_decaying = yes
		}
		add_radicals = {
			value = 0.25
			culture = cu:turkish
		}
		
		change_tag = ERE
		
		remove_diplomatic_pact = {
			country = c:TRS
			type = puppet
		}

		remove_diplomatic_pact = {
			country = c:WAL
			type = puppet
		}

		remove_diplomatic_pact = {
			country = c:BSG
			type = puppet

		remove_diplomatic_pact = {

			type = puppet
		}

		remove_diplomatic_pact = {
			country = c:ALB
			type = puppet

		}

			country = c:GRE
			type = puppet
		}

		remove_diplomatic_pact = {
			country = c:IRQ

		}

			type = protectorate
		}
		
		remove_diplomatic_pact = {
			country = c:CIR
			type = protectorate
		}
	

		c:WRE = {
			create_diplomatic_play = {
				type = dp_annex_war
				target_country = c:GRE
				war = yes

				add_war_goal = {
					holder = c:GRE
					type = annex_country
					target_country = c:WRE
				}
				
			
			}
		}
		

		### Transferation of states!

		#s:STATE_THESSALIA.region_state:WRE = { set_state_owner = c:GRE }
		#s:STATE_ATTICA.region_state:WRE = { set_state_owner = c:GRE }
		#s:STATE_CRETE.region_state:WRE = { set_state_owner = c:GRE }
		#s:STATE_WESTERN_THRACE.region_state:WRE = { set_state_owner = c:ERE }
		
		### Set Constantinople as capital of ERE
		set_capital = STATE_WESTERN_THRACE
		
		###Claiming Thrace
		if = {
			limit = {
				exists = c:ERE
			}
			s:STATE_EASTERN_THRACE = {
				add_claim = c:ERE
			}
		}

		### Creation of Rumelia
		create_country = {
			tag = RUM
			origin = this
			state = s:STATE_MACEDONIA.region_state:WRE
			state = s:STATE_EASTERN_THRACE.region_state:WRE
			state = s:STATE_DOBRUDJA.region_state:WRE
			state = s:STATE_NORTHERN_THRACE.region_state:WRE
			state = s:STATE_BULGARIA.region_state:WRE
			state = s:STATE_WESTERN_SERBIA.region_state:WRE
		}

		### Creation of Macedonia
		create_country = {
			tag = MCD
			origin = this
			state = s:STATE_MACEDONIA.region_state:ALB
			state = s:STATE_SKOPIA.region_state:ALB
		}
		
		### Creation of Syria
		create_country = {
			tag = SYR
			origin = this
			state = s:STATE_ALEPPO.region_state:ERE
			state = s:STATE_DEIR_EZ_ZOR.region_state:ERE
			state = s:STATE_LEBANON.region_state:ERE
			state = s:STATE_PALESTINE.region_state:ERE
			state = s:STATE_SINAI.region_state:ERE
			state = s:STATE_TRANSJORDAN.region_state:ERE
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = dvg_rum_dissolution.1.b
		
		add_radicals = {
			value = 0.33
		}
		add_modifier = {
			name = modifier_army_bolstered
			months = 60
			is_decaying = yes
		}
		add_modifier = {
			name = army_resources_modifier
			months = 60
			is_decaying = yes
		}

			value = 0.20

		}
		add_loyalists = {
			value = 0.40
			pop_type = officers
		}
		ai_chance = {
			base = 1
		}	
	}

		name = dvg_rum_dissolution.1.c
		trigger = {
			any_subject_or_below = {
				has_attitude = {
					who = c:TUR
					attitude = rebellious
				}
			}
			
		}
		every_subject_or_below = {
			limit = {
				has_attitude = {
					who = c:TUR
					attitude = rebellious
				}
			}
			save_scope_as = rum_subject

			create_diplomatic_play = {
				type = dp_independence
				target_country = c:TUR
				war = no 

				add_war_goal = {
					holder = c:TUR
					type = annex_country
					target_country = scope:rum_subject 
				}
			}
		}
		ai_chance = {
			base = 1
		}	
	}
}
