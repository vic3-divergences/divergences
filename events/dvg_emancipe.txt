	duration = 3

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	immediate = { 
		
    }
	trigger = {
		
	}
	option = {	#We will keep border as is 
		name = dvg_emancipe.2.a
		default_option = yes

		add_radicals = {
			value = 0.10
			interest_group = ig:ig_intelligentsia
		}
		add_radicals = {
			value = 0.10
			interest_group = ig:ig_armed_forces
		}
		every_country = {
			limit = {
				is_in_geographic_region = geographic_region_west_africa
				NOT = {
					has_law = law_type:law_slavery_banned
				}
			}
			change_relations = {
				country = THIS
				value = 20
			}
		}

	}

	option = {	#We will press our claims further
		name = dvg_emancipe.2.b

		every_country = {
			limit = {
				OR = {
					is_in_geographic_region = geographic_region_west_africa
					has_interest_marker_in_region = sr:region_senegal
				}
				NOT = {
					has_law = law_type:law_slavery_banned
				}
			}
			change_relations = {
				country = ROOT
				value = -30
			}
			
			every_scope_state = {
				state_region = {
					add_claim = ROOT
				}
			}
		}
		create_incident = {
			country = THIS
			value = 15
		}
	}
}
