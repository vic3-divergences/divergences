namespace = subjecthood_events

subjecthood_events.1 = { # Stall - Minorities are cross
	type = country_event
	placement = root
	title = subjecthood_events.1.t
	desc = {
		first_valid = {
			triggered_desc = {
				desc = subjecthood_events.1.d_1
				trigger = {
					any_primary_culture = {
						count = 1
					}
				}
			}
			triggered_desc = {
				desc = subjecthood_events.1.d_2
				trigger = {
					any_primary_culture = {
						count = 2
					}
				}
			}
			triggered_desc = {
				desc = subjecthood_events.1.d_3
				trigger = {
					any_primary_culture = {
						count > 2
					}
				}
			}
		}
	}
	flavor = subjecthood_events.1.f
	duration = 3
	event_image = {
		video = "ip3_angry_mob_raiding_minorities"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	cooldown = { days = normal_modifier_time }

	category = enactment

	trigger = {
		is_abolishing_subjecthood = yes
		NOT = {
			is_enacting_law = law_type:law_multicultural
		}
		any_scope_culture = {
			ROOT = {
				culture_percent_country = {
					target = prev
					value >= 0.05
				}
			}
			NOT = {
				is_primary_culture_of = ROOT
			}
			ROOT = { # Between 40 and 80 acceptance
				country_average_cultural_acceptance = {
					target = prev
					value >= acceptance_status_2
				}
				country_average_cultural_acceptance = {
					target = prev
					value <= acceptance_status_4
				}
			}
		}
	}

	cancellation_trigger = {
		NOT = {
			exists = root.currently_enacting_law
		}
	}

	immediate = {
		random_primary_culture = {
			save_scope_as = relevant_primary_culture
		}
		if = {
			limit = {
				any_primary_culture = {
					count > 1
				}
			}
			random_primary_culture = {
				limit = {
					NOT = {
						scope:relevant_primary_culture ?= this
					}
				}
				save_scope_as = relevant_primary_culture_2
			}
		}
		random_scope_culture = {
			limit = {
				ROOT = {
					culture_percent_country = {
						target = prev
						value >= 0.05
					}
				}
				NOT = {
					is_primary_culture_of = ROOT
				}
				ROOT = { # Between 40 and 80 acceptance
					country_average_cultural_acceptance = {
						target = prev
						value >= acceptance_status_2
					}
					country_average_cultural_acceptance = {
						target = prev
						value <= acceptance_status_4
					}
				}
			}
			save_scope_as = relevant_minority_culture
		}
		currently_enacting_law = {
    		save_scope_as = current_law_scope
    	}
	}

	option = {
		name = subjecthood_events.1.a
		default_option = yes
		add_enactment_modifier = {
			name = modifier_angered_minorities_with_nationalism
			multiplier = "culture_percent_country(scope:relevant_minority_culture)"
		}
		add_radicals = {
			value = medium_radicals
			culture = scope:relevant_minority_culture
		}
	}

	option = {
		name = subjecthood_events.1.b
		add_enactment_modifier = {
			name = modifier_civic_nationalist_focus
			multiplier = "culture_percent_country(scope:relevant_minority_culture)"
		}
		every_primary_culture = {
			ROOT = {
				add_radicals = {
					value = small_radicals
					culture = prev
				}
			}
		}
	}
}
