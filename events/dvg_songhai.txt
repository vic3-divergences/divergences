#Failing the Songhai 

dvg_songhai.1 = { 
	type = country_event
	placement = root
	title = dvg_songhai.1.t
	desc = dvg_songhai.1.d
	flavor = dvg_songhai.1.f

	duration = 3

	event_image = {
		video = "africa_prosperous_farm"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
		
	}

	option = {
		name = dvg_songhai.1.a
		default_option = yes
		
		every_state_region = {
			limit = {
				has_claim_by = ROOT
			}
			remove_claim = ROOT
		}
		add_loyalists = {
			value = 0.15
		}
		create_incident = {
			country = THIS
			value = -10
		}
	}

	option = {
		name = dvg_songhai.1.b
		#Continue Songhai Legacy

		add_radicals = {
			value = 0.10
		}
		
		add_loyalists ={
			value = 0.15
			culture = cu:songhai
		}
		every_state_region = {
			limit = {
				is_homeland = cu:songhai
			}
			add_claim = ROOT
		}
		create_incident = {
			country = THIS
			value = 15
		}
	}
}

dvg_songhai.3 = { #Songhai Rule Collapses
	type = country_event
	placement = root
	title = dvg_songhai.3.t
	desc = dvg_songhai.3.d
	flavor = dvg_songhai.3.f

	duration = 3

	event_image = {
		video = "africa_prosperous_farm"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
	
	}
	immediate = {
		random_primary_culture = {
			limit = {
				NOT = {
					has_discrimination_trait = language_songhai
				}
			}
			save_scope_as = national_culture_scope
		}
		
	}
	#Ditch the Songhai Legacy
	option = {
		name = dvg_songhai.3.a
		default_option = yes

		remove_primary_culture = cu:songhai
		
		
		add_loyalists ={
			value = 0.15
		}
		add_radicals = {
			value = 0.20
			culture = cu:songhai
		}
	}

	option = {
		#Restore Songhai Legacy
		name = dvg_songhai.3.b
		
		
		add_radicals = {
			value = 0.10
		}
		
		add_loyalists ={
			value = 0.15
			culture = cu:songhai
		}
		every_state_region = {
			limit = {
				is_homeland = cu:songhai
			}
			add_claim = ROOT
		}
		create_incident = {
			country = THIS
			value = 15
		}
	}
}
