namespace = dvg_lanfang

dvg_lanfang.1 = {
	type = country_event
	placement = ROOT

	title = dvg_lanfang.1.t
	desc = dvg_lanfang.1.d
	flavor = dvg_lanfang.1.f

	event_image = {
		video = "gfx/event_pictures/asia_buddhism.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}

	# We are Tiongs!
	option = {
		name = dvg_lanfang.1.a
		default_option = yes

		random_scope_state = {
			limit = {
				is_capital = yes
			}
			state_region = {
				add_homeland = cu:tionghoa
			}
		}

		random_scope_pop = {
			limit = {
				culture = {
					has_discrimination_trait = tradition_sinosphere
				}
			}
			change_pop_culture = {
				target = cu:tionghoa
				value = 0.1
			}
		}
		random_scope_pop = {
			limit = {
				culture = {
					has_discrimination_trait = heritage_han
				}
			}
			change_pop_culture = {
				target = cu:tionghoa
				value = 0.1
			}
		}
		random_scope_pop = {
			limit = {
				culture = {
					has_discrimination_trait = heritage_insulindian
				}
			}
			change_pop_culture = {
				target = cu:tionghoa
				value = 0.1
			}
		}
		if = {
			limit = {
				NOT = {
					has_journal_entry = je_tionghoa_culture
				}
			}
			add_journal_entry = {
				type = je_tionghoa_culture
			}
		}
		
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = dvg_lanfang.1.b
		
		if = {
			limit = {
				country_has_primary_culture = cu:moluccan
			}
			remove_primary_culture = cu:moluccan
		}
		if = {
			limit = {
				country_has_primary_culture = cu:malay
			}
			remove_primary_culture = cu:malay
		}

		add_loyalists = {
			value = 0.15
		}
	

		ai_chance = {
			base = 1
		}
	}
}

	duration = 3

	trigger = {
	}

	# Build Temple!
	option = {
		name = dvg_lanfang.3.a
		default_option = yes

		hidden_effect = {
			
		

		c:BRU = {trigger_event = { id = dvg_lanfang.3 }}

		}

		add_radicals = {
			value = 0.20
			religion = rel:sunni
		}
		add_loyalists = {
			value = 0.20
			religion = rel:mahayana
		}
		if = {
			limit = {
				exists = c:BAL
			}
			change_relations = {
				country = c:BAL
				value = 15
			}
		}
		if = {
			limit = {
				any_interest_group = {
					has_ideology = ideology:ideology_buddhist_moralist
				}
			}
			ig:ig_devout = {
			
				remove_ideology = ideology_buddhist_moralist
				add_ideology = ideology_dharmic
			}
		}
		

		ai_chance = {
			base = 1
		}
	}
}

#The Arrival of the Pillager Prince
dvg_lanfang.7 = {
	type = country_event
	placement = ROOT

	title = dvg_lanfang.7.t
	desc = dvg_lanfang.7.d
	flavor = dvg_lanfang.7.f

	event_image = {
		video = "gfx/event_pictures/asia_westerners_arriving.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	immediate = {
		random_country = {
			limit = {
				country_has_primary_culture = cu:zhourao
				
			}
			save_scope_as = zhourao_scope
		}
		random_character = {
			limit = {
				has_variable = pillager_prince
			}
			transfer_character = ROOT
		}
	}

	# Pay him off
	option = {
		name = dvg_lanfang.7.a
		default_option = yes
		add_radicals = {
			value = 0.35
		}

		
		
		random_country = {
			limit = {
				country_has_primary_culture = cu:zhourao
			}
			trigger_event = {
				id = dvg_zhourao.1200
			}
		}

		add_treasury = -10000
		
		ai_chance = {
			base = 99
		}
	}
	# Refuse to pay him
	option = {
		name = dvg_lanfang.7.b

		hidden_effect = {
			if = {
				limit = {
					AND = {
						has_law = law_type:law_no_police
						has_law = law_type:law_no_home_affairs
					}
				}
				trigger_event = {
					id = dvg_lanfang.8 			days = 7

				}

			}
		}
		add_loyalists = {
			value = 0.25
		}

		ai_chance = {
			base = 1
		}
	}
	#Invite him to stay
	option = {
		name = dvg_lanfang.7.c
		add_radicals = {
			value = 0.45
		}
		add_modifier = {
			name = dvg_underground_innovation
			months = 30
			is_decaying = yes
		}
		every_character = {
			limit = {
				has_variable = pillager_prince
			}
			transfer_character = ROOT
			remove_trait = immortal
			add_character_role = agitator
			add_character_role = general
		}

		ai_chance = {
			base = 1
		}
	}
}
