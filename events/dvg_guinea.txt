dvg_guinea.3 = { #Company Rule Collapses
	type = country_event
	placement = root
	title = dvg_guinea.3.t
	desc = dvg_guinea.3.d
	flavor = dvg_guinea.3.f

	duration = 3

	event_image = {
		video = "asia_confucianism_shinto"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
	
	}
	immediate = {
		random_primary_culture = {
			limit = {
				NOT = {
					has_discrimination_trait_group = heritage_group_african
				}
			}
			save_scope_as = foreign_culture
		}
		
	}

	option = {
		name = dvg_guinea.3.a

		
		
		add_radicals ={
			value = 0.15
		}
	}

	option = {
		name = dvg_guinea.3.b
		default_option = yes
		

		remove_primary_culture = scope:foreign_culture
		if = {
			limit = {
				NOT = {
					exists = c:IBO
				}
			}
			change_tag = IBO
		}

		add_loyalists = {
			value = 0.10
		}

	}

	option = {
		name = dvg_guinea.3.c
		trigger = {
			has_technology_researched = pan-nationalism
		}

		remove_primary_culture = scope:foreign_culture
		add_primary_culture = cu:afro_frankish

		add_loyalists = {
			value = 0.10
		}

	}
}

dvg_guinea.6 = {  #Plantagenia has banned slavery
	type = country_event
	placement = root
	title = dvg_guinea.6.t
	desc = dvg_guinea.6.d
	flavor = dvg_guinea.6.f

	duration = 3

	event_image = {
		video = "asia_buddhism"

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
		
	}
	immediate = {
		overlord = {
			save_scope_as = scope_PLA
		}
	}

	option = {
		name = dvg_guinea.6.a
		default_option = yes
		
		
	}

	option = {
		name = dvg_guinea.6.b
		trigger = {
			any_country = {
				is_country_type = recognized
				NOT = {
					has_law = law_type:law_slavery_banned
				}
				any_primary_culture = {
					has_discrimination_trait = language_francophone
				}
			}
		}
		
		
		trigger_event = {
			id = dvg_guinea.7
		}
	}

	option = {
		name = dvg_guinea.6.c
		
		create_diplomatic_play = {
			type = dp_independence
			target_country = scope:scope_PLA
		}
		
	}
}
