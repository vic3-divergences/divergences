		}
	}
	immediate = {

			limit = {
				has_variable = zimisces_var
			}
			save_scope_as = zimisces_scope
		}
		c:VEN = {
			save_scope_as = scope_VEN
		}
		if = {
			limit = {
				exists = c:GRE
			}
			
			c:GRE = {
				save_scope_as = scope_GRE
			}
		}
		if = {
			limit = {
				NOT = {
					exists = c:GRE
				}
			}
			random_scope_state = {
				limit = {
					state_region = {
						is_homeland = cu:greek
					}
				}
				save_scope_as = scope_greek_state

			}
			
		}
	}
	option = {	#No council
		name = dvg_greece.201.a
		default_option = yes
		
		add_radicals = {
			value = 0.30
			culture = cu:greek
		}
		if = {
			limit = {
				exists = c:GRE
			}
			change_relations = {
				country = c:GRE
				value = -25
			}
			c:GRE = {
				add_liberty_desire = {
					value = 20
				}
			}
		}
		scope:zimisces_scope = {
			exile_character = yes
		}
		set_variable = council_cancelled

		ai_chance = {
			base = 01
		}
	}
	option = {	
		name = dvg_greece.201.b

		trigger_event = {
			id = dvg_greece.202 days = 30
		}
		if = {
			limit = {
				NOT = {
					exists = c:GRE
				}
				
			}
			create_country = {
				tag = GRE
				origin = this
				state = s:STATE_ATTICA.region_state:VEN
				state = s:STATE_PELOPONNESE.region_state:VEN	
				state = s:STATE_THESSALIA.region_state:VEN				
			
			}
			create_diplomatic_pact = {
				country = c:GRE
				type = puppet
			}
			c:GRE = {
				activate_law = law_type:law_autocracy
			}
			
			

			random_scope_character = {
				limit = {
					has_variable = zimisces_var
				}
				transfer_character = c:GRE
				add_trait = immortal
				
			}
			
			
		}

		ai_chance = {
			base = 33
		}
	}
	option = {	#If he loves Greece so much, let's send him there instead!
		name = dvg_greece.201.c
		trigger = {
			exists = c:GRE
		}
		random_state = {
			limit = {
				owner = c:GRE
			}

		}

			limit = {

	}

	trigger = { 
		exists = c:GRE
		c:GRE = {
			is_ai = yes
		}
	}
	immediate = {

		random_character = {
			limit = {
				has_variable = zimisces_var
			}
			save_scope_as = zimisces_scope
		}
		c:GRE = {
			save_scope_as = greek_revolt_country_scope
			create_diplomatic_play = {
				type = dp_independence
				target_country = c:VEN
				war = no
			}
		}
		
		
	
		
		
		if = {
			limit = {
				has_variable = council_start
			}
			remove_variable = council_start
			remove_variable = greece_citizen_var
			remove_variable = greece_voting_var
			set_variable = council_concluded
			
		}
		every_scope_state = {
			limit = {
				state_region = {
					is_homeland = cu:greek
				}
				culture_percent_state = {
					target = cu:greek
					value >= 0.40
				}
			}
			set_state_owner = c:GRE
		}
		
	}
	option = {	 #Kill the Greek Revolt
		name = dvg_greece.203.a
		default_option = yes

		hidden_effect = {
			c:GRE = {
				trigger_event = {
					id = dvg_greece.213 days = 3
				}
			}
		}
		every_diplomatic_play = {
			limit = {
				initiator_is = c:GRE
				target_is = c:VEN
			}
			add_war_goal = {
				holder = c:VEN
				type = annex_country
				target_country = c:GRE
			}
		}

		

		add_loyalists = {
			value = 0.10
		}

		ai_chance = {
			base = 33
		}
	}
	option = {	#Fight Venice
		name = dvg_greece.203.b

		play_as = scope:greek_revolt_country_scope

		every_diplomatic_play = {
			limit = {
				initiator_is = c:GRE
				target_is = c:VEN
			}
			add_war_goal = {
				holder = c:VEN
				type = annex_country
				target_country = c:GRE
			}
		}

		
		hidden_effect = {
			c:GRE = {
				trigger_event = {
					id = dvg_greece.213 days = 3
				}
			}
		}
		


		ai_chance = {
			base = 33
		}
	}
	
}
