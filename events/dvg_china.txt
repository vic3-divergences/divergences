	}
	immediate = {
		hidden_effect = {
			
		}
		random_scope_state = {
			
			limit = {
				state_region = {
					is_homeland = cu:han
					
				}
				OR = {
				has_variable = cult_center
			
				
				turmoil >= 0.33
				}
				owner = ROOT
				is_capital = no
				
				
			
			}
		save_scope_as = KMT_capital
		set_state_owner = c:KMT

		
		}
		create_country = {
			origin = THIS
			state = scope:KMT_capital
		}
		every_scope_state = {
		
		limit = {
			
			state_region = {
				
				is_homeland = cu:han
				
			}
			OR = {
				turmoil >= 0.40
				radical_fraction = {
				value = 0.40
				}
			}
			
		}
		set_state_owner = c:KMT

		}
		c:KMT = {
			set_variable = blue_turban_revolt

			activate_law = law_type:law_national_supremacy
		}
		
	}

	option = {
		name = dvg_china.9.a
		
		create_diplomatic_play = {
			type = dp_rebellion_annex
			target_country = c:KMT
			war = yes

			add_war_goal = {
				holder = c:KMT
				type = annex_country
			}
		}

		remove_modifier = questioned_mandate
		
	}
