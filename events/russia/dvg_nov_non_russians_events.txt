# Non russians start event
dvg_nov_non_russians.1 = {
	type = country_event
	placement = ROOT

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/capitalists_meeting"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	title = dvg_nov_non_russians.1.t
	desc = dvg_nov_non_russians.1.d
	flavor = dvg_nov_non_russians.1.f

	duration = 3

	trigger = {
		exists = c:VYA
		this = c:VYA
		year <= 1840
		NOT = { has_variable = VYA_nonrussians }
		NOT = { has_variable = VYA_nonrussians_end }
	}

	immediate = {
		set_variable = VYA_nonrussians
	}

	option = {
		name = dvg_nov_non_russians.1.a
		default_option = yes

		if = {
			limit = {
				NOT = { has_law = law_type:law_racial_segregation }
			}
			activate_law = law_type:law_racial_segregation
		}

		c:NOV = {
			trigger_event = {
				id = dvg_nov_non_russians.3
				days = 7
				popup = yes
			}
		}

		ai_chance = {
			base = 75
		}
	}

	option = {
		name = dvg_nov_non_russians.1.b

		set_variable = VYA_cringe
		
		ai_chance = {
			base = 25
		}
	}
}

dvg_nov_non_russians.6 = {
	type = country_event
	placement = ROOT

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_before_the_battle.bk2"
	}
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/before_the_battle"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = dvg_nov_non_russians.6.t
	desc = dvg_nov_non_russians.6.d
	flavor = dvg_nov_non_russians.6.f

	duration = 3

	immediate = {
		# Create UDM
		create_country = {
			tag = UDM
			origin = this
			province = p:x003051
		}
		s:STATE_DVG_VYATKA = {
			set_owner_of_provinces = {
				country = c:UDM
				provinces = { x0A78C3 x0E6979 x199917 x1FD22C x2080C0 x29245A x31E881 x336835 x342573 x34999F x40D3F3 x41DF7D x4344A1 x4401F4 x454C45
				x49C31C x4CB925 x4E1217 x5B57D5 x6578DF x730769 x78F48B x7D0DB0 x8BAC4A x8C2EE8 x948973 xA7F595 xAB7912 xADCD83 xB5DD9A xB6C701 xB7D465 xB9522D 
				xC410CF xCB3697 xD754F9 xD80177 xDBB080 xDCF9AE xEF2B33 xEFA1B1 xF4A935 }
			}
		}
		save_temporary_scope_value_as = {
			name = war_reparations_amount
			value = {
				value = c:NOV.income
				multiply = define:NDiplomacy|DEFAULT_WAR_REPARATIONS_MONEY_TRANSFER
			}
		}
	}

	option = {
		name = dvg_nov_non_russians.6.a
		default_option = yes

		add_radicals = {
			value = very_large_radicals
			culture = cu:udmurt
		}

		set_global_variable = UDM_WAR

		c:UDM = {
			activate_law = law_type:law_monarchy
			activate_law = law_type:law_autocracy
			add_loyalists = {
				value = very_large_radicals
				culture = cu:udmurt
			}
			create_diplomatic_play = {
				target_country = c:VYA
				war = yes
				type = dp_conquer_state
				target_state = s:STATE_DVG_VYATKA.region_state:VYA
			
				add_war_goal = {
					holder = c:VYA
					type = annex_country
					target_country = c:UDM
				}
				add_war_goal = {
					holder = c:UDM
					target_country = c:NOV
					type = enforce_treaty_article
					article = {
						article = money_transfer
						source_country = c:NOV
						target_country = c:UDM
						inputs = {
							{ quantity = scope:war_reparations_amount }
						}
					}
				}
			}
		}

		c:NOV = {
			trigger_event = {
				id = dvg_nov_non_russians.7
				days = 0
				popup = yes
			}
		}
	}
}

				limit = {
					NOT = { has_law = law_type:law_national_supremacy }
				}

			}
		}

			base = 40
