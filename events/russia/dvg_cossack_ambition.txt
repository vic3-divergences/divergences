dvg_cossack_ambition.2 = {
	type = country_event
	placement = ROOT

	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/capitalists_meeting"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	title = dvg_cossack_ambition.2.t
	desc = dvg_cossack_ambition.2.d
	flavor = dvg_cossack_ambition.2.f

	duration = 3

	trigger = {
		NOT = { has_variable = sternenko_move }
		has_revolution = no
		NOT = { has_variable = KHK_angry_host }
	}

	immediate = {
		set_variable = sternenko_move
		set_global_variable = move_CSK
		set_strategy = ai_strategy_unify_cossacks
	}

	option = {
		name = dvg_cossack_ambition.2.a
		default_option = yes

		if = {
			limit = {
				NOT = { has_law = law_type:law_autocracy }
			}
			activate_law = law_type:law_autocracy
		}

		remove_modifier = neutrality
		remove_modifier = cossack_lifestyle

		add_radicals = {
			value = 0.1
		}

		change_relations = {
			country = c:ZAZ
			value = -100
		}
		
		change_relations = {
			country = c:DON
			value = -100
		}

		change_relations = {
			country = c:VOR
			value = -100
		}

		hidden_effect = {
			set_secret_goal = {
				country = c:ZAZ
				secret_goal = conquer
			}
			set_secret_goal = {
				country = c:DON
				secret_goal = conquer
			}
			set_secret_goal = {
				country = c:VOR
				secret_goal = conquer
			}
		}

		ai_chance = {
			base = 75
		}
	}

	option = {
		name = dvg_cossack_ambition.2.b
	
		remove_modifier = cossack_lifestyle
		
		ai_chance = {
			base = 25
		}
	}
}

dvg_cossack_ambition.4 = {
    type = country_event
    hidden = yes

    trigger = {
		is_ai = yes
        exists = c:ZAZ
		is_at_war = no
		NOT = { has_truce_with = c:ZAZ }
		NOT = {
			any_diplomatic_play = {
				OR = {
					initiator = ROOT
					target = ROOT
				}
			}
		}
    }

	immediate = {
		create_diplomatic_play = {
			target_country = c:ZAZ
			war = no
			type = dp_unify_cossacks
		}
	}
}
