dvg_ryuukidou.1 = { 
	type = country_event
	placement = root
	title = dvg_ryuukidou.1.t
	desc = dvg_ryuukidou.1.d
	flavor = dvg_ryuukidou.1.f

	duration = 3

	event_image = {
		video = "asia_confucianism_shinto"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
		
	}

	option = {
		name = dvg_ryuukidou.1.a
		default_option = yes
		
		ig:ig_devout = {
			add_modifier = {
				name = great_awakening_modifier
				months = 120
				is_decaying = yes
			} 
		}
		add_loyalists = {
			value = 0.10
			culture = cu:japanese
		}
		every_scope_state = {
			convert_population = {
				target = rel:gelugpa
				value = 0.1
			}
		}
		add_modifier = {
			name = colonial_enthusiasm_settlers
			months = 30
			is_decaying = yes
		}
	}

	option = {
		name = dvg_ryuukidou.1.b

		add_loyalists = {
			value = 0.10
			culture = cu:siberian
		}
		add_loyalists = {
			value = 0.10
			culture = cu:ainu
		}
		add_loyalists = {
			value = 0.10
			culture = cu:yakut
		}
		
		change_tension = {
				country = c:YAK
				value = -15
		}
	}
}

dvg_ryuukidou.2 = { #Settling the border
	type = country_event
	placement = scope:dvg_ryuukidouese_korean_state
	title = dvg_ryuukidou.2.t
	desc = dvg_ryuukidou.2.d
	flavor = dvg_ryuukidou.2.f

	duration = 3

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	
	immediate = { 
		
    }
	trigger = {
		
	}
	option = {	#We will keep border as is 
		name = dvg_ryuukidou.2.a
		default_option = yes

		change_infamy = -5
		
		every_scope_state = {
			limit = {
				OR = {
					region = sr:region_west_siberia
					region = sr:region_east_siberia
				}
				is_split_state = yes
			}
			add_claim = ROOT
			every_neighbouring_state = {
				owner = {
					change_relations = {
						country = THIS
						value = 20
					}
				}
			}
		}
	}

	option = {	#We will press our claims further
		name = dvg_ryuukidou.2.a

		create_incident = {
				country = THIS
				value = 10
		}
		
		every_scope_state = {
			limit = {
				OR = {
					region = sr:region_west_siberia
					region = sr:region_east_siberia
				}
				is_split_state = yes
			}
			add_claim = ROOT
			every_neighbouring_state = {
				owner = {
					change_relations = {
						country = THIS
						value = -20
					}
				}
			}
		}
	}
}
