#Herrerian Revolution 
dvg_gran_colombia.6 = {
	type = country_event
	placement = ROOT

	title = dvg_gran_colombia.6.t
	desc = dvg_gran_colombia.6.d
	flavor = dvg_gran_colombia.6.f
	
	event_image = {
		video = "southamerica_war_civilians"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/war_civilians"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	
	# Colombia joins Herreria
	option = {
		name = dvg_gran_colombia.6.a
		default_option = yes
		
		if = {
			limit = {
				NOT = {
					any_scope_treaty = {
						binds = c:SAN
						any_scope_article = {
							has_type = alliance
						}
					}
				}
			}
			create_treaty = {
				first_country = root
				second_country = c:SAN

				is_draft = no
				binding_period = { years = 5 }

				articles_to_create = {
					{
						article = alliance
					}
				}
			}
		}
		add_journal_entry = {
			type = je_herrerian_revolution_GCO
		}
		c:SAN = {
			change_relations = {
				country = c:GCO
				value = 100
			}
		}
		random_diplomatic_play = {
			limit = {
				any_scope_play_involved = { this = c:SAN }
				any_scope_play_involved = { this = c:GRA }
				is_diplomatic_play_type = dp_annex_war
			}
			add_target_backers = { root }
		}
		ai_chance = {
			base = 99
		}
	}
	#Colombia ignores Herreria
	option = {
		name = dvg_gran_colombia.6.b
		
		trigger_event = { id = dvg_gran_colombia.7 days = 7 }
		ai_chance = {
			base = 1
		}
	}
}

dvg_gran_colombia.18 = {
	type = country_event
	placement = ROOT

	title = dvg_gran_colombia.18.t
	desc = dvg_gran_colombia.18.d
	flavor = dvg_gran_colombia.18.f
	
	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	immediate = {
		every_scope_character = {
			limit = { has_variable = is_peixoto }
			save_scope_as = scope_peixoto
		}
		every_scope_character = {
			limit = { has_variable = is_de_marquez }
			save_scope_as = scope_de_marquez
		}
		every_scope_character = {
			limit = { has_variable = is_castro }
			save_scope_as = scope_castro
		}
		every_scope_character = {
			limit = { has_variable = is_santander }
			save_scope_as = scope_santander
		}
	}
	trigger = {
	}
	
	#Peixoto's coup
	option = {
		name = dvg_gran_colombia.18.a
		
		c:GCO = { 
			change_tag = MEX
			remove_primary_culture = cu:lusitanian
			remove_primary_culture = cu:central_american
		}
		if = {
			limit = {
				NOT = { has_law = law_type:law_no_colonial_affairs }
			}
			activate_law = law_type:law_no_colonial_affairs
		}
		
		hidden_effect = {
			if = {
				limit = { exists = s:STATE_BAHIA.region_state:MEX }
				create_country = {
					tag = LUS
					origin = this
					state = s:STATE_BAHIA.region_state:MEX
					state = s:STATE_SANTA_CATARINA.region_state:MEX
					state = s:STATE_TEJAS.region_state:MEX
					state = s:STATE_RIO_GRANDE.region_state:MEX
					state = s:STATE_BAJA_CALIFORNIA.region_state:MEX
					on_created = {
						add_modifier = {
							name = dvg_revolutionary_militias
							days = 720
							is_decaying = yes
						}
						add_loyalists = {
							value = 0.35
