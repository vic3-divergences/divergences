dvg_caladero.17 = {
	type = country_event
	placement = ROOT

	title = dvg_caladero.17.t
	desc = dvg_caladero.17.d
	flavor = dvg_caladero.17.f
	
	event_image = {
		video = "southamerica_war_civilians"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	immediate = {
		c:GRA = {
			every_scope_character = {
				limit = { has_variable = is_ballivian }
				save_scope_as = scope_ballivian
			}
		}
		every_scope_character = {
			limit = { has_variable = is_tupac }
			save_scope_as = scope_tupac
		}
	}
	trigger = {
	}
	#Viva el Presidente!
	option = {
		name = dvg_caladero.17.a
		default_option = yes
		
		scope:scope_ballivian = {
			transfer_character = c:QUI
			set_character_as_ruler = yes
		}
		scope:scope_tupac = {
			set_character_as_ruler = no
			add_character_role = general
		}
		set_variable = dvg_QUI_ballivian
		if = {
			limit = { is_subject_type = subject_type_protectorate }
			c:SPA = {
				remove_diplomatic_pact = {
					country = c:QUI
					type = protectorate
				}
			}
		}
		else_if = {
			limit = { is_subject_type = subject_type_dominion }
			c:SPA = {
				remove_diplomatic_pact = {
					country = c:QUI
					type = dominion
				}
			}
		}
		else_if = {
			limit = { is_subject_type = subject_type_puppet }
			c:SPA = {
				remove_diplomatic_pact = {
					country = c:QUI
					type = puppet
				}
			}
		}
		else = {
			limit = { is_subject_type = subject_type_colony }
			c:SPA = {
				remove_diplomatic_pact = {
					country = c:QUI
					type = colony
				}
			}
		}
		
		ai_chance = {
			base = 1
		}
	}
}
