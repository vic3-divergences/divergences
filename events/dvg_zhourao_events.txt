    flavor = dvg_zhourao.6.f

    event_image = {
        video = "gfx/event_pictures/unspecific_fire.bk2"
    }

    on_opened_soundeffect = "event:/SFX/Events/southamerica/war_civilians"

    icon = "gfx/interface/icons/event_icons/event_fire.dds"

    duration = 3

    trigger = {
        #has_variable = dvg_yaluo_reform_variable_var_2
        #this = { var:dvg_yaluo_reform_variable_var_2 >= 8 }
        #has_variable = dvg_one_kongsi
        #NOT = {
        #	has_variable = dvg_northern_hui

        #exists = c:MJN (must exist)
    }

    immediate = {
        set_variable = dvg_northern_hui
        ig:ig_landowners = {
            save_scope_as = landowners_ig
        }
	every_country = {
		limit = {
			is_revolutionary = yes
			country_definition = root.country_definition
			owns_entire_state_region = STATE_XINXIANG
			country_has_primary_culture = cu:zhourao
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_XINXIANG
				owner = prev
			}
			set_state_owner = c:MJN
		}
	}
        if = {
            limit = {
                s:STATE_XINXIANG.region_state:YLO = {
                    ig_state_pol_strength_share = {
                        target = scope:landowners_ig
                        value > 0.1
                    }
                }
            }
            s:STATE_XINXIANG.region_state:YLO = {
                set_state_owner = c:MJN
            }
        }
        ordered_state = {
            limit = {
                owner = c:MJN
            }
            order_by = state_population
            save_scope_as = MJN_state
        }

    }

    # Oh dear
    option = {
        name = dvg_zhourao.6.a
        custom_tooltip = {
            text = dvg_effects_of_reforms_tt
            set_variable = dvg_among_kongsus
            trigger_event = { id = dvg_zhourao.7 days = 90 }
            s:STATE_XINXIANG = {
                add_claim = c:YLO
            }
            change_relations = {
                country = c:MJN
                value = -50
            }
        }
        if = {
            limit = {
                ig:ig_landowners = {
                    is_marginal = no
                }
            }
            ordered_scope_pop = {
                limit = {
                    is_pop_type = aristocrats
                }
                order_by = total_size
                max = 6
                move_partial_pop = { 
                    state = scope:MJN_state
                    population_ratio = 0.3
                }
            }
            ordered_scope_pop = {
                limit = {
                    is_pop_type = farmers
                }
                max = 6
                move_partial_pop = { 
                    state = scope:MJN_state
                    population_ratio = 0.1
                }
            }
        }


        add_modifier = {
            name = dvg_modifier_landowners_purged
            months = 60
            is_decaying = yes
        }
        ai_chance = {
            base = 1
        }
    }
}
