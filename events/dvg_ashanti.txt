#Ashanti Revolt
dvg_ashanti.2 = {
	type = country_event
	placement = ROOT

	title = dvg_ashanti.2.t
	desc = dvg_ashanti.2.d
	flavor = dvg_ashanti.2.f

	event_image = {
		video = "africa_prosperous_farm"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		
		
		if = {
			limit = {
				any_country = {
					country_has_primary_culture = cu:akan
					is_diplomatic_play_enemy_of = ROOT
					OR = {
					is_revolutionary = yes
					is_secessionist = yes
					}
				}
				
			}
			random_country = {
				limit = {
					country_has_primary_culture = cu:akan
					OR = {
					is_revolutionary = yes
					is_secessionist = yes
					}
				}
				change_tag = AKN
				save_scope_as = scope_AKN
			}
		}

	}


	
	option = {
		name = dvg_ashanti.2.a
		default_option = yes

		create_diplomatic_play = {
			type = dp_rebellion_annex
			target_country = scope:scope_AKN

			add_war_goal = {
				holder = scope:scope_AKN
				type = annex_country
				target_country = ROOT
			}
		}
		
		if = {
			limit = {
				exists = c:ASH
			}
			scope:scope_AKN = {
				trigger_event = {
					id = dvg_ashanti.3
				}
			}
		}
	}
	option = {
		name = dvg_ashanti.2.b

		play_as = scope:scope_AKN
		

		create_diplomatic_play = {
			type = dp_rebellion_annex
			target_country = scope:scope_AKN

			add_war_goal = {
				holder = scope:scope_AKN
				type = annex_country
				target_country = ROOT
			}
		}
		if = {
			limit = {
				exists = c:ASH
			}
			scope:scope_AKN = {
				trigger_event = {
					id = dvg_ashanti.3
				}
			}
		}

		ai_chance = {
			base = 50
		}
	}
	
}

dvg_ashanti.4 = {
	type = country_event
	placement = ROOT

	title = dvg_ashanti.4.t
	desc = dvg_ashanti.4.d
	flavor = dvg_ashanti.4.f

	event_image = {
		video = "europenorthamerica_judaism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		any_country = {
			country_has_primary_culture = cu:afro_frankish
		}
		any_country = {
			country_has_primary_culture = cu:akan
		}
	}
	immediate = {
		random_country = {
			limit = {
				country_has_primary_culture = cu:afro_frankish
			}
			save_scope_as = scope_CDA
		}
		random_country = {
			limit = {
				country_has_primary_culture = cu:akan
				NOT = {
					is_country_type = decentralized
				}
			}
			save_scope_as = scope_AKN
		}
	}


	option = {
		name = dvg_ashanti.4.a
		default_option = yes

		create_diplomatic_pact = {
			country = scope:scope_AKN
			type = personal_union
		}
		scope:scope_AKN = {
			add_liberty_desire = -100
		}

		every_diplomatic_play = {
			limit = {
				target_is = scope:scope_AKN
			}
			add_target_backers = {
				ROOT
			}
			add_war_goal = {
				holder = ROOT
				type = annex_country
				target_country = scope:scope_CDA
			}
		}
		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_ashanti.4.b
		
		change_relations = {
			country = scope:scope_AKN
			value = -30
		}
		
		ai_chance = {
			base = 1
		}
	}
}
