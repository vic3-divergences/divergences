dvg_lotharingia.5 = { #Cards Fall Apart Split Lotharingia up into warlord states
	type = country_event
	placement = root
	duration = 3

	title = dvg_lotharingia.5.t
	desc = dvg_lotharingia.5.d
	flavor = dvg_lotharingia.5.f

	event_image = {
		video = "europenorthamerica_springtime_of_nations"
	}

	on_created_soundeffect = "event:/SFX/Events/southamerica/war_civilians"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	trigger = {
		exists = c:LOT
		this = c:LOT
		NOT = { has_global_variable = lotharingia_shatters }
	}

	immediate = {
		
		set_global_variable = lotharingia_shatters

		if = {
			limit = {
				has_global_variable = atlantia_independent
			}
			remove_global_variable = atlantia_independent
		}		
		
		every_state_region = {
			limit = {
				state_region = {
					is_homeland = cu:lotharian
				}			
			}
			state_region = {
				add_claim = ROOT
			}
		}
		if = {
			limit = {
				NOT = {
					exists = c:ODO
				}
			}
			change_tag = ODO
	
		
			if = {
			limit = { exists = s:STATE_LORRAINE_ATLANTIQUE.region_state:ODO }
			create_country = { 
				tag = LOT
				origin = this
				state = s:STATE_LORRAINE_ATLANTIQUE.region_state:ODO
				state = s:STATE_BRUXELLES_SUR_ODON.region_state:ODO
				state = s:STATE_SERMOISE_SUR_ODON.region_state:ODO

				on_created = {
					set_variable = warlord_state
				}
			}
			c:LOT = { save_scope_as = scope_ODO
			change_relations = {
				country = c:ATL
				value = -50
			}
			}
			
			}
		}
		

		if = {
			limit = { exists = s:STATE_NOUVELLE_ALSACE.region_state:ODO }
			create_country = { #Nouvelle Alsace
				tag = NEA
				origin = this
				state = s:STATE_NOUVELLE_ALSACE.region_state:ODO
				state = s:STATE_LIBREVILLE.region_state:ODO

				on_created = {
					set_variable = warlord_state
				}
			}
			c:NEA = { save_scope_as = scope_NEA
			change_relations = {
				country = c:ATL
				value = 50
			}
			change_relations = {
				country = c:LOT
				value = 50
			}
			}
		}

		if = {
			limit = { exists = s:STATE_COTE_DU_CAFE.region_state:ODO }
			create_country = { #Atlantia
				tag = ATL 
				origin = this
				state = s:STATE_COTE_DU_CAFE.region_state:ODO
				state = s:STATE_GOUAS.region_state:ODO

				on_created = {
					set_variable = warlord_state
				}
			}
			c:ATL = { save_scope_as = scope_ATL }
			
			
		}
		if = {
			limit = { exists = s:STATE_AVALON.region_state:ODO
			NOT = {exists = c:BEO}
			}
			create_country = { #Beornia
				tag = BEO
				origin = this
				state = s:STATE_AVALON.region_state:ODO
				state = s:STATE_GODWINSON.region_state:ODO
				state = s:STATE_EOFERWIC.region_state:ODO
				state = s:STATE_WESSEX.region_state:ODO
				state = s:STATE_NEW_CORNWALL.region_state:ODO
				state = s:STATE_NEW_LONDON.region_state:ODO
				state = s:STATE_NORTHUMBRIA.region_state:ODO

				on_created = {
					set_variable = warlord_state
				}
			}
			c:BEO = { save_scope_as = scope_BEO }
			
			
		}

		if = {
			limit = { 
				OR = {
					cu:guarani = {
						culture_current_fervor > 49
					}
					has_variable = white_lot_favored
				}
				NOT = {
					any_country = {
						country_has_primary_culture = cu:guarani
					}
				}
			}
			create_country = { #Taragui
				tag = TGI
				origin = this
				state = s:STATE_CHACO_BOREAL.region_state:ODO
				state = s:STATE_MARCHES_OUEST.region_state:ODO
				state = s:STATE_GRAND_CHACO.region_state:ODO
				state = s:STATE_EOFERWIC.region_state:ODO
				state = s:STATE_NORTHUMBRIA.region_state:ODO
				state = s:STATE_AMAZONIE_INFERIEURE.region_state:ODO

				on_created = {
					set_variable = warlord_state
				}
			}
			c:TGI = { save_scope_as = scope_TGI
			change_relations = {
				country = c:ATL
				value = 25
			}
			change_relations = {
				country = c:ODO
				value = 25
			}
			change_relations = {
				country = c:NEA
				value = 25
			}
			}
		}
		if = {
			limit = {
				any_scope_country = {
					country_has_primary_culture = cu:guarani
				}
			}
			save_scope_as = scope_TGI
		}

		every_subject_or_below = {
			limit = {
				is_subject_of = ROOT
				country_is_in_south_america = yes
				OR = {
					is_subject_type = subject_type_vassal
					is_subject_type = subject_type_protectorate
					is_subject_type = subject_type_dominion
					is_subject_type = subject_type_puppet
				}
				liberty_desire > 50
			}
			
			make_independent = yes
		}

		c:ODO = {
		 if = {
			limit = {
				has_variable = warlord_state
				NOR = {
					has_variable = var_warlord_two
					has_variable = var_warlord_three
					has_variable = var_warlord_four 
				}
			}
			save_scope_as = scope_random_warlord_one
			set_variable = var_warlord_one
			}
			
		}

		c:NEA = {
		if = {
			limit = {
				has_variable = warlord_state
				NOR = {
					has_variable = var_warlord_one

		}

			save_scope_as = scope_ATL
		if = {
			limit = {
				has_variable = warlord_state
				NOR = {
					has_variable = var_warlord_one

	}
	option = {

			}
		}


		create_diplomatic_play = {

	}
