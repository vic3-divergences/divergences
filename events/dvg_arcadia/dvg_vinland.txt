dvg_vinland.10 = { #End of Dominion
	type = country_event
	placement = ROOT

	title = dvg_vinland.10.t
	desc = dvg_vinland.10.d
	flavor = dvg_vinland.10.f

	duration = 3

	event_image = {
		video = "ep1_transfer_of_authority"

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	trigger = {
		exists = c:VIN
		c:VIN = {
			is_subject_of = ROOT
		}
	}

	immediate = {
		every_country = {
			limit = {
				has_variable = support_vin
			}
			save_scope_as = vin_support
		}

		every_country = {
			limit = {
				has_variable = support_sca
			}
			save_scope_as = sca_support
		}
	}
	
	option = {
		name = dvg_vinland.10.a
		default_option = yes

		c:VIN = {
			make_independent = yes
		}
		every_diplomatic_play = {
			limit = {
				initiator_is = ROOT
				target_is = c:VIN
			}
			end_play = yes
		}

		ai_chance = {
			base = 90
		}
	}

	option = {
		name = dvg_vinland.10.b

		create_diplomatic_play = {
			type = dp_rebellion_annex
			target_country = c:VIN
			add_war_goal = { 
				type = independence
				holder = c:VIN
			}
			add_initiator_backers = {
				scope:sca_support
			}

			add_target_backers = {
				scope:vin_support
			}
		}
			
		c:VIN = {
			trigger_event = {
				id = dvg_vinland.12
			}
		}

		ai_chance = {
			base = 10
		}
	}
}
