tibet_expedition_events.2 = { #Choose commander
	type = country_event
    placement = ROOT

    title = expedition_events.2.t
    desc = expedition_events.2.d
    flavor = expedition_events.2.f

    duration = 3

	event_image = {
		video = "ep1_ca_mountain_pass_civilian"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	trigger = {
		any_scope_character = {
			OR = {
				has_role = general
				has_role = admiral
			}
			is_ruler = no
			is_busy = no
		}
	}

	cancellation_trigger = {
		NOR = {
			exists = scope:expedition_leader_1
			exists = scope:expedition_leader_2
			exists = scope:expedition_leader_3
		}
	}

	immediate = {
		if = {
			limit = {
				any_scope_character = {
					is_busy = no
					is_ruler = no
					OR = {
						has_role = general
						has_role = admiral
					}
					is_ruler = no
					has_trait = explorer
				}
			}
			random_scope_character = {
				limit = {
					is_busy = no
					is_ruler = no
					OR = {
						has_role = general
						has_role = admiral
					}
					has_trait = explorer
				}
				save_scope_as = expedition_leader_1
			}
		}
		random_scope_character = {
			limit = {
				is_busy = no
				is_ruler = no
				OR = {
					has_role = general
					has_role = admiral
				}
				trigger_if = {
					limit = { exists = scope:expedition_leader_1 }
					NOT = {
						this = scope:expedition_leader_1
					}
				}
			}
			save_scope_as = expedition_leader_2
		}
		random_scope_character = {
			limit = {
				is_busy = no
				is_ruler = no
				OR = {
					has_role = general
					has_role = admiral
				}
				NOR = {
					AND = {
						exists = scope:expedition_leader_1
						this = scope:expedition_leader_1
					}
					this = scope:expedition_leader_2
				}
			}
			save_scope_as = expedition_leader_3
		}
	}

	option = {
		name = expedition_events.602.a
		trigger = {
			exists = scope:expedition_leader_1
		}
		scope:expedition_leader_1 = {
			set_character_busy_and_immortal = yes
			set_variable = expedition_leader_var
			ROOT = {
				set_variable = { name = expedition_leader_storage_var value = prev }
			}
		}
		trigger_event = { id = tibet_expedition_events.3 days = 30 }
	}

	option = {
		name = expedition_events.602.b
		default_option = yes
		trigger = {
			exists = scope:expedition_leader_2
		}
		scope:expedition_leader_2 = {
			set_character_busy_and_immortal = yes
			set_variable = expedition_leader_var
			ROOT = {
				set_variable = { name = expedition_leader_storage_var value = prev }
			}
		}
		trigger_event = { id = tibet_expedition_events.3 days = 30 }
	}

	option = {
		name = expedition_events.602.c
		trigger = {
			exists = scope:expedition_leader_3
		}
		scope:expedition_leader_3 = {
			set_character_busy_and_immortal = yes
			set_variable = expedition_leader_var

tibet_expedition_events.4 = { # Yaks cross the border
	type = country_event
    placement = scope:current_expedition_location

    title = tibet_expedition_events.4.t
    desc = tibet_expedition_events.4.d
    flavor = tibet_expedition_events.4.f

    duration = 3

	event_image = {
		video = "ep1_ca_mountain_pass_civilian"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
		NOT = {
			any_interest_group = {
				OR = {
					has_modifier = hostility_resisted
					has_modifier = resisted_being_hostile
				}
			}
		}
		var:tibet_expedition_progress_var < 4
	}

	immediate = {

		random_scope_character = {
			limit = { has_variable = expedition_leader_var }
			save_scope_as = expedition_leader
		}
	}

	cancellation_trigger = {
		OR = {
			NOT = {
				any_scope_character = {
					has_variable = expedition_leader_var
				}
			}
			NOT = {
				exists = c:TIB
			}
		}
	}

	option = {
		name = tibet_expedition_events.4.a
		default_option = yes

		ai_chance = {
			base = 1
		}

		every_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_armed_forces
					has_ideology = ideology:ideology_jingoist
					leader = {
						has_ideology = ideology:ideology_jingoist_leader
 					}
				}
			}
			add_modifier = {
				name = hostility_resisted
				days = normal_modifier_time
			}
		}

		if = {
			limit = {
				NOT = {
					any_interest_group = {
						NOT = {
							is_interest_group_type = ig_armed_forces
						}
	    				OR = {
	    					has_ideology = ideology:ideology_jingoist
	    					leader = {
	    						has_ideology = ideology:ideology_jingoist_leader
	     					}
	  					}
					}
				}
			}
			random_interest_group = {
				limit = {
					NOR = {
						is_interest_group_type = ig_armed_forces
						is_interest_group_type = ig_rural_folk # RF is not normally a valid IG for Jingoists
						leader = { has_ideology = ideology:ideology_jingoist_leader }
					}
				}
				leader = {
					set_ideology = ideology:ideology_jingoist_leader
				}
				add_modifier = {
					name = hostility_resisted
					days = normal_modifier_time
				}
			}
		}

		custom_tooltip = {
			text = je_tibet_expedition_advance
			change_variable = { name = tibet_expedition_progress_var add = 1 }
		}
	}

	option = {
		name = tibet_expedition_events.4.b


		ai_chance = {
			base = 0
		}

		every_interest_group = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_isolationist
					leader = {
						has_ideology = ideology:ideology_pacifist
 					}
				}
			}
			add_modifier = {
				name = resisted_being_hostile
				days = normal_modifier_time
			}
		}
	}
}

	}

		name = tibet_expedition_events.21.a
		default_option = yes

		if = { # trigger tibet_expedition_events.26 only if overlord exists & c:TIB is subject (so this is not sent to an independent c:TIB with top_overlord)
			limit = {
				exists = scope:overlord_of_tibet
				c:TIB ?= {
					is_subject = yes
				}
			}
			scope:overlord_of_tibet = {
				trigger_event = {
					id = tibet_expedition_events.26 days = 1
				}
				set_variable = former_tibetan_overlord
			}
		}

		scope:expedition_country = {
			trigger_event = {
				id = tibet_expedition_events.27 days = 1
			}
		}

		hidden_effect = { # clearing out existing relations
			if = {
				limit = {
					is_subject = yes
				}
				make_independent = yes
			}
		}

		if = {
			limit = {
				country_rank <= rank_value:minor_power
				scope:expedition_country = {
					OR = {
						is_country_type = recognized
						is_country_type = colonial
					}
					country_rank >= rank_value:minor_power
				}
			}
			scope:expedition_country = {
				create_diplomatic_pact = {
					country = c:TIB
					type = protectorate
				}
			}
		}
		else_if = {
			limit = {
				country_rank <= rank_value:minor_power
				scope:expedition_country = {
					is_country_type = unrecognized
					country_rank >= rank_value:unrecognized_regional_power
				}
			}
			scope:expedition_country = {
				create_diplomatic_pact = {
					country = c:TIB
					type = tributary
				}
			}
		}
		else = {
			scope:expedition_country = {
				create_diplomatic_pact = {
					country = c:TIB
					type = vassal
				}
			}
		}

		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					has_dlc_feature = lobbies
					any_political_lobby = {
						is_political_lobby_type = lobby_pro_country
						target = {
							scope:expedition_country = this
						}
					}
				}
				add = 25
			}
		}
	}

	option = {
		name = tibet_expedition_events.21.b

		change_relations = {
			country = scope:expedition_country
			value = -40
		}
		scope:expedition_country = {
			set_variable = received_subjugation_refusal_var # variable for expedition_country to fail their Tibet Expedition JE if this option is selected
		}

		scope:expedition_country = {
			post_notification = tibet_refuses_demands
		}

		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					any_scope_character = {
						is_ruler = yes
						OR = {
							has_trait = ambitious
							has_trait = arrogant
							has_trait = imperious
						}
					}
				}
				add = 25
			}
			modifier = {
				trigger = {
					has_dlc_feature = lobbies
					any_political_lobby = {
						is_political_lobby_type = lobby_anti_country
						target = {
							scope:expedition_country = this
						}
					}
				}

		}
