			base = 50

					has_game_rule = anglois_culture_choice_french

					has_game_rule = anglois_culture_choice_english
				}
				add = -100
			}
			modifier = {
				trigger = {
					has_game_rule = anglois_culture_choice_anglois
				}
				add = -100
			}
		}

		name = dvg_dua.1.b

		set_variable = supporting_english_var

			type = je_english_integration

					add_homeland = cu:anglois
					remove_homeland = cu:french
				}
			}

			every_scope_state = {
				limit = {
					owner = ROOT
					state_region = {
						is_homeland = cu:british
					}
				}

				state_region = {
					add_homeland = cu:anglois
					remove_homeland = cu:british
				}
			}
		}

		custom_tooltip = {
			text = dvg_anglois_assimilate
			every_scope_state = {
				limit = {
					owner = ROOT
					state_region = {
						is_homeland = cu:anglois
					}
				}
				add_modifier = {
					name = dvg_assimilate
					months = 60
				}
			}
		}
		
		add_loyalists = {
			value = 0.2
			culture = cu:anglois
		}

		add_loyalists = {
			value = 0.3
			culture = cu:french
		}

		add_loyalists = {
			value = 0.2
			culture = cu:occitan
		}

		add_loyalists = {
			value = 0.3
			culture = cu:british
		}
	}
