dvg_australie.4 = {
	type = country_event
	placement = ROOT

	title = dvg_australie.4.t
	desc = dvg_australie.4.d
	flavor = dvg_australie.4.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_gold_prospectors.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		
	}
	immediate = {
		top_overlord = {
			save_scope_as = migration_country_b
			random_primary_culture = {
				save_scope_as = migration_culture_b
			}
			random_subject_or_below = {
				limit = {
					any_scope_state = {
							state_region = {
								is_homeland = cu:javan
							}
					}
				}
				save_scope_as = migration_country_a
				
				cu:javan = {
					save_scope_as = migration_culture_a
				}
			}
		}
		random_scope_state = {
			save_scope_as = owned_state
		}
		c:BEI = {
			save_scope_as = scope_BEI
		}
	}

	option = {
		name = dvg_australie.4.a
		default_option = yes
		
		random_scope_state = {
			create_mass_migration = {
			origin = scope:migration_country_b
			culture = scope:migration_culture_b
			}
		}

		add_modifier = {
			name = sa_migration_cost
			months = 70
			is_decaying = yes
		}
		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_australie.4.b
		trigger = {
			OR = {
				exists = c:BEI
				c:BRG = {
					any_scope_state = {
						any_scope_pop = {
							has_discrimination_trait = heritage_insulindian
						}
					}
				}
			}
		}
		hidden_effect = {
			every_scope_state = {
					add_cultural_community = cu:javan
					random_scope_pop = {
						change_pop_culture = {
							target = cu:javan
							value = 0.50
						}
					}
			}
		}
		
		top_overlord = {
			random_subject_or_below = {
				limit = {
					any_scope_state = {
							state_region = {
								is_homeland = cu:javan
							}
					}
				}
				move_pop = scope:owned_state
			}
		}
		random_scope_state = {
			create_mass_migration = {
			origin = scope:migration_country_a
			culture = scope:migration_culture_a
			}
		}
		

		add_radicals = {
			value = 0.10
			strata = lower
		}
		cu:australian = {
			add_fervor = 5
		}
		
	}
}

dvg_australie.9 = {
	type = country_event
	placement = ROOT

	title = dvg_australie.9.t
	desc = dvg_australie.9.d
	flavor = dvg_australie.9.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_gold_prospectors.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		OR = {
			cu:aborigine_malay = {
				culture_current_fervor > 24
			}
			cu:aborigine_french = {
				culture_current_fervor > 24
			}
		}
		OR = {
			any_scope_state = {
			state_region = s:STATE_YOLNGU
			turmoil > 0.33
			}
			radical_fraction = {
				value = 0.50
			}
		}
		NOT = {
			has_global_variable = yolngu_defection
		}
		OR = {
			any_scope_state = {
				count > 1
			}
			any_country = {
				is_adjacent_to_country = ROOT
				is_country_type = decentralized
			}
		}
		
	}
	immediate = {
		set_global_variable = yolngu_defection

		if = {
			limit = {
				exists = c:GRJ
			}
			save_scope_as = aboriginal_country
		}
		
		if = {
			limit = {
				NOT = {
					exists = c:GRJ
				}
				any_country = {
						NOT = {
							country_has_primary_culture = cu:australian
						}
						is_adjacent_to_country = ROOT
						OR = {
							country_has_primary_culture = cu:aborigine_malay
							country_has_primary_culture = cu:aborigine
						}
					}
			}
				
			random_country = {
				limit = {
					is_adjacent_to_country = ROOT
					NOT = {
							country_has_primary_culture = cu:australian
					}
					OR = {
							country_has_primary_culture = cu:aborigine_malay
							country_has_primary_culture = cu:aborigine
					}
				}
				save_scope_as = aboriginal_country
			}
		}
			

		if = {
			limit = {
				any_scope_state = {
					count > 1
				}
			}
			random_scope_state = {
				limit = {
					is_capital = no
				}
				save_scope_as = aboriginal_state
			}		
		}

		if = {
			limit = {
				NOT = {
					exists = c:GRJ
					any_country = {
						NOT = {
							country_has_primary_culture = cu:australian
						}
						is_adjacent_to_country = ROOT
						OR = {
							country_has_primary_culture = cu:aborigine_malay
							country_has_primary_culture = cu:aborigine
						}
					}
				}
			}
			create_country = {
						tag = YLN
						origin = this
						state = scope:aboriginal_state


			}
			c:YLN = {
				save_scope_as = aboriginal_country
			}
		}

		
		if = {
			limit = {
				scope:aboriginal_country = {
					is_country_type = decentralized
				}
			}
			scope:aboriginal_country = {
				set_country_type = unrecognized
			}
		}

		add_modifier = {
			name = dvg_racial_tensions
			months = 50
			is_decaying = yes
		}
	}

	

	option = {
		name = dvg_australie.9.a
		default_option = yes

			limit = {
				exists = c:GRJ
			}

				add_involvement = {
					strategic_region = sr:region_zhourao
					value = 5000
