dvg_australie.4 = {
	type = country_event
	placement = ROOT

	title = dvg_australie.4.t
	desc = dvg_australie.4.d
	flavor = dvg_australie.4.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_gold_prospectors.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		
	}
	immediate = {
		top_overlord = {
			save_scope_as = migration_country_b
			random_primary_culture = {
				save_scope_as = migration_culture_b
			}
			random_subject_or_below = {
				limit = {
					any_scope_state = {
							state_region = {
								is_homeland = cu:javan
							}
					}
				}
				save_scope_as = migration_country_a
				
				cu:javan = {
					save_scope_as = migration_culture_a
				}
			}
		}
		
	}

	option = {
		name = dvg_australie.4.a
		default_option = yes
		
		random_scope_state = {
			create_mass_migration = {
			origin = scope:migration_country_b
			culture = scope:migration_culture_b
			}
		}

		add_modifier = {
			name = sa_migration_cost
			months = 60
			is_decaying = yes
		}
		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_australie.4.b
		trigger = {
			OR = {
				exists = c:BEI
				c:BRG = {
					any_scope_state = {
						any_scope_pop = {
							has_discrimination_trait = heritage_insulindian
						}
					}
				}
			}
		}
		random_scope_state = {
			create_mass_migration = {
			origin = scope:migration_country_a
			culture = scope:migration_culture_a
			}
		}
		

		add_radicals = {
			value = 0.10
			strata = lower
		}
		cu:australian = {
			add_fervor = 10
		}
		
	}
}

dvg_australie.5 = {

	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		has_technology_researched = nationalism
	}

	immediate = {
		random_scope_culture = {
			limit = {
				country_average_cultural_acceptance = {
					target = THIS
					value < 60
				}
			}

		}
		
	}


	
	option = {
		name = dvg_australie.5.a

			value = 0.15

		}

			value = 0.10
			culture = scope:discriminated_culture_scope
		}
	
	}
	option = {
		name = dvg_australie.5.b

		add_radicals = {
			value = 0.15
			culture = cu:australian
		}
		if = {

			}

		}

			value = 0.10

		}

			base = 50
		}
	}
}
