dvg_australie.4 = {
	type = country_event
	placement = ROOT

	title = dvg_australie.4.t
	desc = dvg_australie.4.d
	flavor = dvg_australie.4.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_gold_prospectors.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		
	}
	immediate = {
		c:BRG ?= {
			save_scope_as = migration_country_b
		}
		cu:burgundian = {
			save_scope_as = migration_culture_b
		}
		capital = {
			save_scope_as = migration_target_state
		}
		random_country = {
			limit = {
				country_rank <= rank_value:major_power
				custom_tooltip = {
					text = is_in_indonesia_tt
					is_in_geographic_region = geographic_region_south_east_asia
				}
				average_sol <= root.average_sol
				any_primary_culture = {
					culture_can_have_mass_migration_to_country = ROOT
				}
				any_scope_state = {
					is_incorporated = yes
					custom_tooltip = {
						text = is_in_indonesia_tt
						is_in_geographic_region = geographic_region_south_east_asia
					}
					count >= 3
				}
			}
			save_scope_as = migration_country
			random_scope_state = {
				limit = {
					is_incorporated = yes
					is_in_geographic_region = geographic_region_south_east_asia
					any_scope_culture = {
						has_homeland = PREV
						culture_can_have_mass_migration_to_country = ROOT
					}
				}
				save_scope_as = migration_state
				random_scope_culture = {
					limit = {
						has_homeland = PREV
						culture_can_have_mass_migration_to_country = ROOT
					}
					save_scope_as = migration_culture
				}
			}
		}
		ordered_scope_state = {
			limit = {
				OR = {
						region = sr:region_zhourao
						region = sr:region_oceania
				}
				state_is_eligible_as_mass_migration_target = yes
			}
			order_by = migration_pull
			position = 0
			check_range_bounds = no
			save_scope_as = migration_target_state
		}
		scope:migration_target_state = {
			create_mass_migration = {
				origin = scope:migration_country
				culture = scope:migration_culture
			}
		}
	}

	option = {
		name = dvg_australie.4.a
		default_option = yes
		
		
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = dvg_australie.4.b
		trigger = {
			OR = {
				exists = c:BEI
				c:BRG = {
					any_scope_state = {
						any_scope_pop = {
							has_discrimination_trait = heritage_insulindian
						}
					}
				}
			}
		}

		add_radicals = {
			value = 0.10
			strata = lower
		}
		cu:australian = {
			add_fervor = 10
		}
		
	}
}

dvg_australie.5 = {
	type = country_event
	placement = ROOT

	title = dvg_australie.5.t
	desc = dvg_australie.5.d
	flavor = dvg_australie.5.f

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
		has_technology_researched = nationalism
	}

	immediate = {
		random_scope_culture = {
			limit = {
				country_average_cultural_acceptance = {
					target = THIS
					value < 60
				}
			}
			save_scope_as = discriminated_culture_scope

		}
		
	}


	
	option = {
		name = dvg_australie.5.a
		default_option = yes

		add_loyalists = {
			value = 0.15
			culture = cu:australian
		}

		add_radicals = {
			value = 0.10
			culture = scope:discriminated_culture_scope
		}
	
	}
	option = {
		name = dvg_australie.5.b

		add_radicals = {
			value = 0.15
			culture = cu:australian
		}
		if = {
			limit = {
				is_subject = yes
			}
			add_liberty_desire = 5
		}

		add_loyalists = {
			value = 0.10
			culture = scope:discriminated_culture_scope
		}
		
		ai_chance = {
			base = 50
		}
	}
}
