dvg_afghanistan.5 = {
	type = country_event
	placement = ROOT

	title = dvg_afghanistan.5.t
	desc = dvg_afghanistan.5.d
	flavor = dvg_afghanistan.5.f
	
	event_image = {
		video = "gfx/event_pictures/asia_hinduism_sikhism.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/middleeast/courtroom_upheaval"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 1

	trigger = {
		NOT = {
			has_modifier = decision_cooldown
		}
	}

	immediate = {
		add_modifier = {
			name = decision_cooldown
			months = 12
		}
	}
	
	# Close down gudawaras
	option = {
		name = dvg_afghanistan.5.a
		default_option = yes
		
		add_radicals = {
			value = 0.20
			religion = rel:sikh
		}
		add_loyalists = {
			value = 0.15
			religion = rel:sunni
		}
		add_modifier = {
			name = dvg_modifier_muslim_kitchens
			months = 12
		}
		if = {
			limit = {
				has_journal_entry = je_panjabi_revolt
			}
			change_variable = {
				name = punjab_var
				add = 4
			}
		}

		ai_chance = {
			base = 100
		}	
	}

	option = {
		name = dvg_afghanistan.5.b
		# Restrict Gudawaras

		add_radicals = {
			value = 0.10
			religion = rel:sikh
		}
		add_radicals = {
			value = 0.10
			religion = rel:sunni
		}
		
		if = {
			limit = {
				has_journal_entry = je_panjabi_revolt
			}
			change_variable = {
				name = punjab_var
				add = 2
			}
		}

		ai_chance = {
			base = 100
		}	
	}

	option = {
		name = dvg_afghanistan.5.c
		# Allow Gudawaras

		
		
		add_loyalists = {
			value = 0.25
			religion = rel:sikh
		}
		add_radicals = {
			value = 0.15
			religion = rel:sunni
		}
		random_scope_pop = {
			change_pop_religion = {
				target = rel:sikh
				value = 0.10
			}
		}
		if = {
			limit = {
				has_journal_entry = je_panjabi_revolt
			}
			change_variable = {
				name = punjab_var
				subtract = 3
			}
		}
		ai_chance = {
			base = 100
		}	
	}
}

		}

			}
		}
		add_loyalists = {
			value = 0.20
			religion = rel:sunni
		}
		scope:tax_state = {
			add_loyalists = {
				value = 0.15
			}
			add_modifier = {
				name = tax_modifier_high

			}
		}

			base = 100

				value = 0.15

				name = temporary_tax_cut_proportional

			}
		}
		if = {
			limit = {
				has_journal_entry = je_panjabi_revolt
			}
			change_variable = {
				name = punjab_var
				subtract = 2
			}
		}
		

		ai_chance = {
			base = 100
		}	
	}
	#Side with the Sikhs
	option = {
		name = dvg_afghanistan.6.c		
		
		add_loyalists = {
			value = 0.15
			religion = rel:sikh
		}

			value = 0.25
			religion = rel:sunni
		}

			limit = {
				has_journal_entry = je_panjabi_revolt
			}
			change_variable = {
				name = punjab_var
				subtract = 4
			}
		}

			base = 100
