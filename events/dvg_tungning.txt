# China Reunification
dvg_tungning.1 = {
	type = country_event
	placement = ROOT

	title = dvg_tungning.1.t
	desc = dvg_tungning.1.d
	flavor = dvg_tungning.1.f

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/military_parade"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
	}
	immediate = {
		c:NAN = { save_scope_as = scope_NAN}
		c:FRS = { save_scope_as = scope_FRS}
	}

	option = {
		name = dvg_tungning.1.a
		default_option = yes
		
		add_journal_entry = {
			type = je_tungning_start
		
		}

		create_diplomatic_play = {
			type = dp_unify_china

			war = no

			target_country = c:NAN

			add_war_goal = {
				holder = c:NAN
				type = return_state
				target_state = s:STATE_ZHEJIANG.region_state:FRS
			}
			add_war_goal = {
				holder = c:NAN
				type = return_state
				target_state = s:STATE_FUJIAN.region_state:FRS
			}
			add_war_goal = {
				holder = c:NAN
				type = return_state
				target_state = s:STATE_SHAOZHOU.region_state:FRS
			}

			
		}
		ai_chance = {
			base = 99
		}
	}
	option = {
		name = dvg_tungning.1.b
		default_option = yes
		
		add_journal_entry = {
			type = je_tungning_start
		
		}
		add_radicals = {
			value = 0.3
		}
		ai_chance = {
			base = 01
		}
		
	}
}
