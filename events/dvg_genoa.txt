	type = country_event
	placement = root
	title = genoa.18.t
	desc = genoa.18.d
	flavor = genoa.18.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_before_the_battle.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/before_the_battle"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {
	}   
		
	option = {
		name = genoa.18.a
		default_option = yes
		ai_chance = {
			base = 33
		}  

		#WAR
		custom_tooltip = {
			text = dvg_this_will_lead_to_war_with_corsica
			create_diplomatic_play = {
				name = corsica_independence_war
			
				target_country = c:COR
				
				war = yes
				
				type = dp_annex_war
				
				add_war_goal = {
					holder = c:COR
					type = humiliation
					target_country = c:GEN
				}

				add_war_goal = {
					holder = c:COR
					type = war_reparations
					target_country = c:GEN
				}
			}
		}
	}

	option = {
		name = genoa.18.b
		ai_chance = {
			base = 33
		} 
		add_modifier = { 
			name = dvg_genoa_losses_corsica 
			months = 60
		}   
	}
}

	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	trigger = { #Still says bugged, but I think it works???
		exists = c:COR
		AND = {
			c:COR = {
				has_truce_with = c:GEN 
			}
			c:COR = {
				has_diplomatic_pact = {
					who = c:TUS
					type = alliance
				}
			}
		}
	}
		
	option = {
		name = genoa.20.a
		default_option = yes
		ai_chance = {
			base = 100
		}  
		play_as = c:TUS
		hidden_effect = {
			c:TUS = {
				trigger_event = {
					id = genoa.21 
					days = 30
					popup = yes
				}
			}
		}  
	}	
}

# Etruria annexs Corsica
genoa.21 = { # TUS annex COR event
	type = country_event
	placement = root

	title = genoa.21.t
	desc = genoa.21.d
	flavor = genoa.21.f

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	immediate = {
		c:COR = {
			save_scope_as = annex_corsica
		}
	}
		
	option = {
		name = genoa.21.a
		default_option = yes
		annex = scope:annex_corsica

		create_character = {
			first_name = "Mateo"					
			last_name = "Falcone"
			culture = cu:central_italian
			is_general = yes
			traits = { experienced_defensive_strategist bandit brave romantic} #Might be too OP, decrease 
			birth_date = 1785.10.19
			interest_group = ig_armed_forces
			ideology = ideology_traditionalist
		}

		ai_chance = {
			base = 50
		}
	}

	option = {
		name = genoa.21.b
		default_option = yes
		annex = scope:annex_corsica

		add_loyalists = {
			value = 0.15
			culture=cu:central_italian
		}  

		ai_chance = {
			base = 50
		}
	}	
}
