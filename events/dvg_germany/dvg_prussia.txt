	}

	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_default.dds"
	
	trigger = {
	}
	immediate = {
		
	}
	option = { #new königsberg
		name = dvg_prussia.3.a
		default_option = yes
		
		if = {
			limit = {
				owns_entire_state_region = STATE_EAST_PRUSSIA
			}
			s:STATE_EAST_PRUSSIA.region_state:PRU = {
				add_modifier = {
				name = dvg_rennovated_city
				days = -1
				}
			}
		}
	}
	option = { #new berlin
		name = dvg_prussia.3.b
		trigger = {
			has_global_variable = prussia_to_west
		}
		if = {
			limit = {
				owns_entire_state_region = STATE_BRANDENBURG
			}
			s:STATE_BRANDENBURG.region_state:PRU = {
				add_modifier = {
				name = dvg_rennovated_city
				days = -1
				}
			}
			set_capital = STATE_BRANDENBURG
		}
	}
	option = { #new riga
		name = dvg_prussia.3.c
		trigger = {
			has_global_variable = prussia_to_east
		}
		if = {
			limit = {
				owns_entire_state_region = STATE_LATVIA
			}
			s:STATE_LATVIA.region_state:PRU = {
				add_modifier = {
				name = dvg_rennovated_city
				days = -1
				}
			}
			set_capital = STATE_LATVIA
		}
	}
}

	option = { # yes
		name = dvg_prussia.14.a

		create_treaty = {
			name = treaty_name_scando_prussia
			first_country = c:SCA
			second_country = c:PRU
			is_draft = no
			binding_period = { years = 25  }
			articles_to_create = { 
				{
					article = foreign_investment_rights
					source_country = c:PRU
					target_country = c:SCA
				}
				{
					article = military_access
					source_country = c:PRU
					target_country = c:SCA
				}
			}
		}
		c:SCA = {
			create_diplomatic_pact = {
				country = c:PRU
				type = protectorate
			}
			add_treasury = 400000
			every_scope_state = {
				limit = {
					state_region = {
						is_homeland = cu:balters
						is_not_in_geographic_region = geographic_region_north_germany_old
					}
				}
			set_state_owner = c:PRU
			}
		}
		c:PRU = {
			annex = c:CRL
		}
		
		
	}
	option = { # no
