dvg_turan.1 = {
	type = country_event
	placement = ROOT

	title = dvg_turan.1.t
	desc = dvg_turan.1.d
	flavor = dvg_turan.1.f

	event_image = {
		video = "gfx/event_pictures/asia_buddhism.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	immediate = {
		create_character = {
			historical = yes
			first_name = Sándor
			last_name = Vámbéry 
			noble = yes
			is_agitator = yes
			birth_date = 1809.12.05
			interest_group = ig:ig_intelligentsia
			ideology = ideology_adventurer
			traits = {
				romantic

				basic_diplomat

				basic_entrepreneur

				literary
			}
			on_created = {
				save_scope_as = vambery_scope
			}
		}
	}

	#No to Turan nonsense
	option = {
		name = dvg_turan.1.a
		


		

		ai_chance = {
			base = 1
		}
	}	
	#We will fund this
	option = {
		name = dvg_turan.1.b
		default_option = yes

		add_modifier = {
			name = expedition_taxes_modifier
		}
		random_country = {
			limit = {
				any_primary_culture = {
					has_discrimination_trait = heritage_finnic
				}
				empty_agitator_slots >= 1
			}
			trigger_event = dvg_turan.2
			save_scope_as = turan_event_country
		}
		scope:vambery_scope = {
			transfer_character = scope:turan_event_country
		}
		
		trigger_event = {
				id = dvg_turan.2 days = 75
		}
		ai_chance = {
			base = 100
		}
	}
	
}

dvg_turan.7 = {
	type = country_event
	placement = ROOT

	title = dvg_turan.7.t
	desc = dvg_turan.7.d
	flavor = dvg_turan.7.f

	event_image = {
		video = "gfx/event_pictures/asia_buddhism.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/aristocrats"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"

	duration = 3

	trigger = {
	}
	immediate = {
		random_primary_culture = {
			save_scope_as = primary_culture_scope
		}
	}

	#Promote Pan-Turanism
	option = {
		name = dvg_turan.7.a
		

		if = {
			limit = {
				has_technology_researched = nationalism
			}
			add_technology_progress = {
				technology = pan-nationalism
				progress = 3500
			}
		}
		add_loyalists = {
			value = 0.15
			interest_group = ig:ig_intelligentsia
		}

		ai_chance = {
			base = 1
		}
	}	
	#Ban this nonsense
	option = {
		name = dvg_turan.7.b
		default_option = yes

		add_loyalists = {
			value = 0.15
			culture = scope:primary_culture_scope
		}
		add_radicals = {
			value = 0.15
			interest_group = ig:ig_intelligentsia
		}
		
		ai_chance = {
			base = 100
		}
	}
	#Convert to Gelugpa Buddhism
	option = {
		name = dvg_turan.7.c

		change_pop_religion = {
			target = rel:gelugpa
			value = 0.15
		}
		
		ai_chance = {
			base = 100
		}
	}
	
}
