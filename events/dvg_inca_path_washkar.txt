			limit = {
				exists = c:GCO
				NOT = { exists = c:UCA}
			}
			s:STATE_ECUADOR = {
				set_owner_of_provinces = {
					country = c:GCO
					provinces = { "x6ECA62" }
				}
			}
		}

		if = {
			limit = {
				exists = c:UCA
			}
			s:STATE_ECUADOR = {
				set_owner_of_provinces = {
					country = c:UCA
					provinces = { "x6ECA62" }
				}
			}
		}
	}

	option = {
		name = dvg_inca_path_washkar.9.a

		scope:scope_sapa_inka = {
			kill_character = yes
		}
		scope:scope_amachaq_kosciuzko = {
			add_character_role = politician
			set_as_interest_group_leader = yes
		}


		add_journal_entry = { type = je_amachaq }
		trigger_event = { id = dvg_inca_path_washkar.13 days = 180 } #A Secret Meeting
	}
}
